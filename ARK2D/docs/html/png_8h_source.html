00289 /*
00290  * Y2K compliance in libpng:
00291  * =========================
00292  *
00293  *    February 3, 2011
00294  *
00295  *    Since the PNG Development group is an ad-hoc body, we can't make
00296  *    an official declaration.
00297  *
00298  *    This is your unofficial assurance that libpng from version 0.71 and
00299  *    upward through 1.5.1 are Y2K compliant.  It is my belief that
00300  *    earlier versions were also Y2K compliant.
00301  *
00302  *    Libpng only has three year fields.  One is a 2-byte unsigned integer
00303  *    that will hold years up to 65535.  The other two hold the date in text
00304  *    format, and will hold years up to 9999.
00305  *
00306  *    The integer is
00307  *        "png_uint_16 year" in png_time_struct.
00308  *
00309  *    The strings are
00310  *        "png_charp time_buffer" in png_struct and
00311  *        "near_time_buffer", which is a local character string in png.c.
00312  *
00313  *    There are seven time-related functions:
00314  *        png.c: png_convert_to_rfc_1123() in png.c
00315  *          (formerly png_convert_to_rfc_1152() in error)
00316  *        png_convert_from_struct_tm() in pngwrite.c, called in pngwrite.c
00317  *        png_convert_from_time_t() in pngwrite.c
00318  *        png_get_tIME() in pngget.c
00319  *        png_handle_tIME() in pngrutil.c, called in pngread.c
00320  *        png_set_tIME() in pngset.c
00321  *        png_write_tIME() in pngwutil.c, called in pngwrite.c
00322  *
00323  *    All handle dates properly in a Y2K environment.  The
00324  *    png_convert_from_time_t() function calls gmtime() to convert from system
00325  *    clock time, which returns (year - 1900), which we properly convert to
00326  *    the full 4-digit year.  There is a possibility that applications using
00327  *    libpng are not passing 4-digit years into the png_convert_to_rfc_1123()
00328  *    function, or that they are incorrectly passing only a 2-digit year
00329  *    instead of "year - 1900" into the png_convert_from_struct_tm() function, 
00330  *    but this is not under our control.  The libpng documentation has always
00331  *    stated that it works with 4-digit years, and the APIs have been
00332  *    documented as such.
00333  *
00334  *    The tIME chunk itself is also Y2K compliant.  It uses a 2-byte unsigned
00335  *    integer to hold the year, and can hold years as large as 65535.
00336  *
00337  *    zlib, upon which libpng depends, is also Y2K compliant.  It contains
00338  *    no date-related code.
00339  *
00340  *       Glenn Randers-Pehrson
00341  *       libpng maintainer
00342  *       PNG Development Group
00343  */

02117  * interlaced images within the application.

02121 /* Two macros to return the first row and first column of the original,
02122  * full, image which appears in a given pass.  'pass' is in the range 0
02123  * to 6 and the result is in the range 0 to 7.
02124  */
02125 #define PNG_PASS_START_ROW(pass) (((1U&~(pass))<<(3-((pass)>>1)))&7)
02126 #define PNG_PASS_START_COL(pass) (((1U& (pass))<<(3-(((pass)+1)>>1)))&7)
02127 
02128 /* Two macros to help evaluate the number of rows or columns in each
02129  * pass.  This is expressed as a shift - effectively log2 of the number or
02130  * rows or columns in each 8x8 tile of the original image.
02131  */
02132 #define PNG_PASS_ROW_SHIFT(pass) ((pass)>2?(8-(pass))>>1:3)
02133 #define PNG_PASS_COL_SHIFT(pass) ((pass)>1?(7-(pass))>>1:3)
02134 
02135 /* Hence two macros to determine the number of rows or columns in a given
02136  * pass of an image given its height or width.  In fact these macros may
02137  * return non-zero even though the sub-image is empty, because the other
02138  * dimension may be empty for a small image.
02139  */
02140 #define PNG_PASS_ROWS(height, pass) (((height)+(((1<<PNG_PASS_ROW_SHIFT(pass))\
02141    -1)-PNG_PASS_START_ROW(pass)))>>PNG_PASS_ROW_SHIFT(pass))
02142 #define PNG_PASS_COLS(width, pass) (((width)+(((1<<PNG_PASS_COL_SHIFT(pass))\
02143    -1)-PNG_PASS_START_COL(pass)))>>PNG_PASS_COL_SHIFT(pass))

02145 /* For the progressive reader it is necessary to find the row in the output
02146  * image given a row in an interlaced image, so two more macros:
02147  */
02148 #define PNG_ROW_FROM_PASS_ROW(yIn, pass) \
02149    (((yIn)<<PNG_PASS_ROW_SHIFT(pass))+PNG_PASS_START_ROW(pass))
02150 #define PNG_COL_FROM_PASS_COL(xIn, pass) \
02151    (((xIn)<<PNG_PASS_COL_SHIFT(pass))+PNG_PASS_START_COL(pass))

02154  * or column is in a particular pass.  These use a common utility macro that
02155  * returns a mask for a given pass - the offset 'off' selects the row or
02156  * column version.  The mask has the appropriate bit set for each column in
02157  * the tile.
02158  */

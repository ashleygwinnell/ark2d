01826     {

01860         readPos %= TotalBufferDataSize;
01861         writePos %= TotalBufferDataSize;

02063 static ALint FramesFromBytes(ALint offset, ALenum format, ALint channels)

02068         offset /= 36 * channels;

02070         offset *= 65;
02071     }
02072     else if(format==AL_FORMAT_MONO_MULAW || format==AL_FORMAT_STEREO_MULAW ||

02074             format==AL_FORMAT_61CHN_MULAW || format==AL_FORMAT_71CHN_MULAW)
02075     {
02076         /* muLaw has 1 byte per sample */
02077         offset /= 1 * channels;
02078     }
02079     else if(format == AL_FORMAT_REAR_MULAW)
02080     {
02081         /* Rear is 2 channels */
02082         offset /= 1 * 2;
02083     }
02084     else if(format == AL_FORMAT_REAR8)

02086     else if(format == AL_FORMAT_REAR16)

02089         offset /= 4 * 2;
02090     else

02092         ALuint bytes = aluBytesFromFormat(format);

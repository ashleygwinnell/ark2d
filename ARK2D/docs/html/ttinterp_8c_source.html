00062   /*                                                                       */
00063   /*    The current execution context is a static variable, which fields   */
00064   /*    are accessed directly by the interpreter during execution.  The    */
00065   /*    context is named `cur'.                                            */

00070   /*                                                                       */

02690   /*                                                                       */
02691   /* <Function>                                                            */
02692   /*    Normalize                                                          */
02693   /*                                                                       */
02694   /* <Description>                                                         */
02695   /*    Norms a vector.                                                    */
02696   /*                                                                       */
02697   /* <Input>                                                               */
02698   /*    Vx :: The horizontal input vector coordinate.                      */
02699   /*    Vy :: The vertical input vector coordinate.                        */
02700   /*                                                                       */
02701   /* <Output>                                                              */
02702   /*    R  :: The normed unit vector.                                      */
02703   /*                                                                       */
02704   /* <Return>                                                              */
02705   /*    Returns FAILURE if a vector parameter is zero.                     */
02706   /*                                                                       */
02707   /* <Note>                                                                */
02708   /*    In case Vx and Vy are both zero, Normalize() returns SUCCESS, and  */
02709   /*    R is undefined.                                                    */
02710   /*                                                                       */
02711 
02712 
02713   static FT_Bool
02714   Normalize( EXEC_OP_ FT_F26Dot6      Vx,
02715                       FT_F26Dot6      Vy,
02716                       FT_UnitVector*  R )
02717   {
02718     FT_F26Dot6  W;
02719     FT_Bool     S1, S2;
02720 
02721     FT_UNUSED_EXEC;
02722 
02723 
02724     if ( FT_ABS( Vx ) < 0x10000L && FT_ABS( Vy ) < 0x10000L )
02725     {
02726       Vx *= 0x100;
02727       Vy *= 0x100;
02728 
02729       W = TT_VecLen( Vx, Vy );
02730 
02731       if ( W == 0 )
02732       {
02733         /* XXX: UNDOCUMENTED! It seems that it is possible to try   */
02734         /*      to normalize the vector (0,0).  Return immediately. */
02735         return SUCCESS;
02736       }
02737 
02738       R->x = (FT_F2Dot14)FT_MulDiv( Vx, 0x4000L, W );
02739       R->y = (FT_F2Dot14)FT_MulDiv( Vy, 0x4000L, W );
02740 
02741       return SUCCESS;
02742     }
02743 
02744     W = TT_VecLen( Vx, Vy );
02745 
02746     Vx = FT_MulDiv( Vx, 0x4000L, W );
02747     Vy = FT_MulDiv( Vy, 0x4000L, W );
02748 
02749     W = Vx * Vx + Vy * Vy;
02750 
02751     /* Now, we want that Sqrt( W ) = 0x4000 */
02752     /* Or 0x10000000 <= W < 0x10004000        */
02753 
02754     if ( Vx < 0 )
02755     {
02756       Vx = -Vx;
02757       S1 = TRUE;
02758     }
02759     else
02760       S1 = FALSE;
02761 
02762     if ( Vy < 0 )
02763     {
02764       Vy = -Vy;
02765       S2 = TRUE;
02766     }
02767     else
02768       S2 = FALSE;
02769 
02770     while ( W < 0x10000000L )
02771     {
02772       /* We need to increase W by a minimal amount */
02773       if ( Vx < Vy )
02774         Vx++;
02775       else
02776         Vy++;
02777 
02778       W = Vx * Vx + Vy * Vy;
02779     }
02780 
02781     while ( W >= 0x10004000L )
02782     {
02783       /* We need to decrease W by a minimal amount */
02784       if ( Vx < Vy )
02785         Vx--;
02786       else
02787         Vy--;
02788 
02789       W = Vx * Vx + Vy * Vy;
02790     }
02791 
02792     /* Note that in various cases, we can only  */
02793     /* compute a Sqrt(W) of 0x3FFF, eg. Vx = Vy */
02794 
02795     if ( S1 )
02796       Vx = -Vx;
02797 
02798     if ( S2 )
02799       Vy = -Vy;
02800 
02801     R->x = (FT_F2Dot14)Vx;   /* Type conversion */
02802     R->y = (FT_F2Dot14)Vy;   /* Type conversion */
02803 
02804     return SUCCESS;
02805   }

04548   /* Opcode range: 0x2C                                                    */
04549   /* Stack:        uint32 -->                                              */
04550   /*                                                                       */
04551   static void
04552   Ins_FDEF( INS_ARG )
04553   {
04554     FT_ULong       n;
04555     TT_DefRecord*  rec;
04556     TT_DefRecord*  limit;
04557 
04558 
04559     /* some font programs are broken enough to redefine functions! */
04560     /* We will then parse the current table.                       */
04561 
04562     rec   = CUR.FDefs;
04563     limit = rec + CUR.numFDefs;
04564     n     = args[0];
04565 
04566     for ( ; rec < limit; rec++ )
04567     {
04568       if ( rec->opc == n )
04569         break;
04570     }
04571 
04572     if ( rec == limit )
04573     {
04574       /* check that there is enough room for new functions */
04575       if ( CUR.numFDefs >= CUR.maxFDefs )
04576       {
04577         CUR.error = TT_Err_Too_Many_Function_Defs;
04578         return;
04579       }
04580       CUR.numFDefs++;
04581     }
04582 
04583     /* Although FDEF takes unsigned 32-bit integer,  */
04584     /* func # must be within unsigned 16-bit integer */
04585     if ( n > 0xFFFFU )
04586     {
04587       CUR.error = TT_Err_Too_Many_Function_Defs;
04588       return;
04589     }
04590 
04591     rec->range  = CUR.curRange;
04592     rec->opc    = (FT_UInt16)n;
04593     rec->start  = CUR.IP + 1;
04594     rec->active = TRUE;
04595 
04596     if ( n > CUR.maxFunc )
04597       CUR.maxFunc = (FT_UInt16)n;
04598 
04599     /* Now skip the whole function definition. */
04600     /* We don't allow nested IDEFS & FDEFs.    */
04601 
04602     while ( SKIP_Code() == SUCCESS )
04603     {
04604       switch ( CUR.opcode )
04605       {
04606       case 0x89:    /* IDEF */
04607       case 0x2C:    /* FDEF */
04608         CUR.error = TT_Err_Nested_DEFS;
04609         return;
04610 
04611       case 0x2D:   /* ENDF */
04612         return;
04613       }
04614     }
04615   }

04628 
04629     FT_UNUSED_ARG;

04631

04667   /*                                                                       */
04668   /* CALL[]:       CALL function                                           */
04669   /* Opcode range: 0x2B                                                    */
04670   /* Stack:        uint32? -->                                             */
04671   /*                                                                       */
04672   static void
04673   Ins_CALL( INS_ARG )
04674   {
04675     FT_ULong       F;
04676     TT_CallRec*    pCrec;
04677     TT_DefRecord*  def;
04678 
04679 
04680     /* first of all, check the index */
04681 
04682     F = args[0];
04683     if ( BOUNDSL( F, CUR.maxFunc + 1 ) )
04684       goto Fail;
04685 
04686     /* Except for some old Apple fonts, all functions in a TrueType */
04687     /* font are defined in increasing order, starting from 0.  This */
04688     /* means that we normally have                                  */
04689     /*                                                              */
04690     /*    CUR.maxFunc+1 == CUR.numFDefs                             */
04691     /*    CUR.FDefs[n].opc == n for n in 0..CUR.maxFunc             */
04692     /*                                                              */
04693     /* If this isn't true, we need to look up the function table.   */
04694 
04695     def = CUR.FDefs + F;
04696     if ( CUR.maxFunc + 1 != CUR.numFDefs || def->opc != F )
04697     {
04698       /* look up the FDefs table */
04699       TT_DefRecord*  limit;
04700 
04701 
04702       def   = CUR.FDefs;
04703       limit = def + CUR.numFDefs;
04704 
04705       while ( def < limit && def->opc != F )
04706         def++;
04707 
04708       if ( def == limit )
04709         goto Fail;
04710     }
04711 
04712     /* check that the function is active */
04713     if ( !def->active )
04714       goto Fail;
04715 
04716     /* check the call stack */
04717     if ( CUR.callTop >= CUR.callSize )
04718     {
04719       CUR.error = TT_Err_Stack_Overflow;
04720       return;
04721     }
04722 
04723     pCrec = CUR.callStack + CUR.callTop;
04724 
04725     pCrec->Caller_Range = CUR.curRange;
04726     pCrec->Caller_IP    = CUR.IP + 1;
04727     pCrec->Cur_Count    = 1;
04728     pCrec->Cur_Restart  = def->start;
04729 
04730     CUR.callTop++;
04731 
04732     INS_Goto_CodeRange( def->range,
04733                         def->start );
04734 
04735     CUR.step_ins = FALSE;
04736     return;
04737 
04738   Fail:
04739     CUR.error = TT_Err_Invalid_Reference;
04740   }

06614   static void
06615   _iup_worker_interpolate( IUP_Worker  worker,
06616                            FT_UInt     p1,
06617                            FT_UInt     p2,
06618                            FT_UInt     ref1,
06619                            FT_UInt     ref2 )
06620   {
06621     FT_UInt     i;
06622     FT_F26Dot6  orus1, orus2, org1, org2, delta1, delta2;
06623 
06624 
06625     if ( p1 > p2 )
06626       return;
06627 
06628     if ( BOUNDS( ref1, worker->max_points ) ||
06629          BOUNDS( ref2, worker->max_points ) )
06630       return;
06631 
06632     orus1 = worker->orus[ref1].x;
06633     orus2 = worker->orus[ref2].x;
06634 
06635     if ( orus1 > orus2 )
06636     {
06637       FT_F26Dot6  tmp_o;

06639 
06640 
06641       tmp_o = orus1;
06642       orus1 = orus2;
06643       orus2 = tmp_o;
06644 
06645       tmp_r = ref1;
06646       ref1  = ref2;
06647       ref2  = tmp_r;
06648     }
06649 
06650     org1   = worker->orgs[ref1].x;
06651     org2   = worker->orgs[ref2].x;
06652     delta1 = worker->curs[ref1].x - org1;
06653     delta2 = worker->curs[ref2].x - org2;
06654 
06655     if ( orus1 == orus2 )
06656     {
06657       /* simple shift of untouched points */
06658       for ( i = p1; i <= p2; i++ )
06659       {
06660         FT_F26Dot6  x = worker->orgs[i].x;
06661 
06662 
06663         if ( x <= org1 )
06664           x += delta1;
06665         else
06666           x += delta2;
06667 
06668         worker->curs[i].x = x;
06669       }
06670     }
06671     else
06672     {
06673       FT_Fixed  scale       = 0;
06674       FT_Bool   scale_valid = 0;
06675 
06676 
06677       /* interpolation */
06678       for ( i = p1; i <= p2; i++ )
06679       {
06680         FT_F26Dot6  x = worker->orgs[i].x;
06681 
06682 
06683         if ( x <= org1 )
06684           x += delta1;
06685 
06686         else if ( x >= org2 )
06687           x += delta2;
06688 
06689         else
06690         {
06691           if ( !scale_valid )
06692           {
06693             scale_valid = 1;
06694             scale       = TT_MULDIV( org2 + delta2 - ( org1 + delta1 ),
06695                                      0x10000L, orus2 - orus1 );
06696           }
06697 
06698           x = ( org1 + delta1 ) +
06699               TT_MULFIX( worker->orus[i].x - orus1, scale );
06700         }
06701         worker->curs[i].x = x;
06702       }
06703     }

06801                                      first_touched - 1,

07368   /*                                                                       */
07369   /*  This function executes a run of opcodes.  It will exit in the        */
07370   /*  following cases:                                                     */
07371   /*                                                                       */
07372   /*  - Errors (in which case it returns FALSE).                           */
07373   /*                                                                       */
07374   /*  - Reaching the end of the main code range (returns TRUE).            */
07375   /*    Reaching the end of a code range within a function call is an      */
07376   /*    error.                                                             */
07377   /*                                                                       */
07378   /*  - After executing one single opcode, if the flag `Instruction_Trap'  */
07379   /*    is set to TRUE (returns TRUE).                                     */
07380   /*                                                                       */
07381   /*  On exit with TRUE, test IP < CodeSize to know whether it comes from  */
07382   /*  an instruction trap or a normal termination.                         */
07383   /*                                                                       */
07384   /*                                                                       */
07385   /*  Note: The documented DEBUG opcode pops a value from the stack.  This */
07386   /*        behaviour is unsupported; here a DEBUG opcode is always an     */
07387   /*        error.                                                         */
07388   /*                                                                       */
07389   /*                                                                       */
07390   /* THIS IS THE INTERPRETER'S MAIN LOOP.                                  */
07391   /*                                                                       */
07392   /*  Instructions appear in the specification's order.                    */

07399   FT_EXPORT_DEF( FT_Error )
07400   TT_RunIns( TT_ExecContext  exc )
07401   {
07402     FT_Long  ins_counter = 0;  /* executed instructions counter */
07403 
07404 
07405 #ifdef TT_CONFIG_OPTION_STATIC_RASTER
07406     cur = *exc;
07407 #endif
07408 
07409     /* set CVT functions */
07410     CUR.tt_metrics.ratio = 0;
07411     if ( CUR.metrics.x_ppem != CUR.metrics.y_ppem )
07412     {
07413       /* non-square pixels, use the stretched routines */
07414       CUR.func_read_cvt  = Read_CVT_Stretched;
07415       CUR.func_write_cvt = Write_CVT_Stretched;
07416       CUR.func_move_cvt  = Move_CVT_Stretched;
07417     }
07418     else
07419     {
07420       /* square pixels, use normal routines */
07421       CUR.func_read_cvt  = Read_CVT;
07422       CUR.func_write_cvt = Write_CVT;
07423       CUR.func_move_cvt  = Move_CVT;
07424     }
07425 
07426     COMPUTE_Funcs();
07427     COMPUTE_Round( (FT_Byte)exc->GS.round_state );
07428 
07429     do
07430     {
07431       CUR.opcode = CUR.code[CUR.IP];
07432 
07433       FT_TRACE7(( "  " ));
07434       FT_TRACE7(( opcode_name[CUR.opcode] ));
07435       FT_TRACE7(( "\n" ));
07436 
07437       if ( ( CUR.length = opcode_length[CUR.opcode] ) < 0 )
07438       {
07439         if ( CUR.IP + 1 > CUR.codeSize )
07440           goto LErrorCodeOverflow_;
07441 
07442         CUR.length = 2 - CUR.length * CUR.code[CUR.IP + 1];
07443       }
07444 
07445       if ( CUR.IP + CUR.length > CUR.codeSize )
07446         goto LErrorCodeOverflow_;
07447 
07448       /* First, let's check for empty stack and overflow */
07449       CUR.args = CUR.top - ( Pop_Push_Count[CUR.opcode] >> 4 );
07450 
07451       /* `args' is the top of the stack once arguments have been popped. */
07452       /* One can also interpret it as the index of the last argument.    */
07453       if ( CUR.args < 0 )
07454       {
07455         CUR.error = TT_Err_Too_Few_Arguments;
07456         goto LErrorLabel_;
07457       }
07458 
07459       CUR.new_top = CUR.args + ( Pop_Push_Count[CUR.opcode] & 15 );
07460 
07461       /* `new_top' is the new top of the stack, after the instruction's */

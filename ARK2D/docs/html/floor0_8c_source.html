00104 /* initialize Bark scale and normalization lookups.  We could do this
00105    with static tables, but Vorbis allows a number of possible
00106    combinations, so it's best to do it computationally.
00107 
00108    The below is authoritative in terms of defining scale mapping.
00109    Note that the scale depends on the sampling rate as well as the
00110    linear block and mapping sizes */

00112 static void floor0_map_lazy_init(vorbis_block      *vb,
00113                                  vorbis_info_floor *infoX,
00114                                  vorbis_look_floor0 *look){
00115   if(!look->linearmap[vb->W]){
00116     vorbis_dsp_state   *vd=vb->vd;
00117     vorbis_info        *vi=vd->vi;
00118     codec_setup_info   *ci=vi->codec_setup;
00119     vorbis_info_floor0 *info=(vorbis_info_floor0 *)infoX;
00120     int W=vb->W;
00121     int n=ci->blocksizes[W]/2,j;
00122 
00123     /* we choose a scaling constant so that:
00124        floor(bark(rate/2-1)*C)=mapped-1
00125      floor(bark(rate/2)*C)=mapped */
00126     float scale=look->ln/toBARK(info->rate/2.f);
00127 
00128     /* the mapping from a linear scale to a smaller bark scale is
00129        straightforward.  We do *not* make sure that the linear mapping
00130        does not skip bark-scale bins; the decoder simply skips them and
00131        the encoder may do what it wishes in filling them.  They're
00132        necessary in some mapping combinations to keep the scale spacing
00133        accurate */
00134     look->linearmap[W]=_ogg_malloc((n+1)*sizeof(**look->linearmap));
00135     for(j=0;j<n;j++){
00136       int val=floor( toBARK((info->rate/2.f)/n*j)
00137                      *scale); /* bark numbers represent band edges */
00138       if(val>=look->ln)val=look->ln-1; /* guard against the approximation */
00139       look->linearmap[W][j]=val;
00140     }
00141     look->linearmap[W][j]=-1;
00142     look->n[W]=n;
00143   }
00144 }

00974          goto PNG_FP_End; ** Because SAW_DOT is always set */
00975 
00976       case PNG_FP_FRACTION + PNG_FP_DIGIT:

00979 
00980       case PNG_FP_FRACTION + PNG_FP_E:

01009          goto PNG_FP_End; */

01080 /* Function to format a floating point value in ASCII with a given
01081  * precision.
01082  */
01083 void /* PRIVATE */
01084 png_ascii_from_fp(png_structp png_ptr, png_charp ascii, png_size_t size,
01085     double fp, unsigned int precision)
01086 {
01087    /* We use standard functions from math.h, but not printf because
01088     * that would require stdio.  The caller must supply a buffer of
01089     * sufficient size or we will png_error.  The tests on size and
01090     * the space in ascii[] consumed are indicated below.
01091     */
01092    if (precision < 1)
01093       precision = DBL_DIG;
01094 
01095    /* Enforce the limit of the implementation precision too. */
01096    if (precision > DBL_DIG+1)
01097       precision = DBL_DIG+1;
01098 
01099    /* Basic sanity checks */
01100    if (size >= precision+5) /* See the requirements below. */
01101    {
01102       if (fp < 0)
01103       {
01104          fp = -fp;
01105          *ascii++ = 45; /* '-'  PLUS 1 TOTAL 1*/
01106          --size;
01107       }
01108 
01109       if (fp >= DBL_MIN && fp <= DBL_MAX)
01110       {
01111          int exp_b10;       /* A base 10 exponent */
01112          double base;   /* 10^exp_b10 */
01113 
01114          /* First extract a base 10 exponent of the number,
01115           * the calculation below rounds down when converting
01116           * from base 2 to base 10 (multiply by log10(2) -
01117           * 0.3010, but 77/256 is 0.3008, so exp_b10 needs to
01118           * be increased.  Note that the arithmetic shift
01119           * performs a floor() unlike C arithmetic - using a
01120           * C multiply would break the following for negative
01121           * exponents.
01122           */
01123          (void)frexp(fp, &exp_b10); /* exponent to base 2 */
01124 
01125          exp_b10 = (exp_b10 * 77) >> 8; /* <= exponent to base 10 */
01126 
01127          /* Avoid underflow here. */
01128          base = png_pow10(exp_b10); /* May underflow */
01129 
01130          while (base < DBL_MIN || base < fp)
01131          {
01132             /* And this may overflow. */
01133             double test = png_pow10(exp_b10+1);
01134 
01135             if (test <= DBL_MAX)
01136                ++exp_b10, base = test;
01137 
01138             else
01139                break;
01140          }
01141 
01142          /* Normalize fp and correct exp_b10, after this fp is in the
01143           * range [.1,1) and exp_b10 is both the exponent and the digit
01144           * *before* which the decimal point should be inserted
01145           * (starting with 0 for the first digit).  Note that this
01146           * works even if 10^exp_b10 is out of range because of the
01147           * test on DBL_MAX above.
01148           */
01149          fp /= base;
01150          while (fp >= 1) fp /= 10, ++exp_b10;
01151

01153           * less than .1, this is ok because the code below can
01154           * handle the leading zeros this generates, so no attempt
01155           * is made to correct that here.
01156           */

01163              * the number compared to using E-n.

01166             {

01169             }

01172

01174              * inserting a '.' before a digit if the exponent is 0.

01176             clead = czero; /* Count of leading zeros */
01177             cdigits = 0;   /* Count of digits in list. */
01178 
01179             do
01180             {

01182 
01183                fp *= 10;

01185                 * the separation is done in one step.  At the end
01186                 * of the loop don't break the number into parts so
01187                 * that the final digit is rounded.
01188                 */
01189                if (cdigits+czero-clead+1 < (int)precision)

01473 /* This API takes signed arguments and rounds the result to the nearest
01474  * integer (or, for a fixed point number - the standard argument - to
01475  * the nearest .00001).  Overflow and divide by zero are signalled in
01476  * the result, a boolean - true on success, false on overflow.

01478 int
01479 png_muldiv(png_fixed_point_p res, png_fixed_point a, png_int_32 times,
01480     png_int_32 divisor)
01481 {
01482    /* Return a * times / divisor, rounded. */
01483    if (divisor != 0)
01484    {
01485       if (a == 0 || times == 0)
01486       {
01487          *res = 0;
01488          return 1;
01489       }
01490       else
01491       {
01492 #ifdef PNG_FLOATING_ARITHMETIC_SUPPORTED
01493          double r = a;
01494          r *= times;
01495          r /= divisor;
01496          r = floor(r+.5);
01497 
01498          /* A png_fixed_point is a 32 bit integer. */
01499          if (r <= 2147483647. && r >= -2147483648.)
01500          {
01501             *res = (png_fixed_point)r;
01502             return 1;

01523

01525           * have 31 bits each, however the result may be 32 bits.
01526           */
01527          s16 = (A >> 16) * (T & 0xffff) +
01528                            (A & 0xffff) * (T >> 16);
01529          /* Can't overflow because the a*times bit is only 30
01530           * bits at most.
01531           */
01532          s32 = (A >> 16) * (T >> 16) + (s16 >> 16);
01533          s00 = (A & 0xffff) * (T & 0xffff);
01534 
01535          s16 = (s16 & 0xffff) << 16;
01536          s00 += s16;
01537 
01538          if (s00 < s16)
01539             ++s32; /* carry */
01540 
01541          if (s32 < D) /* else overflow */
01542          {
01543             /* s32.s00 is now the 64 bit product, do a standard
01544              * division, we know that s32 < D, so the maximum
01545              * required shift is 31.
01546              */
01547             int bitshift = 32;
01548             png_fixed_point result = 0; /* NOTE: signed */
01549 
01550             while (--bitshift >= 0)
01551             {
01552                png_uint_32 d32, d00;
01553 
01554                if (bitshift > 0)
01555                   d32 = D >> (32-bitshift), d00 = D << bitshift;
01556 
01557                else
01558                   d32 = 0, d00 = D;
01559 
01560                if (s32 > d32)
01561                {
01562                   if (s00 < d00) --s32; /* carry */
01563                   s32 -= d32, s00 -= d00, result += 1<<bitshift;
01564                }
01565 
01566                else
01567                   if (s32 == d32 && s00 >= d00)
01568                      s32 = 0, s00 -= d00, result += 1<<bitshift;
01569             }
01570 
01571             /* Handle the rounding. */
01572             if (s00 >= (D >> 1))
01573                ++result;
01574 
01575             if (negative)
01576                result = -result;
01577 
01578             /* Check for overflow. */
01579             if ((negative && result <= 0) || (!negative && result >= 0))
01580             {
01581                *res = result;
01582                return 1;
01583             }
01584          }
01585 #endif
01586       }

01793 {

01798     * input), return 7.99998 for the overflow (log 0) case - so the result is
01799     * always at most 19 bits.

01806

01809 
01810    if ((x & 0x80) == 0)
01811       lg2 += 1, x <<= 1;

01814    return (png_int_32)((lg2 << 16) + ((png_8bit_l2[x-128]+32768)>>16));

01817 /* The above gives exact (to 16 binary places) log2 values for 8 bit images,
01818  * for 16 bit images we use the most significant 8 bits of the 16 bit value to
01819  * get an approximation then multiply the approximation by a correction factor
01820  * determined by the remaining up to 8 bits.  This requires an additional step
01821  * in the 16 bit case.
01822  *
01823  * We want log2(value/65535), we have log2(v'/255), where:
01824  *
01825  *    value = v' * 256 + v''
01826  *          = v' * f
01827  *
01828  * So f is value/v', which is equal to (256+v''/v') since v' is in the range 128
01829  * to 255 and v'' is in the range 0 to 255 f will be in the range 256 to less
01830  * than 258.  The final factor also needs to correct for the fact that our 8 bit
01831  * value is scaled by 255, whereas the 16 bit values must be scaled by 65535.
01832  *
01833  * This gives a final formula using a calculated value 'x' which is value/v' and
01834  * scaling by 65536 to match the above table:
01835  *
01836  *   log2(x/257) * 65536
01837  *
01838  * Since these numbers are so close to '1' we can use simple linear
01839  * interpolation between the two end values 256/257 (result -368.61) and 258/257
01840  * (result 367.179).  The values used below are scaled by a further 64 to give
01841  * 16 bit precision in the interpolation:
01842  *
01843  * Start (256): -23591
01844  * Zero  (257):      0
01845  * End   (258):  23499
01846  */
01847 static png_int_32
01848 png_log16bit(png_uint_32 x)
01849 {
01850    unsigned int lg2 = 0;
01851 
01852    /* As above, but now the input has 16 bits. */
01853    if ((x &= 0xffff) == 0)
01854       return 0xffffffff;
01855 
01856    if ((x & 0xff00) == 0)
01857       lg2  = 8, x <<= 8;
01858 
01859    if ((x & 0xf000) == 0)
01860       lg2 += 4, x <<= 4;
01861 
01862    if ((x & 0xc000) == 0)
01863       lg2 += 2, x <<= 2;
01864 
01865    if ((x & 0x8000) == 0)
01866       lg2 += 1, x <<= 1;
01867 
01868    /* Calculate the base logarithm from the top 8 bits as a 28 bit fractional
01869     * value.
01870     */
01871    lg2 <<= 28;
01872    lg2 += (png_8bit_l2[(x>>8)-128]+8) >> 4;
01873 
01874    /* Now we need to interpolate the factor, this requires a division by the top
01875     * 8 bits.  Do this with maximum precision.
01876     */
01877    x = ((x << 16) + (x >> 9)) / (x >> 8);
01878 
01879    /* Since we divided by the top 8 bits of 'x' there will be a '1' at 1<<24,
01880     * the value at 1<<16 (ignoring this) will be 0 or 1; this gives us exactly
01881     * 16 bits to interpolate to get the low bits of the result.  Round the
01882     * answer.  Note that the end point values are scaled by 64 to retain overall
01883     * precision and that 'lg2' is current scaled by an extra 12 bits, so adjust
01884     * the overall scaling by 6-12.  Round at every step.
01885     */
01886    x -= 1U << 24;
01887 
01888    if (x <= 65536U) /* <= '257' */
01889       lg2 += ((23591U * (65536U-x)) + (1U << (16+6-12-1))) >> (16+6-12);
01890 
01891    else
01892       lg2 -= ((23499U * (x-65536U)) + (1U << (16+6-12-1))) >> (16+6-12);
01893 
01894    /* Safe, because the result can't have more than 20 bits: */
01895    return (png_int_32)((lg2 + 2048) >> 12);

02164 png_build_16to8_table(png_structp png_ptr, png_uint_16pp *ptable,
02165    PNG_CONST unsigned int shift, PNG_CONST png_fixed_point gamma_val)
02166 {
02167    PNG_CONST unsigned int num = 1U << (8U - shift);
02168    PNG_CONST unsigned int max = (1U << (16U - shift))-1U;
02169    unsigned int i;
02170    png_uint_32 last;
02171 
02172    png_uint_16pp table = *ptable =
02173        (png_uint_16pp)png_calloc(png_ptr, num * png_sizeof(png_uint_16p));
02174 
02175    /* 'num' is the number of tables and also the number of low bits of low
02176     * bits of the input 16 bit value used to select a table.  Each table is
02177     * itself index by the high 8 bits of the value.
02178     */
02179    for (i = 0; i < num; i++)
02180       table[i] = (png_uint_16p)png_malloc(png_ptr,
02181           256 * png_sizeof(png_uint_16));

02183    /* 'gamma_val' is set to the reciprocal of the value calculated above, so
02184     * pow(out,g) is an *input* value.  'last' is the last input value set.
02185     *
02186     * In the loop 'i' is used to find output values.  Since the output is 8
02187     * bit there are only 256 possible values.  The tables are set up to
02188     * select the closest possible output value for each input by finding
02189     * the input value at the boundary between each pair of output values
02190     * and filling the table up to that boundary with the lower output
02191     * value.
02192     *
02193     * The boundary values are 0.5,1.5..253.5,254.5.  Since these are 9 bit
02194     * values the code below uses a 16 bit value in i; the values start at
02195     * 128.5 (for 0.5) and step by 257, for a total of 254 values (the last
02196     * entries are filled with 255).  Start i at 128 and fill all 'last'
02197     * table entries <= 'max'
02198     */
02199    last = 0;
02200    for (i = 0; i < 255; ++i) /* 8 bit output value */
02201    {
02202       /* Find the corresponding maximum input value */
02203       png_uint_16 out = (png_uint_16)(i * 257U); /* 16 bit output value */
02204 
02205       /* Find the boundary value in 16 bits: */
02206       png_uint_32 bound = png_gamma_16bit_correct(out+128U, gamma_val);
02207 
02208       /* Adjust (round) to (16-shift) bits: */
02209       bound = (bound * max + 32768U)/65535U + 1U;
02210 
02211       while (last < bound)
02212       {
02213          table[last & (0xffU >> shift)][last >> (8U - shift)] = out;
02214          last++;
02215       }
02216    }
02217 
02218    /* And fill in the final entries. */
02219    while (last < (num << 8))
02220    {
02221       table[last & (0xff >> shift)][last >> (8U - shift)] = 65535U;
02222       last++;
02223    }

00930 /* Overload/Side effect: On input, the *q vector holds either the
00931    quantized energy (for elements with the flag set) or the absolute
00932    values of the *r vector (for elements with flag unset).  On output,
00933    *q holds the quantized energy for all elements */
00934 static float noise_normalize(vorbis_look_psy *p, int limit, float *r, float *q, float *f, int *flags, float acc, int i, int n, int *out){
00935 
00936   vorbis_info_psy *vi=p->vi;
00937   float **sort = alloca(n*sizeof(*sort));
00938   int j,count=0;
00939   int start = (vi->normal_p ? vi->normal_start-i : n);
00940   if(start>n)start=n;
00941 
00942   /* force classic behavior where only energy in the current band is considered */
00943   acc=0.f;
00944 
00945   /* still responsible for populating *out where noise norm not in
00946      effect.  There's no need to [re]populate *q in these areas */
00947   for(j=0;j<start;j++){
00948     if(!flags || !flags[j]){ /* lossless coupling already quantized.
00949                                 Don't touch; requantizing based on
00950                                 energy would be incorrect. */
00951       float ve = q[j]/f[j];
00952       if(r[j]<0)
00953         out[j] = -rint(sqrt(ve));
00954       else
00955         out[j] = rint(sqrt(ve));
00956     }
00957   }
00958 
00959   /* sort magnitudes for noise norm portion of partition */
00960   for(;j<n;j++){
00961     if(!flags || !flags[j]){ /* can't noise norm elements that have
00962                                 already been loslessly coupled; we can
00963                                 only account for their energy error */
00964       float ve = q[j]/f[j];
00965       /* Despite all the new, more capable coupling code, for now we
00966          implement noise norm as it has been up to this point. Only
00967          consider promotions to unit magnitude from 0.  In addition
00968          the only energy error counted is quantizations to zero. */
00969       /* also-- the original point code only applied noise norm at > pointlimit */
00970       if(ve<.25f && (!flags || j>=limit-i)){
00971         acc += ve;
00972         sort[count++]=q+j; /* q is fabs(r) for unflagged element */
00973       }else{
00974         /* For now: no acc adjustment for nonzero quantization.  populate *out and q as this value is final. */
00975         if(r[j]<0)
00976           out[j] = -rint(sqrt(ve));
00977         else
00978           out[j] = rint(sqrt(ve));
00979         q[j] = out[j]*out[j]*f[j];
00980       }
00981     }/* else{
00982         again, no energy adjustment for error in nonzero quant-- for now
00983         }*/
00984   }
00985 
00986   if(count){
00987     /* noise norm to do */
00988     qsort(sort,count,sizeof(*sort),apsort);
00989     for(j=0;j<count;j++){
00990       int k=sort[j]-q;
00991       if(acc>=vi->normal_thresh){
00992         out[k]=unitnorm(r[k]);
00993         acc-=1.f;
00994         q[k]=f[k];
00995       }else{
00996         out[k]=0;
00997         q[k]=0.f;
00998       }
00999     }
01000   }
01001 
01002   return acc;
01003 }

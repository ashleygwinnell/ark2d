00025 /* Note that the lpc-lsp conversion finds the roots of polynomial with
00026    an iterative root polisher (CACM algorithm 283).  It *is* possible
00027    to confuse this algorithm into not converging; that should only
00028    happen with absurdly closely spaced roots (very sharp peaks in the
00029    LPC f response) which in turn should be impossible in our use of
00030    the code.  If this *does* happen anyway, it's a bug in the floor
00031    finder; find the cause of the confusion (probably a single bin
00032    spike or accidental near-float-limit resolution problems) and
00033    correct it. */

00045    (float), a lookup based float implementation, and an integer

00302 /* Newton-Raphson-Maehly actually functioned as a decent root finder,
00303    but there are root sets for which it gets into limit cycles
00304    (exacerbated by zero suppression) and fails.  We can't afford to
00305    fail, even if the failure is 1 in 100,000,000, so we now use
00306    Laguerre and later polish with Newton-Raphson (which can then
00307    afford to fail) */

00310 static int Laguerre_With_Deflation(float *a,int ord,float *r){
00311   int i,m;
00312   double lastdelta=0.f;
00313   double *defl=alloca(sizeof(*defl)*(ord+1));
00314   for(i=0;i<=ord;i++)defl[i]=a[i];
00315 
00316   for(m=ord;m>0;m--){
00317     double new=0.f,delta;
00318 
00319     /* iterate a root */
00320     while(1){
00321       double p=defl[m],pp=0.f,ppp=0.f,denom;
00322 
00323       /* eval the polynomial and its first two derivatives */
00324       for(i=m;i>0;i--){
00325         ppp = new*ppp + pp;
00326         pp  = new*pp  + p;
00327         p   = new*p   + defl[i-1];
00328       }
00329 
00330       /* Laguerre's method */
00331       denom=(m-1) * ((m-1)*pp*pp - m*p*ppp);
00332       if(denom<0)
00333         return(-1);  /* complex root!  The LPC generator handed us a bad filter */
00334 
00335       if(pp>0){
00336         denom = pp + sqrt(denom);
00337         if(denom<EPSILON)denom=EPSILON;
00338       }else{
00339         denom = pp - sqrt(denom);
00340         if(denom>-(EPSILON))denom=-(EPSILON);
00341       }
00342 
00343       delta  = m*p/denom;
00344       new   -= delta;
00345 
00346       if(delta<0.f)delta*=-1;

00348       if(fabs(delta/new)<10e-12)break;

00353 
00354     /* forward deflation */
00355 
00356     for(i=m;i>0;i--)

00599         Position[1] -= ALContext->Listener.Position[1];
00600         Position[2] -= ALContext->Listener.Position[2];

00604         aluMatrixVector(Direction, 0.0f, Matrix);

00614     SourceToListener[2] = -Position[2];

00639             Distance=__min(Distance,MaxDist);
00640             if(MaxDist < MinDist)
00641                 break;

00643         case AL_INVERSE_DISTANCE:
00644             if(MinDist > 0.0f)
00645             {

00647                     flAttenuation = MinDist / (MinDist + (Rolloff * (Distance - MinDist)));

00654             break;

00657             Distance=__max(Distance,MinDist);
00658             Distance=__min(Distance,MaxDist);
00659             if(MaxDist < MinDist)
00660                 break;

00664             if(MaxDist != MinDist)
00665             {

00667                 for(i = 0;i < NumSends;i++)
00668                     RoomAttenuation[i] = 1.0f - (RoomRolloff[i]*(Distance-MinDist)/(MaxDist - MinDist));
00669             }
00670             break;
00671 
00672         case AL_EXPONENT_DISTANCE_CLAMPED:
00673             Distance=__max(Distance,MinDist);
00674             Distance=__min(Distance,MaxDist);
00675             if(MaxDist < MinDist)
00676                 break;
00677             //fall-through
00678         case AL_EXPONENT_DISTANCE:
00679             if(Distance > 0.0f && MinDist > 0.0f)
00680             {
00681                 flAttenuation = aluPow(Distance/MinDist, -Rolloff);
00682                 for(i = 0;i < NumSends;i++)
00683                     RoomAttenuation[i] = aluPow(Distance/MinDist, -RoomRolloff[i]);
00684             }
00685             break;
00686 
00687         case AL_NONE:
00688             break;

00692     DryMix = SourceVolume * flAttenuation;
00693     for(i = 0;i < NumSends;i++)
00694         WetGain[i] = SourceVolume * RoomAttenuation[i];
00695 
00696     effectiveDist = 0.0f;
00697     if(MinDist > 0.0f)
00698         effectiveDist = (MinDist/flAttenuation - MinDist)*MetersPerUnit;
00699 
00700     // Distance-based air absorption
00701     if(ALSource->AirAbsorptionFactor > 0.0f && effectiveDist > 0.0f)
00702     {

00704 
00705         // Absorption calculation is done in dB
00706         absorb = (ALSource->AirAbsorptionFactor*AIRABSORBGAINDBHF) *
00707                  effectiveDist;
00708         // Convert dB to linear gain before applying
00709         absorb = aluPow(10.0f, absorb/20.0f);
00710 
00711         DryGainHF *= absorb;
00712     }

00714     //3. Apply directional soundcones
00715     Angle = aluAcos(aluDotproduct(Direction,SourceToListener)) * 180.0f/M_PI;
00716     if(Angle >= InnerAngle && Angle <= OuterAngle)
00717     {
00718         ALfloat scale = (Angle-InnerAngle) / (OuterAngle-InnerAngle);
00719         ConeVolume = (1.0f+(ALSource->flOuterGain-1.0f)*scale);
00720         ConeHF = (1.0f+(OuterGainHF-1.0f)*scale);
00721     }
00722     else if(Angle > OuterAngle)
00723     {
00724         ConeVolume = (1.0f+(ALSource->flOuterGain-1.0f));
00725         ConeHF = (1.0f+(OuterGainHF-1.0f));

00727     else
00728     {

00730         ConeHF = 1.0f;

00766         if(Slot->AuxSendAuto)
00767         {

00771                 WetGainHF[i] *= ConeHF;

00794                                        ALSource->AirAbsorptionFactor * effectiveDist);
00795             }
00796         }
00797         else
00798         {

00813     }

00815     {

00817         WetGainHF[i] = 1.0f;
00818     }
00819 
00820     // Apply filter gains and filters
00821     switch(ALSource->DirectFilter.type)
00822     {
00823         case AL_FILTER_LOWPASS:

00827     }

00829 
00830     // Calculate Velocity
00831     if(DopplerFactor != 0.0f)
00832     {
00833         ALfloat flVSS, flVLS;

00839             flVSS = (flMaxVelocity - 1.0f);
00840         else if(flVSS <= -flMaxVelocity)

00845             flVLS = (flMaxVelocity - 1.0f);
00846         else if(flVLS <= -flMaxVelocity)
00847             flVLS = -flMaxVelocity + 1.0f;

00852     }
00853     else
00854         ALSource->Params.Pitch = ALSource->flPitch;

00856     // Use energy-preserving panning algorithm for multi-speaker playback
00857     length = __max(OrigDist, MinDist);
00858     if(length > 0.0f)
00859     {
00860         ALfloat invlen = 1.0f/length;
00861         Position[0] *= invlen;
00862         Position[1] *= invlen;
00863         Position[2] *= invlen;
00864     }

00866     pos = aluCart2LUTpos(-Position[2], Position[0]);

00871     // has low complexity
00872     AmbientGain = 1.0/aluSqrt(Device->NumChan) * (1.0-DirGain);

00880     }
00881 
00882     /* Update filter coefficients. */
00883     cw = cos(2.0*M_PI * LOWPASSFREQCUTOFF / Frequency);
00884 
00885     /* Spatialized sources use four chained one-pole filters, so we need to
00886      * take the fourth root of the squared gain, which is the same as the
00887      * square root of the base gain. */
00888     ALSource->Params.iirFilter.coeff = lpCoeffCalc(aluSqrt(DryGainHF), cw);
00889 
00890     for(i = 0;i < NumSends;i++)
00891     {
00892         /* The wet path uses two chained one-pole filters, so take the
00893          * base gain (square root of the squared gain) */
00894         ALSource->Params.Send[i].iirFilter.coeff = lpCoeffCalc(WetGainHF[i], cw);
00895     }
00896 }

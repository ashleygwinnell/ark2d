00257 /* _zip_checkcons:
00258    Checks the consistency of the central directory by comparing central
00259    directory entries with local headers and checking for plausible
00260    file and header offsets. Returns -1 if not plausible, else the
00261    difference between the lowest and the highest fileposition reached */
00262 
00263 static int
00264 _zip_checkcons(FILE *fp, struct zip_cdir *cd, struct zip_error *error)
00265 {
00266     int i;
00267     unsigned int min, max, j;
00268     struct zip_dirent temp;
00269 
00270     if (cd->nentry) {
00271         max = cd->entry[0].offset;
00272         min = cd->entry[0].offset;
00273     }
00274     else
00275         min = max = 0;
00276 
00277     for (i=0; i<cd->nentry; i++) {
00278         if (cd->entry[i].offset < min)
00279             min = cd->entry[i].offset;
00280         if (min > cd->offset) {
00281             _zip_error_set(error, ZIP_ER_NOZIP, 0);
00282             return -1;
00283         }
00284         
00285         j = cd->entry[i].offset + cd->entry[i].comp_size
00286             + cd->entry[i].filename_len + LENTRYSIZE;
00287         if (j > max)
00288             max = j;
00289         if (max > cd->offset) {
00290             _zip_error_set(error, ZIP_ER_NOZIP, 0);
00291             return -1;
00292         }
00293         
00294         if (fseeko(fp, cd->entry[i].offset, SEEK_SET) != 0) {
00295             _zip_error_set(error, ZIP_ER_SEEK, 0);
00296             return -1;
00297         }
00298         
00299         if (_zip_dirent_read(&temp, fp, NULL, NULL, 1, error) == -1)
00300             return -1;
00301         
00302         if (_zip_headercomp(cd->entry+i, 0, &temp, 1) != 0) {
00303             _zip_error_set(error, ZIP_ER_INCONS, 0);
00304             _zip_dirent_finalize(&temp);
00305             return -1;
00306         }
00307         _zip_dirent_finalize(&temp);
00308     }
00309 
00310     return max - min;
00311 }

00355 static int
00356 _zip_headercomp(struct zip_dirent *h1, int local1p, struct zip_dirent *h2,
00357            int local2p)
00358 {
00359     if ((h1->version_needed != h2->version_needed)
00360 #if 0

00388             || (h1->comp_size != h2->comp_size)

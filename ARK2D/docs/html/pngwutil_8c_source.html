00740 /* Write an IDAT chunk */
00741 void /* PRIVATE */
00742 png_write_IDAT(png_structp png_ptr, png_bytep data, png_size_t length)
00743 {
00744    PNG_IDAT;
00745 
00746    png_debug(1, "in png_write_IDAT");
00747 
00748    /* Optimize the CMF field in the zlib stream. */
00749    /* This hack of the zlib stream is compliant to the stream specification. */
00750    if (!(png_ptr->mode & PNG_HAVE_IDAT) &&
00751        png_ptr->compression_type == PNG_COMPRESSION_TYPE_BASE)
00752    {
00753       unsigned int z_cmf = data[0];  /* zlib compression method and flags */
00754       if ((z_cmf & 0x0f) == 8 && (z_cmf & 0xf0) <= 0x70)
00755       {
00756          /* Avoid memory underflows and multiplication overflows.
00757           *
00758           * The conditions below are practically always satisfied;
00759           * however, they still must be checked.
00760           */
00761          if (length >= 2 &&
00762              png_ptr->height < 16384 && png_ptr->width < 16384)
00763          {
00764             png_uint_32 uncompressed_idat_size = png_ptr->height *
00765                 ((png_ptr->width *
00766                 png_ptr->channels * png_ptr->bit_depth + 15) >> 3);
00767             unsigned int z_cinfo = z_cmf >> 4;
00768             unsigned int half_z_window_size = 1 << (z_cinfo + 7);
00769             while (uncompressed_idat_size <= half_z_window_size &&
00770                 half_z_window_size >= 256)
00771             {
00772                z_cinfo--;
00773                half_z_window_size >>= 1;
00774             }
00775 
00776             z_cmf = (z_cmf & 0x0f) | (z_cinfo << 4);
00777 
00778             if (data[0] != z_cmf)
00779             {
00780                int tmp;
00781                data[0] = (png_byte)z_cmf;
00782                tmp = data[1] & 0xe0;
00783                tmp += 0x1f - ((z_cmf << 8) + tmp) % 0x1f;
00784                data[1] = (png_byte)tmp;
00785             }
00786          }
00787       }
00788 
00789       else
00790          png_error(png_ptr,
00791              "Invalid zlib compression method or flags in IDAT");
00792    }
00793 
00794    png_write_chunk(png_ptr, png_IDAT, data, length);
00795    png_ptr->mode |= PNG_HAVE_IDAT;
00796 }

01271 /* Check that the tEXt or zTXt keyword is valid per PNG 1.0 specification,
01272  * and if invalid, correct the keyword rather than discarding the entire
01273  * chunk.  The PNG 1.0 specification requires keywords 1-79 characters in
01274  * length, forbids leading or trailing whitespace, multiple internal spaces,
01275  * and the non-break space (0x80) from ISO 8859-1.  Returns keyword length.
01276  *

01278  * by the calling routine.  This avoids problems with trying to write to
01279  * static keywords without having to have duplicate copies of the strings.
01280  */
01281 png_size_t /* PRIVATE */
01282 png_check_keyword(png_structp png_ptr, png_const_charp key, png_charpp new_key)
01283 {
01284    png_size_t key_len;
01285    png_const_charp ikp;
01286    png_charp kp, dp;
01287    int kflag;
01288    int kwarn=0;
01289 
01290    png_debug(1, "in png_check_keyword");
01291 
01292    *new_key = NULL;
01293 
01294    if (key == NULL || (key_len = png_strlen(key)) == 0)
01295    {
01296       png_warning(png_ptr, "zero length keyword");
01297       return ((png_size_t)0);
01298    }
01299 
01300    png_debug1(2, "Keyword to be checked is '%s'", key);
01301 
01302    *new_key = (png_charp)png_malloc_warn(png_ptr, (png_uint_32)(key_len + 2));
01303 
01304    if (*new_key == NULL)
01305    {
01306       png_warning(png_ptr, "Out of memory while procesing keyword");
01307       return ((png_size_t)0);
01308    }
01309 
01310    /* Replace non-printing characters with a blank and print a warning */
01311    for (ikp = key, dp = *new_key; *ikp != '\0'; ikp++, dp++)
01312    {
01313       if ((png_byte)*ikp < 0x20 ||
01314          ((png_byte)*ikp > 0x7E && (png_byte)*ikp < 0xA1))
01315       {
01316 #ifdef PNG_CONSOLE_IO_SUPPORTED
01317          char msg[40];
01318

01321          png_warning(png_ptr, msg);
01322 #else
01323          png_warning(png_ptr, "invalid character in keyword");
01324 #endif
01325          *dp = ' ';
01326       }
01327 
01328       else
01329       {
01330          *dp = *ikp;
01331       }
01332    }
01333    *dp = '\0';
01334 
01335    /* Remove any trailing white space. */
01336    kp = *new_key + key_len - 1;
01337    if (*kp == ' ')
01338    {
01339       png_warning(png_ptr, "trailing spaces removed from keyword");
01340 
01341       while (*kp == ' ')
01342       {
01343          *(kp--) = '\0';
01344          key_len--;
01345       }
01346    }
01347 
01348    /* Remove any leading white space. */
01349    kp = *new_key;
01350    if (*kp == ' ')
01351    {
01352       png_warning(png_ptr, "leading spaces removed from keyword");
01353 
01354       while (*kp == ' ')
01355       {
01356          kp++;
01357          key_len--;
01358       }
01359    }
01360 
01361    png_debug1(2, "Checking for multiple internal spaces in '%s'", kp);
01362 
01363    /* Remove multiple internal spaces. */
01364    for (kflag = 0, dp = *new_key; *kp != '\0'; kp++)
01365    {
01366       if (*kp == ' ' && kflag == 0)
01367       {
01368          *(dp++) = *kp;
01369          kflag = 1;
01370       }
01371 
01372       else if (*kp == ' ')
01373       {
01374          key_len--;
01375          kwarn = 1;
01376       }
01377 
01378       else
01379       {
01380          *(dp++) = *kp;
01381          kflag = 0;
01382       }
01383    }
01384    *dp = '\0';
01385    if (kwarn)
01386       png_warning(png_ptr, "extra interior spaces removed from keyword");
01387 
01388    if (key_len == 0)
01389    {
01390       png_free(png_ptr, *new_key);
01391       png_warning(png_ptr, "Zero length keyword");
01392    }
01393 
01394    if (key_len > 79)
01395    {
01396       png_warning(png_ptr, "keyword length must be 1 - 79 characters");
01397       (*new_key)[79] = '\0';
01398       key_len = 79;
01399    }
01400 
01401    return (key_len);
01402 }

02180 /* This filters the row, chooses which filter to use, if it has not already
02181  * been specified by the application, and then writes the row out with the
02182  * chosen filter.
02183  */
02184 #define PNG_MAXSUM (((png_uint_32)(-1)) >> 1)
02185 #define PNG_HISHIFT 10
02186 #define PNG_LOMASK ((png_uint_32)0xffffL)
02187 #define PNG_HIMASK ((png_uint_32)(~PNG_LOMASK >> PNG_HISHIFT))
02188 void /* PRIVATE */
02189 png_write_find_filter(png_structp png_ptr, png_row_infop row_info)
02190 {
02191    png_bytep best_row;
02192 #ifdef PNG_WRITE_FILTER_SUPPORTED
02193    png_bytep prev_row, row_buf;
02194    png_uint_32 mins, bpp;
02195    png_byte filter_to_do = png_ptr->do_filter;
02196    png_size_t row_bytes = row_info->rowbytes;
02197 #ifdef PNG_WRITE_WEIGHTED_FILTER_SUPPORTED
02198    int num_p_filters = (int)png_ptr->num_prev_filters;
02199 #endif
02200 
02201    png_debug(1, "in png_write_find_filter");
02202 
02203 #ifndef PNG_WRITE_WEIGHTED_FILTER_SUPPORTED
02204   if (png_ptr->row_number == 0 && filter_to_do == PNG_ALL_FILTERS)
02205   {
02206      /* These will never be selected so we need not test them. */
02207      filter_to_do &= ~(PNG_FILTER_UP | PNG_FILTER_PAETH);
02208   }
02209 #endif
02210 
02211    /* Find out how many bytes offset each pixel is */
02212    bpp = (row_info->pixel_depth + 7) >> 3;
02213 
02214    prev_row = png_ptr->prev_row;
02215 #endif
02216    best_row = png_ptr->row_buf;
02217 #ifdef PNG_WRITE_FILTER_SUPPORTED
02218    row_buf = best_row;
02219    mins = PNG_MAXSUM;
02220 
02221    /* The prediction method we use is to find which method provides the
02222     * smallest value when summing the absolute values of the distances
02223     * from zero, using anything >= 128 as negative numbers.  This is known
02224     * as the "minimum sum of absolute differences" heuristic.  Other
02225     * heuristics are the "weighted minimum sum of absolute differences"
02226     * (experimental and can in theory improve compression), and the "zlib
02227     * predictive" method (not implemented yet), which does test compressions
02228     * of lines using different filter methods, and then chooses the
02229     * (series of) filter(s) that give minimum compressed data size (VERY
02230     * computationally expensive).
02231     *
02232     * GRR 980525:  consider also
02233     *
02234     *   (1) minimum sum of absolute differences from running average (i.e.,
02235     *       keep running sum of non-absolute differences & count of bytes)
02236     *       [track dispersion, too?  restart average if dispersion too large?]
02237     *
02238     *  (1b) minimum sum of absolute differences from sliding average, probably
02239     *       with window size <= deflate window (usually 32K)
02240     *
02241     *   (2) minimum sum of squared differences from zero or running average
02242     *       (i.e., ~ root-mean-square approach)
02243     */
02244 
02245 
02246    /* We don't need to test the 'no filter' case if this is the only filter
02247     * that has been chosen, as it doesn't actually do anything to the data.
02248     */
02249    if ((filter_to_do & PNG_FILTER_NONE) && filter_to_do != PNG_FILTER_NONE)
02250    {
02251       png_bytep rp;
02252       png_uint_32 sum = 0;
02253       png_size_t i;
02254       int v;
02255 
02256       for (i = 0, rp = row_buf + 1; i < row_bytes; i++, rp++)
02257       {
02258          v = *rp;
02259          sum += (v < 128) ? v : 256 - v;
02260       }
02261 
02262 #ifdef PNG_WRITE_WEIGHTED_FILTER_SUPPORTED
02263       if (png_ptr->heuristic_method == PNG_FILTER_HEURISTIC_WEIGHTED)
02264       {
02265          png_uint_32 sumhi, sumlo;
02266          int j;
02267          sumlo = sum & PNG_LOMASK;
02268          sumhi = (sum >> PNG_HISHIFT) & PNG_HIMASK; /* Gives us some footroom */
02269 
02270          /* Reduce the sum if we match any of the previous rows */
02271          for (j = 0; j < num_p_filters; j++)
02272          {
02273             if (png_ptr->prev_filters[j] == PNG_FILTER_VALUE_NONE)
02274             {
02275                sumlo = (sumlo * png_ptr->filter_weights[j]) >>
02276                    PNG_WEIGHT_SHIFT;
02277 
02278                sumhi = (sumhi * png_ptr->filter_weights[j]) >>
02279                    PNG_WEIGHT_SHIFT;
02280             }
02281          }
02282 
02283          /* Factor in the cost of this filter (this is here for completeness,
02284           * but it makes no sense to have a "cost" for the NONE filter, as
02285           * it has the minimum possible computational cost - none).
02286           */
02287          sumlo = (sumlo * png_ptr->filter_costs[PNG_FILTER_VALUE_NONE]) >>
02288              PNG_COST_SHIFT;
02289 
02290          sumhi = (sumhi * png_ptr->filter_costs[PNG_FILTER_VALUE_NONE]) >>
02291              PNG_COST_SHIFT;
02292 
02293          if (sumhi > PNG_HIMASK)
02294             sum = PNG_MAXSUM;
02295 
02296          else
02297             sum = (sumhi << PNG_HISHIFT) + sumlo;
02298       }
02299 #endif
02300       mins = sum;
02301    }
02302 
02303    /* Sub filter */
02304    if (filter_to_do == PNG_FILTER_SUB)

02306    {

02308       png_size_t i;
02309 
02310       for (i = 0, rp = row_buf + 1, dp = png_ptr->sub_row + 1; i < bpp;
02311            i++, rp++, dp++)
02312       {
02313          *dp = *rp;
02314       }
02315 
02316       for (lp = row_buf + 1; i < row_bytes;

02319          *dp = (png_byte)(((int)*rp - (int)*lp) & 0xff);
02320       }
02321 
02322       best_row = png_ptr->sub_row;
02323    }
02324 
02325    else if (filter_to_do & PNG_FILTER_SUB)

02329       png_size_t i;
02330       int v;
02331 
02332 #ifdef PNG_WRITE_WEIGHTED_FILTER_SUPPORTED

02343

02345          {
02346             if (png_ptr->prev_filters[j] == PNG_FILTER_VALUE_SUB)
02347             {
02348                lmlo = (lmlo * png_ptr->inv_filter_weights[j]) >>
02349                    PNG_WEIGHT_SHIFT;
02350 
02351                lmhi = (lmhi * png_ptr->inv_filter_weights[j]) >>
02352                    PNG_WEIGHT_SHIFT;
02353             }
02354          }
02355 
02356          lmlo = (lmlo * png_ptr->inv_filter_costs[PNG_FILTER_VALUE_SUB]) >>
02357              PNG_COST_SHIFT;
02358 
02359          lmhi = (lmhi * png_ptr->inv_filter_costs[PNG_FILTER_VALUE_SUB]) >>
02360              PNG_COST_SHIFT;
02361

02374 
02375          sum += (v < 128) ? v : 256 - v;
02376       }
02377 
02378       for (lp = row_buf + 1; i < row_bytes;
02379          i++, rp++, lp++, dp++)
02380       {
02381          v = *dp = (png_byte)(((int)*rp - (int)*lp) & 0xff);
02382 
02383          sum += (v < 128) ? v : 256 - v;
02384 
02385          if (sum > lmins)  /* We are already worse, don't continue. */
02386             break;
02387       }
02388 
02389 #ifdef PNG_WRITE_WEIGHTED_FILTER_SUPPORTED
02390       if (png_ptr->heuristic_method == PNG_FILTER_HEURISTIC_WEIGHTED)
02391       {
02392          int j;
02393          png_uint_32 sumhi, sumlo;
02394          sumlo = sum & PNG_LOMASK;
02395          sumhi = (sum >> PNG_HISHIFT) & PNG_HIMASK;
02396 
02397          for (j = 0; j < num_p_filters; j++)
02398          {
02399             if (png_ptr->prev_filters[j] == PNG_FILTER_VALUE_SUB)
02400             {
02401                sumlo = (sumlo * png_ptr->inv_filter_weights[j]) >>
02402                    PNG_WEIGHT_SHIFT;
02403 
02404                sumhi = (sumhi * png_ptr->inv_filter_weights[j]) >>
02405                    PNG_WEIGHT_SHIFT;
02406             }
02407          }
02408 
02409          sumlo = (sumlo * png_ptr->inv_filter_costs[PNG_FILTER_VALUE_SUB]) >>
02410              PNG_COST_SHIFT;

02413              PNG_COST_SHIFT;

02417

02421 #endif

02437           pp = prev_row + 1; i < row_bytes;
02438           i++, rp++, pp++, dp++)
02439       {
02440          *dp = (png_byte)(((int)*rp - (int)*pp) & 0xff);
02441       }
02442

02445

02450       png_size_t i;
02451       int v;
02452

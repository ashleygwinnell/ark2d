01105 /* An internal API to validate the format of a floating point number.
01106  * The result is the index of the next character.  If the number is
01107  * not valid it will be the index of a character in the supposed number.
01108  *
01109  * The format of a number is defined in the PNG extensions specification
01110  * and this API is strictly conformant to that spec, not anyone elses!
01111  *
01112  * The format as a regular expression is:
01113  *
01114  * [+-]?[0-9]+.?([Ee][+-]?[0-9]+)?
01115  *
01116  * or:
01117  *
01118  * [+-]?.[0-9]+(.[0-9]+)?([Ee][+-]?[0-9]+)?
01119  *
01120  * The complexity is that either integer or fraction must be present and the
01121  * fraction is permitted to have no digits only if the integer is present.
01122  *
01123  * NOTE: The dangling E problem.
01124  *   There is a PNG valid floating point number in the following:
01125  *
01126  *       PNG floating point numb1.ers are not greedy.
01127  *
01128  *   Working this out requires *TWO* character lookahead (because of the
01129  *   sign), the parser does not do this - it will fail at the 'r' - this
01130  *   doesn't matter for PNG sCAL chunk values, but it requires more care
01131  *   if the value were ever to be embedded in something more complex.  Use
01132  *   ANSI-C strtod if you need the lookahead.
01133  */
01134 /* State table for the parser. */
01135 #define PNG_FP_INTEGER    0  /* before or in integer */
01136 #define PNG_FP_FRACTION   1  /* before or in fraction */
01137 #define PNG_FP_EXPONENT   2  /* before or in exponent */
01138 #define PNG_FP_STATE      3  /* mask for the above */
01139 #define PNG_FP_SAW_SIGN   4  /* Saw +/- in current state */
01140 #define PNG_FP_SAW_DIGIT  8  /* Saw a digit in current state */
01141 #define PNG_FP_SAW_DOT   16  /* Saw a dot in current state */
01142 #define PNG_FP_SAW_E     32  /* Saw an E (or e) in current state */
01143 #define PNG_FP_SAW_ANY   60  /* Saw any of the above 4 */
01144 #define PNG_FP_WAS_VALID 64  /* Preceding substring is a valid fp number */
01145 #define PNG_FP_INVALID  128  /* Available for callers as a distinct value */
01146 
01147 /* Result codes for the parser (boolean - true meants ok, false means
01148  * not ok yet.)
01149  */
01150 #define PNG_FP_MAYBE      0  /* The number may be valid in the future */
01151 #define PNG_FP_OK         1  /* The number is valid */
01152 
01153 /* The actual parser.  This can be called repeatedly, it updates
01154  * the index into the string and the state variable (which must
01155  * be initialzed to 0).  It returns a result code, as above.  There
01156  * is no point calling the parser any more if it fails to advance to
01157  * the end of the string - it is stuck on an invalid character (or
01158  * terminated by '\0').
01159  *
01160  * Note that the pointer will consume an E or even an E+ then leave
01161  * a 'maybe' state even though a preceding integer.fraction is valid.
01162  * The PNG_FP_WAS_VALID flag indicates that a preceding substring was
01163  * a valid number.  It's possible to recover from this by calling
01164  * the parser again (from the start, with state 0) but with a string
01165  * that omits the last character (i.e. set the size to the index of
01166  * the problem character.)  This has not been tested within libpng.
01167  */
01168 PNG_EXTERN int png_check_fp_number PNGARG((png_const_charp string,
01169     png_size_t size, int *statep, png_size_tp whereami));

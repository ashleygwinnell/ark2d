00740     FT_UInt       count;

00744     {
00745 #undef  FT_STRUCTURE
00746 #define FT_STRUCTURE  TT_NameTableRec
00747 
00748       FT_FRAME_START( 6 ),
00749         FT_FRAME_USHORT( format ),
00750         FT_FRAME_USHORT( numNameRecords ),
00751         FT_FRAME_USHORT( storageOffset ),
00752       FT_FRAME_END
00753     };
00754 
00755     static const FT_Frame_Field  name_record_fields[] =
00756     {
00757 #undef  FT_STRUCTURE
00758 #define FT_STRUCTURE  TT_NameEntryRec
00759 
00760       /* no FT_FRAME_START */
00761         FT_FRAME_USHORT( platformID ),
00762         FT_FRAME_USHORT( encodingID ),
00763         FT_FRAME_USHORT( languageID ),
00764         FT_FRAME_USHORT( nameID ),
00765         FT_FRAME_USHORT( stringLength ),
00766         FT_FRAME_USHORT( stringOffset ),
00767       FT_FRAME_END
00768     };
00769 
00770 
00771     table         = &face->name_table;
00772     table->stream = stream;
00773 
00774     error = face->goto_table( face, TTAG_name, stream, &table_len );
00775     if ( error )
00776       goto Exit;
00777 
00778     table_pos = FT_STREAM_POS();
00779 
00780 
00781     if ( FT_STREAM_READ_FIELDS( name_table_fields, table ) )
00782       goto Exit;
00783 
00784     /* Some popular Asian fonts have an invalid `storageOffset' value   */
00785     /* (it should be at least "6 + 12*num_names").  However, the string */
00786     /* offsets, computed as "storageOffset + entry->stringOffset", are  */
00787     /* valid pointers within the name table...                          */
00788     /*                                                                  */
00789     /* We thus can't check `storageOffset' right now.                   */
00790     /*                                                                  */
00791     storage_start = table_pos + 6 + 12*table->numNameRecords;
00792     storage_limit = table_pos + table_len;
00793 
00794     if ( storage_start > storage_limit )
00795     {
00796       FT_ERROR(( "tt_face_load_name: invalid `name' table\n" ));
00797       error = SFNT_Err_Name_Table_Missing;
00798       goto Exit;
00799     }
00800 
00801     /* Allocate the array of name records. */
00802     count                 = table->numNameRecords;
00803     table->numNameRecords = 0;
00804 
00805     if ( FT_NEW_ARRAY( table->names, count ) ||
00806          FT_FRAME_ENTER( count * 12 )        )
00807       goto Exit;
00808 
00809     /* Load the name records and determine how much storage is needed */
00810     /* to hold the strings themselves.                                */
00811     {
00812       TT_NameEntryRec*  entry = table->names;
00813 
00814 
00815       for ( ; count > 0; count-- )
00816       {
00817         if ( FT_STREAM_READ_FIELDS( name_record_fields, entry ) )
00818           continue;
00819 
00820         /* check that the name is not empty */
00821         if ( entry->stringLength == 0 )
00822           continue;
00823 
00824         /* check that the name string is within the table */
00825         entry->stringOffset += table_pos + table->storageOffset;
00826         if ( entry->stringOffset                       < storage_start ||
00827              entry->stringOffset + entry->stringLength > storage_limit )
00828         {
00829           /* invalid entry - ignore it */
00830           entry->stringOffset = 0;
00831           entry->stringLength = 0;
00832           continue;
00833         }
00834 
00835         entry++;
00836       }
00837 
00838       table->numNameRecords = (FT_UInt)( entry - table->names );
00839     }
00840 
00841     FT_FRAME_EXIT();
00842 
00843     /* everything went well, update face->num_names */
00844     face->num_names = (FT_UShort) table->numNameRecords;
00845 
00846   Exit:
00847     return error;
00848   }

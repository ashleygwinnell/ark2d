00053 _zip_mkstemp(char *path)
00054 {
00055         int fd;   
00056         char *start, *trv;
00057         struct stat sbuf;
00058         pid_t pid;
00059 
00060         /* To guarantee multiple calls generate unique names even if
00061            the file is not created. 676 different possibilities with 7
00062            or more X's, 26 with 6 or less. */
00063         static char xtra[2] = "aa";
00064         int xcnt = 0;
00065 
00066         pid = getpid();
00067 
00068         /* Move to end of path and count trailing X's. */
00069         for (trv = path; *trv; ++trv)
00070                 if (*trv == 'X')
00071                         xcnt++;
00072                 else
00073                         xcnt = 0;       
00074 
00075         /* Use at least one from xtra.  Use 2 if more than 6 X's. */
00076         if (*(trv - 1) == 'X')
00077                 *--trv = xtra[0];
00078         if (xcnt > 6 && *(trv - 1) == 'X')
00079                 *--trv = xtra[1];
00080 
00081         /* Set remaining X's to pid digits with 0's to the left. */
00082         while (*--trv == 'X') {
00083                 *trv = (pid % 10) + '0';
00084                 pid /= 10;
00085         }
00086 
00087         /* update xtra for next call. */
00088         if (xtra[0] != 'z')
00089                 xtra[0]++;
00090         else {
00091                 xtra[0] = 'a';
00092                 if (xtra[1] != 'z')
00093                         xtra[1]++;
00094                 else
00095                         xtra[1] = 'a';
00096         }
00097 
00098         /*
00099          * check the target directory; if you have six X's and it
00100          * doesn't exist this runs for a *very* long time.
00101          */
00102         for (start = trv + 1;; --trv) {
00103                 if (trv <= path)
00104                         break;
00105                 if (*trv == '/') {
00106                         *trv = '\0';
00107                         if (stat(path, &sbuf))
00108                                 return (0);
00109                         if (!S_ISDIR(sbuf.st_mode)) {
00110                                 errno = ENOTDIR;
00111                                 return (0);
00112                         }
00113                         *trv = '/';
00114                         break;
00115                 }
00116         }
00117 
00118         for (;;) {
00119                 if ((fd=open(path, O_CREAT|O_EXCL|O_RDWR|O_BINARY, 0600)) >= 0)
00120                         return (fd);
00121                 if (errno != EEXIST)
00122                         return (0);
00123 
00124                 /* tricky little algorithm for backward compatibility */
00125                 for (trv = start;;) {
00126                         if (!*trv)
00127                                 return (0);
00128                         if (*trv == 'z')
00129                                 *trv++ = 'a';
00130                         else {
00131                                 if (isdigit((unsigned char)*trv))
00132                                         *trv = 'a';
00133                                 else
00134                                         ++*trv;
00135                                 break;
00136                         }
00137                 }
00138         }

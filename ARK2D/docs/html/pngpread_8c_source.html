00907 void /* PRIVATE */
00908 png_process_IDAT_data(png_structp png_ptr, png_bytep buffer,
00909    png_size_t buffer_length)
00910 {
00911    /* The caller checks for a non-zero buffer length. */
00912    if (!(buffer_length > 0) || buffer == NULL)
00913       png_error(png_ptr, "No IDAT data (internal error)");
00914 
00915    /* This routine must process all the data it has been given
00916     * before returning, calling the row callback as required to
00917     * handle the uncompressed results.
00918     */
00919    png_ptr->zstream.next_in = buffer;
00920    png_ptr->zstream.avail_in = (uInt)buffer_length;
00921 
00922    /* Keep going until the decompressed data is all processed
00923     * or the stream marked as finished.
00924     */
00925    while (png_ptr->zstream.avail_in > 0 &&
00926           !(png_ptr->flags & PNG_FLAG_ZLIB_FINISHED))
00927    {
00928       int ret;
00929 
00930       /* We have data for zlib, but we must check that zlib
00931        * has someplace to put the results.  It doesn't matter
00932        * if we don't expect any results -- it may be the input
00933        * data is just the LZ end code.
00934        */
00935       if (!(png_ptr->zstream.avail_out > 0))
00936       {
00937          png_ptr->zstream.avail_out =
00938              (uInt) PNG_ROWBYTES(png_ptr->pixel_depth,
00939              png_ptr->iwidth) + 1;
00940 
00941          png_ptr->zstream.next_out = png_ptr->row_buf;
00942       }
00943 
00944       /* Using Z_SYNC_FLUSH here means that an unterminated
00945        * LZ stream (a stream with a missing end code) can still
00946        * be handled, otherwise (Z_NO_FLUSH) a future zlib
00947        * implementation might defer output and therefore
00948        * change the current behavior (see comments in inflate.c
00949        * for why this doesn't happen at present with zlib 1.2.5).
00950        */
00951       ret = inflate(&png_ptr->zstream, Z_SYNC_FLUSH);
00952 
00953       /* Check for any failure before proceeding. */
00954       if (ret != Z_OK && ret != Z_STREAM_END)
00955       {
00956          /* Terminate the decompression. */
00957          png_ptr->flags |= PNG_FLAG_ZLIB_FINISHED;
00958 
00959          /* This may be a truncated stream (missing or
00960           * damaged end code).  Treat that as a warning.
00961           */
00962          if (png_ptr->row_number >= png_ptr->num_rows ||
00963              png_ptr->pass > 6)
00964             png_warning(png_ptr, "Truncated compressed data in IDAT");
00965 
00966          else
00967             png_error(png_ptr, "Decompression error in IDAT");
00968 
00969          /* Skip the check on unprocessed input */
00970          return;
00971       }
00972 
00973       /* Did inflate output any data? */
00974       if (png_ptr->zstream.next_out != png_ptr->row_buf)
00975       {
00976          /* Is this unexpected data after the last row?
00977           * If it is, artificially terminate the LZ output
00978           * here.
00979           */
00980          if (png_ptr->row_number >= png_ptr->num_rows ||
00981              png_ptr->pass > 6)
00982          {
00983             /* Extra data. */
00984             png_warning(png_ptr, "Extra compressed data in IDAT");
00985             png_ptr->flags |= PNG_FLAG_ZLIB_FINISHED;
00986 
00987             /* Do no more processing; skip the unprocessed
00988              * input check below.
00989              */
00990             return;
00991          }
00992 
00993          /* Do we have a complete row? */
00994          if (png_ptr->zstream.avail_out == 0)
00995             png_push_process_row(png_ptr);
00996       }
00997 
00998       /* And check for the end of the stream. */
00999       if (ret == Z_STREAM_END)
01000          png_ptr->flags |= PNG_FLAG_ZLIB_FINISHED;
01001    }
01002 
01003    /* All the data should have been processed, if anything
01004     * is left at this point we have bytes of IDAT data
01005     * after the zlib end code.
01006     */
01007    if (png_ptr->zstream.avail_in > 0)
01008       png_warning(png_ptr, "Extra compression data in IDAT");
01009 }

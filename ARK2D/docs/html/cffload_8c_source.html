00841   static FT_Error
00842   cff_charset_load( CFF_Charset  charset,
00843                     FT_UInt      num_glyphs,
00844                     FT_Stream    stream,
00845                     FT_ULong     base_offset,
00846                     FT_ULong     offset,
00847                     FT_Bool      invert )
00848   {
00849     FT_Memory  memory = stream->memory;
00850     FT_Error   error  = CFF_Err_Ok;
00851     FT_UShort  glyph_sid;
00852 
00853 
00854     /* If the the offset is greater than 2, we have to parse the */
00855     /* charset table.                                            */
00856     if ( offset > 2 )
00857     {
00858       FT_UInt  j;
00859 
00860 
00861       charset->offset = base_offset + offset;
00862 
00863       /* Get the format of the table. */
00864       if ( FT_STREAM_SEEK( charset->offset ) ||
00865            FT_READ_BYTE( charset->format )   )
00866         goto Exit;
00867 
00868       /* Allocate memory for sids. */
00869       if ( FT_NEW_ARRAY( charset->sids, num_glyphs ) )
00870         goto Exit;
00871 
00872       /* assign the .notdef glyph */
00873       charset->sids[0] = 0;
00874 
00875       switch ( charset->format )
00876       {
00877       case 0:
00878         if ( num_glyphs > 0 )
00879         {
00880           if ( FT_FRAME_ENTER( ( num_glyphs - 1 ) * 2 ) )
00881             goto Exit;
00882 
00883           for ( j = 1; j < num_glyphs; j++ )
00884           {
00885             FT_UShort sid = FT_GET_USHORT();
00886 
00887 
00888             /* this constant is given in the CFF specification */
00889             if ( sid < 65000L )
00890               charset->sids[j] = sid;
00891             else
00892             {
00893               FT_TRACE0(( "cff_charset_load:"
00894                           " invalid SID value %d set to zero\n", sid ));
00895               charset->sids[j] = 0;
00896             }
00897           }
00898 
00899           FT_FRAME_EXIT();
00900         }
00901         break;
00902 
00903       case 1:
00904       case 2:
00905         {
00906           FT_UInt  nleft;
00907           FT_UInt  i;
00908 
00909 
00910           j = 1;
00911 
00912           while ( j < num_glyphs )
00913           {
00914             /* Read the first glyph sid of the range. */
00915             if ( FT_READ_USHORT( glyph_sid ) )
00916               goto Exit;
00917 
00918             /* Read the number of glyphs in the range.  */
00919             if ( charset->format == 2 )
00920             {
00921               if ( FT_READ_USHORT( nleft ) )
00922                 goto Exit;
00923             }
00924             else
00925             {
00926               if ( FT_READ_BYTE( nleft ) )
00927                 goto Exit;
00928             }
00929 
00930             /* check whether the range contains at least one valid glyph; */
00931             /* the constant is given in the CFF specification             */
00932             if ( glyph_sid >= 65000L )
00933             {
00934               FT_ERROR(( "cff_charset_load: invalid SID range\n" ));
00935               error = CFF_Err_Invalid_File_Format;
00936               goto Exit;
00937             }
00938 
00939             /* try to rescue some of the SIDs if `nleft' is too large */
00940             if ( nleft > 65000L - 1L || glyph_sid >= 65000L - nleft )
00941             {
00942               FT_ERROR(( "cff_charset_load: invalid SID range trimmed\n" ));
00943               nleft = ( FT_UInt )( 65000L - 1L - glyph_sid );
00944             }
00945 
00946             /* Fill in the range of sids -- `nleft + 1' glyphs. */
00947             for ( i = 0; j < num_glyphs && i <= nleft; i++, j++, glyph_sid++ )
00948               charset->sids[j] = glyph_sid;
00949           }
00950         }
00951         break;
00952 
00953       default:
00954         FT_ERROR(( "cff_charset_load: invalid table format\n" ));
00955         error = CFF_Err_Invalid_File_Format;
00956         goto Exit;
00957       }
00958     }
00959     else
00960     {
00961       /* Parse default tables corresponding to offset == 0, 1, or 2.  */
00962       /* CFF specification intimates the following:                   */
00963       /*                                                              */
00964       /* In order to use a predefined charset, the following must be  */
00965       /* true: The charset constructed for the glyphs in the font's   */
00966       /* charstrings dictionary must match the predefined charset in  */
00967       /* the first num_glyphs.                                        */
00968 
00969       charset->offset = offset;  /* record charset type */
00970 
00971       switch ( (FT_UInt)offset )
00972       {
00973       case 0:
00974         if ( num_glyphs > 229 )
00975         {
00976           FT_ERROR(( "cff_charset_load: implicit charset larger than\n"
00977                      "predefined charset (Adobe ISO-Latin)\n" ));
00978           error = CFF_Err_Invalid_File_Format;
00979           goto Exit;
00980         }
00981 
00982         /* Allocate memory for sids. */
00983         if ( FT_NEW_ARRAY( charset->sids, num_glyphs ) )
00984           goto Exit;
00985 
00986         /* Copy the predefined charset into the allocated memory. */
00987         FT_ARRAY_COPY( charset->sids, cff_isoadobe_charset, num_glyphs );
00988 
00989         break;
00990 
00991       case 1:
00992         if ( num_glyphs > 166 )
00993         {
00994           FT_ERROR(( "cff_charset_load: implicit charset larger than\n"
00995                      "predefined charset (Adobe Expert)\n" ));
00996           error = CFF_Err_Invalid_File_Format;
00997           goto Exit;
00998         }
00999 
01000         /* Allocate memory for sids. */
01001         if ( FT_NEW_ARRAY( charset->sids, num_glyphs ) )
01002           goto Exit;
01003 
01004         /* Copy the predefined charset into the allocated memory.     */
01005         FT_ARRAY_COPY( charset->sids, cff_expert_charset, num_glyphs );
01006 
01007         break;
01008 
01009       case 2:
01010         if ( num_glyphs > 87 )
01011         {
01012           FT_ERROR(( "cff_charset_load: implicit charset larger than\n"
01013                      "predefined charset (Adobe Expert Subset)\n" ));
01014           error = CFF_Err_Invalid_File_Format;
01015           goto Exit;
01016         }
01017 
01018         /* Allocate memory for sids. */
01019         if ( FT_NEW_ARRAY( charset->sids, num_glyphs ) )
01020           goto Exit;
01021 
01022         /* Copy the predefined charset into the allocated memory.     */
01023         FT_ARRAY_COPY( charset->sids, cff_expertsubset_charset, num_glyphs );
01024 
01025         break;
01026 
01027       default:
01028         error = CFF_Err_Invalid_File_Format;
01029         goto Exit;
01030       }
01031     }
01032 
01033     /* we have to invert the `sids' array for subsetted CID-keyed fonts */
01034     if ( invert )
01035       error = cff_charset_compute_cids( charset, num_glyphs, memory );
01036 
01037   Exit:
01038     /* Clean up if there was an error. */
01039     if ( error )
01040     {
01041       FT_FREE( charset->sids );
01042       FT_FREE( charset->cids );
01043       charset->format = 0;
01044       charset->offset = 0;
01045       charset->sids   = 0;
01046     }
01047 
01048     return error;
01049   }

01061   static FT_Error
01062   cff_encoding_load( CFF_Encoding  encoding,
01063                      CFF_Charset   charset,
01064                      FT_UInt       num_glyphs,
01065                      FT_Stream     stream,
01066                      FT_ULong      base_offset,
01067                      FT_ULong      offset )
01068   {
01069     FT_Error   error = CFF_Err_Ok;
01070     FT_UInt    count;
01071     FT_UInt    j;
01072     FT_UShort  glyph_sid;
01073     FT_UInt    glyph_code;
01074 
01075 
01076     /* Check for charset->sids.  If we do not have this, we fail. */
01077     if ( !charset->sids )
01078     {
01079       error = CFF_Err_Invalid_File_Format;
01080       goto Exit;
01081     }
01082 
01083     /* Zero out the code to gid/sid mappings. */
01084     for ( j = 0; j < 256; j++ )
01085     {
01086       encoding->sids [j] = 0;
01087       encoding->codes[j] = 0;
01088     }
01089 
01090     /* Note: The encoding table in a CFF font is indexed by glyph index;  */
01091     /* the first encoded glyph index is 1.  Hence, we read the character  */
01092     /* code (`glyph_code') at index j and make the assignment:            */
01093     /*                                                                    */
01094     /*    encoding->codes[glyph_code] = j + 1                             */
01095     /*                                                                    */
01096     /* We also make the assignment:                                       */
01097     /*                                                                    */
01098     /*    encoding->sids[glyph_code] = charset->sids[j + 1]               */
01099     /*                                                                    */
01100     /* This gives us both a code to GID and a code to SID mapping.        */
01101 
01102     if ( offset > 1 )
01103     {
01104       encoding->offset = base_offset + offset;
01105 
01106       /* we need to parse the table to determine its size */
01107       if ( FT_STREAM_SEEK( encoding->offset ) ||
01108            FT_READ_BYTE( encoding->format )   ||
01109            FT_READ_BYTE( count )              )
01110         goto Exit;
01111 
01112       switch ( encoding->format & 0x7F )
01113       {
01114       case 0:
01115         {
01116           FT_Byte*  p;
01117 
01118 
01119           /* By convention, GID 0 is always ".notdef" and is never */
01120           /* coded in the font.  Hence, the number of codes found  */
01121           /* in the table is `count+1'.                            */
01122           /*                                                       */
01123           encoding->count = count + 1;
01124 
01125           if ( FT_FRAME_ENTER( count ) )
01126             goto Exit;
01127 
01128           p = (FT_Byte*)stream->cursor;
01129 
01130           for ( j = 1; j <= count; j++ )
01131           {
01132             glyph_code = *p++;
01133 
01134             /* Make sure j is not too big. */
01135             if ( j < num_glyphs )
01136             {
01137               /* Assign code to GID mapping. */
01138               encoding->codes[glyph_code] = (FT_UShort)j;
01139 
01140               /* Assign code to SID mapping. */
01141               encoding->sids[glyph_code] = charset->sids[j];
01142             }
01143           }
01144 
01145           FT_FRAME_EXIT();
01146         }
01147         break;
01148 
01149       case 1:
01150         {
01151           FT_UInt  nleft;
01152           FT_UInt  i = 1;
01153           FT_UInt  k;
01154 
01155 
01156           encoding->count = 0;
01157 
01158           /* Parse the Format1 ranges. */
01159           for ( j = 0;  j < count; j++, i += nleft )
01160           {
01161             /* Read the first glyph code of the range. */
01162             if ( FT_READ_BYTE( glyph_code ) )
01163               goto Exit;
01164 
01165             /* Read the number of codes in the range. */
01166             if ( FT_READ_BYTE( nleft ) )
01167               goto Exit;
01168 
01169             /* Increment nleft, so we read `nleft + 1' codes/sids. */
01170             nleft++;
01171 
01172             /* compute max number of character codes */
01173             if ( (FT_UInt)nleft > encoding->count )
01174               encoding->count = nleft;
01175 
01176             /* Fill in the range of codes/sids. */
01177             for ( k = i; k < nleft + i; k++, glyph_code++ )
01178             {
01179               /* Make sure k is not too big. */
01180               if ( k < num_glyphs && glyph_code < 256 )
01181               {
01182                 /* Assign code to GID mapping. */
01183                 encoding->codes[glyph_code] = (FT_UShort)k;
01184 
01185                 /* Assign code to SID mapping. */
01186                 encoding->sids[glyph_code] = charset->sids[k];
01187               }
01188             }
01189           }
01190 
01191           /* simple check; one never knows what can be found in a font */
01192           if ( encoding->count > 256 )
01193             encoding->count = 256;
01194         }
01195         break;
01196 
01197       default:
01198         FT_ERROR(( "cff_encoding_load: invalid table format\n" ));
01199         error = CFF_Err_Invalid_File_Format;
01200         goto Exit;
01201       }
01202 
01203       /* Parse supplemental encodings, if any. */
01204       if ( encoding->format & 0x80 )
01205       {
01206         FT_UInt  gindex;
01207 
01208 
01209         /* count supplements */
01210         if ( FT_READ_BYTE( count ) )
01211           goto Exit;
01212 
01213         for ( j = 0; j < count; j++ )
01214         {
01215           /* Read supplemental glyph code. */
01216           if ( FT_READ_BYTE( glyph_code ) )
01217             goto Exit;
01218 
01219           /* Read the SID associated with this glyph code. */
01220           if ( FT_READ_USHORT( glyph_sid ) )
01221             goto Exit;
01222 
01223           /* Assign code to SID mapping. */
01224           encoding->sids[glyph_code] = glyph_sid;
01225 
01226           /* First, look up GID which has been assigned to */
01227           /* SID glyph_sid.                                */
01228           for ( gindex = 0; gindex < num_glyphs; gindex++ )
01229           {
01230             if ( charset->sids[gindex] == glyph_sid )
01231             {
01232               encoding->codes[glyph_code] = (FT_UShort)gindex;
01233               break;
01234             }
01235           }
01236         }
01237       }
01238     }
01239     else
01240     {
01241       /* We take into account the fact a CFF font can use a predefined */
01242       /* encoding without containing all of the glyphs encoded by this */
01243       /* encoding (see the note at the end of section 12 in the CFF    */
01244       /* specification).                                               */
01245 
01246       switch ( (FT_UInt)offset )
01247       {
01248       case 0:
01249         /* First, copy the code to SID mapping. */
01250         FT_ARRAY_COPY( encoding->sids, cff_standard_encoding, 256 );
01251         goto Populate;
01252 
01253       case 1:
01254         /* First, copy the code to SID mapping. */
01255         FT_ARRAY_COPY( encoding->sids, cff_expert_encoding, 256 );
01256 
01257       Populate:
01258         /* Construct code to GID mapping from code to SID mapping */
01259         /* and charset.                                           */
01260 
01261         encoding->count = 0;
01262 
01263         error = cff_charset_compute_cids( charset, num_glyphs,
01264                                           stream->memory );
01265         if ( error )
01266           goto Exit;
01267 
01268         for ( j = 0; j < 256; j++ )
01269         {
01270           FT_UInt  sid = encoding->sids[j];
01271           FT_UInt  gid = 0;
01272 
01273 
01274           if ( sid )
01275             gid = cff_charset_cid_to_gindex( charset, sid );
01276 
01277           if ( gid != 0 )
01278           {
01279             encoding->codes[j] = (FT_UShort)gid;
01280 
01281             if ( encoding->count < j + 1 )
01282               encoding->count = j + 1;
01283           }
01284           else
01285           {
01286             encoding->codes[j] = 0;
01287             encoding->sids [j] = 0;
01288           }
01289         }
01290         break;
01291 
01292       default:
01293         FT_ERROR(( "cff_encoding_load: invalid table format\n" ));
01294         error = CFF_Err_Invalid_File_Format;
01295         goto Exit;
01296       }
01297     }
01298 
01299   Exit:
01300 
01301     /* Clean up if there was an error. */
01302     return error;
01303   }

01424   FT_LOCAL_DEF( FT_Error )
01425   cff_font_load( FT_Library library,
01426                  FT_Stream  stream,
01427                  FT_Int     face_index,
01428                  CFF_Font   font,
01429                  FT_Bool    pure_cff )
01430   {
01431     static const FT_Frame_Field  cff_header_fields[] =
01432     {
01433 #undef  FT_STRUCTURE
01434 #define FT_STRUCTURE  CFF_FontRec
01435 
01436       FT_FRAME_START( 4 ),
01437         FT_FRAME_BYTE( version_major ),
01438         FT_FRAME_BYTE( version_minor ),
01439         FT_FRAME_BYTE( header_size ),
01440         FT_FRAME_BYTE( absolute_offsize ),
01441       FT_FRAME_END
01442     };
01443 
01444     FT_Error         error;
01445     FT_Memory        memory = stream->memory;
01446     FT_ULong         base_offset;
01447     CFF_FontRecDict  dict;
01448     CFF_IndexRec     string_index;
01449 
01450 
01451     FT_ZERO( font );
01452     FT_ZERO( &string_index );
01453 
01454     font->stream = stream;
01455     font->memory = memory;
01456     dict         = &font->top_font.font_dict;
01457     base_offset  = FT_STREAM_POS();
01458 
01459     /* read CFF font header */
01460     if ( FT_STREAM_READ_FIELDS( cff_header_fields, font ) )
01461       goto Exit;
01462 
01463     /* check format */
01464     if ( font->version_major   != 1 ||
01465          font->header_size      < 4 ||
01466          font->absolute_offsize > 4 )
01467     {
01468       FT_TRACE2(( "[not a CFF font header]\n" ));
01469       error = CFF_Err_Unknown_File_Format;
01470       goto Exit;
01471     }
01472 
01473     /* skip the rest of the header */
01474     if ( FT_STREAM_SKIP( font->header_size - 4 ) )
01475       goto Exit;
01476 
01477     /* read the name, top dict, string and global subrs index */
01478     if ( FT_SET_ERROR( cff_index_init( &font->name_index,
01479                                        stream, 0 ) )                  ||
01480          FT_SET_ERROR( cff_index_init( &font->font_dict_index,
01481                                        stream, 0 ) )                  ||
01482          FT_SET_ERROR( cff_index_init( &string_index,
01483                                        stream, 1 ) )                  ||
01484          FT_SET_ERROR( cff_index_init( &font->global_subrs_index,
01485                                        stream, 1 ) )                  ||
01486          FT_SET_ERROR( cff_index_get_pointers( &string_index,
01487                                                &font->strings,
01488                                                &font->string_pool ) ) )
01489       goto Exit;
01490 
01491     font->num_strings = string_index.count;
01492 
01493     /* well, we don't really forget the `disabled' fonts... */
01494     font->num_faces = font->name_index.count;
01495     if ( face_index >= (FT_Int)font->num_faces )
01496     {
01497       FT_ERROR(( "cff_font_load: incorrect face index = %d\n",
01498                  face_index ));
01499       error = CFF_Err_Invalid_Argument;
01500     }
01501 
01502     /* in case of a font format check, simply exit now */
01503     if ( face_index < 0 )
01504       goto Exit;
01505 
01506     /* now, parse the top-level font dictionary */
01507     error = cff_subfont_load( &font->top_font,
01508                               &font->font_dict_index,
01509                               face_index,
01510                               stream,
01511                               base_offset,
01512                               library );
01513     if ( error )
01514       goto Exit;
01515 
01516     if ( FT_STREAM_SEEK( base_offset + dict->charstrings_offset ) )
01517       goto Exit;
01518 
01519     error = cff_index_init( &font->charstrings_index, stream, 0 );
01520     if ( error )
01521       goto Exit;
01522 
01523     /* now, check for a CID font */
01524     if ( dict->cid_registry != 0xFFFFU )
01525     {
01526       CFF_IndexRec  fd_index;
01527       CFF_SubFont   sub;
01528       FT_UInt       idx;
01529 
01530 
01531       /* this is a CID-keyed font, we must now allocate a table of */
01532       /* sub-fonts, then load each of them separately              */
01533       if ( FT_STREAM_SEEK( base_offset + dict->cid_fd_array_offset ) )
01534         goto Exit;
01535 
01536       error = cff_index_init( &fd_index, stream, 0 );
01537       if ( error )
01538         goto Exit;
01539 
01540       if ( fd_index.count > CFF_MAX_CID_FONTS )
01541       {
01542         FT_TRACE0(( "cff_font_load: FD array too large in CID font\n" ));
01543         goto Fail_CID;
01544       }
01545 
01546       /* allocate & read each font dict independently */
01547       font->num_subfonts = fd_index.count;
01548       if ( FT_NEW_ARRAY( sub, fd_index.count ) )
01549         goto Fail_CID;
01550 
01551       /* set up pointer table */
01552       for ( idx = 0; idx < fd_index.count; idx++ )
01553         font->subfonts[idx] = sub + idx;
01554 
01555       /* now load each subfont independently */
01556       for ( idx = 0; idx < fd_index.count; idx++ )
01557       {
01558         sub = font->subfonts[idx];
01559         error = cff_subfont_load( sub, &fd_index, idx,
01560                                   stream, base_offset, library );
01561         if ( error )
01562           goto Fail_CID;
01563       }
01564 
01565       /* now load the FD Select array */
01566       error = CFF_Load_FD_Select( &font->fd_select,
01567                                   font->charstrings_index.count,
01568                                   stream,
01569                                   base_offset + dict->cid_fd_select_offset );
01570 
01571     Fail_CID:
01572       cff_index_done( &fd_index );
01573 
01574       if ( error )
01575         goto Exit;
01576     }
01577     else
01578       font->num_subfonts = 0;
01579 
01580     /* read the charstrings index now */
01581     if ( dict->charstrings_offset == 0 )
01582     {
01583       FT_ERROR(( "cff_font_load: no charstrings offset\n" ));
01584       error = CFF_Err_Unknown_File_Format;
01585       goto Exit;
01586     }
01587 
01588     font->num_glyphs = font->charstrings_index.count;
01589 
01590     error = cff_index_get_pointers( &font->global_subrs_index,
01591                                     &font->global_subrs, NULL );
01592 
01593     if ( error )
01594       goto Exit;
01595 
01596     /* read the Charset and Encoding tables if available */
01597     if ( font->num_glyphs > 0 )
01598     {
01599       FT_Bool  invert = FT_BOOL( dict->cid_registry != 0xFFFFU && pure_cff );
01600 
01601 
01602       error = cff_charset_load( &font->charset, font->num_glyphs, stream,
01603                                 base_offset, dict->charset_offset, invert );
01604       if ( error )
01605         goto Exit;
01606 
01607       /* CID-keyed CFFs don't have an encoding */
01608       if ( dict->cid_registry == 0xFFFFU )
01609       {
01610         error = cff_encoding_load( &font->encoding,
01611                                    &font->charset,
01612                                    font->num_glyphs,
01613                                    stream,
01614                                    base_offset,
01615                                    dict->encoding_offset );
01616         if ( error )
01617           goto Exit;
01618       }
01619     }
01620 
01621     /* get the font name (/CIDFontName for CID-keyed fonts, */
01622     /* /FontName otherwise)                                 */
01623     font->font_name = cff_index_get_name( font, face_index );
01624 
01625   Exit:
01626     cff_index_done( &string_index );
01627 
01628     return error;
01629   }

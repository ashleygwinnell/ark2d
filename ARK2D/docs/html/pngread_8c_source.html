01319     */
01320    png_read_info(png_ptr, info_ptr);
01321    if (info_ptr->height > PNG_UINT_32_MAX/png_sizeof(png_bytep))
01322       png_error(png_ptr, "Image is too high to process with png_read_png()");
01323 
01324    /* -------------- image transformations start here ------------------- */
01325 
01326 #ifdef PNG_READ_16_TO_8_SUPPORTED
01327    /* Tell libpng to strip 16 bit/color files down to 8 bits per color.
01328     */
01329    if (transforms & PNG_TRANSFORM_STRIP_16)
01330       png_set_strip_16(png_ptr);
01331 #endif
01332 
01333 #ifdef PNG_READ_STRIP_ALPHA_SUPPORTED
01334    /* Strip alpha bytes from the input data without combining with
01335     * the background (not recommended).
01336     */
01337    if (transforms & PNG_TRANSFORM_STRIP_ALPHA)
01338       png_set_strip_alpha(png_ptr);
01339 #endif
01340 
01341 #if defined(PNG_READ_PACK_SUPPORTED) && !defined(PNG_READ_EXPAND_SUPPORTED)

01344     */
01345    if (transforms & PNG_TRANSFORM_PACKING)
01346       png_set_packing(png_ptr);
01347 #endif
01348 
01349 #ifdef PNG_READ_PACKSWAP_SUPPORTED
01350    /* Change the order of packed pixels to least significant bit first
01351     * (not useful if you are using png_set_packing).
01352     */
01353    if (transforms & PNG_TRANSFORM_PACKSWAP)
01354       png_set_packswap(png_ptr);
01355 #endif
01356 
01357 #ifdef PNG_READ_EXPAND_SUPPORTED
01358    /* Expand paletted colors into true RGB triplets
01359     * Expand grayscale images to full 8 bits from 1, 2, or 4 bits/pixel
01360     * Expand paletted or RGB images with transparency to full alpha
01361     * channels so the data will be available as RGBA quartets.
01362     */
01363    if (transforms & PNG_TRANSFORM_EXPAND)
01364       if ((png_ptr->bit_depth < 8) ||
01365           (png_ptr->color_type == PNG_COLOR_TYPE_PALETTE) ||
01366           (png_get_valid(png_ptr, info_ptr, PNG_INFO_tRNS)))
01367          png_set_expand(png_ptr);

01370    /* We don't handle background color or gamma transformation or quantizing.
01371     */
01372 
01373 #ifdef PNG_READ_INVERT_SUPPORTED
01374    /* Invert monochrome files to have 0 as white and 1 as black
01375     */
01376    if (transforms & PNG_TRANSFORM_INVERT_MONO)
01377       png_set_invert_mono(png_ptr);
01378 #endif
01379 
01380 #ifdef PNG_READ_SHIFT_SUPPORTED
01381    /* If you want to shift the pixel values from the range [0,255] or
01382     * [0,65535] to the original [0,7] or [0,31], or whatever range the

01397    if (transforms & PNG_TRANSFORM_BGR)
01398       png_set_bgr(png_ptr);
01399 #endif

01401 #ifdef PNG_READ_SWAP_ALPHA_SUPPORTED
01402    /* Swap the RGBA or GA data to ARGB or AG (or BGRA to ABGR) */

00478   for(i=0;i<fits;i++){
00479     double weight = (a[i].bn+a[i].an)*info->twofitweight/(a[i].an+1)+1.;
00480 
00481     xb+=a[i].xb + a[i].xa * weight;
00482     yb+=a[i].yb + a[i].ya * weight;
00483     x2b+=a[i].x2b + a[i].x2a * weight;
00484     y2b+=a[i].y2b + a[i].y2a * weight;
00485     xyb+=a[i].xyb + a[i].xya * weight;
00486     bn+=a[i].bn + a[i].an * weight;
00487   }
00488 
00489   if(*y0>=0){
00490     xb+=   x0;
00491     yb+=  *y0;
00492     x2b+=  x0 *  x0;
00493     y2b+= *y0 * *y0;
00494     xyb+= *y0 *  x0;
00495     bn++;
00496   }
00497 
00498   if(*y1>=0){
00499     xb+=   x1;
00500     yb+=  *y1;
00501     x2b+=  x1 *  x1;
00502     y2b+= *y1 * *y1;
00503     xyb+= *y1 *  x1;
00504     bn++;
00505   }
00506 
00507   {
00508     double denom=(bn*x2b-xb*xb);
00509 
00510     if(denom>0.){
00511       double a=(yb*x2b-xyb*xb)/denom;
00512       double b=(bn*xyb-xb*yb)/denom;
00513       *y0=rint(a+b*x0);
00514       *y1=rint(a+b*x1);
00515 
00516       /* limit to our range! */
00517       if(*y0>1023)*y0=1023;
00518       if(*y1>1023)*y1=1023;
00519       if(*y0<0)*y0=0;
00520       if(*y1<0)*y1=0;
00521 
00522       return 0;
00523     }else{
00524       *y0=0;
00525       *y1=0;
00526       return 1;
00527     }
00528   }
00529 }

00531 static int inspect_error(int x0,int x1,int y0,int y1,const float *mask,
00532                          const float *mdct,
00533                          vorbis_info_floor1 *info){
00534   int dy=y1-y0;
00535   int adx=x1-x0;
00536   int ady=abs(dy);
00537   int base=dy/adx;
00538   int sy=(dy<0?base-1:base+1);
00539   int x=x0;
00540   int y=y0;
00541   int err=0;
00542   int val=vorbis_dBquant(mask+x);
00543   int mse=0;
00544   int n=0;
00545 
00546   ady-=abs(base*adx);
00547 
00548   mse=(y-val);
00549   mse*=mse;
00550   n++;
00551   if(mdct[x]+info->twofitatten>=mask[x]){
00552     if(y+info->maxover<val)return(1);
00553     if(y-info->maxunder>val)return(1);
00554   }
00555 
00556   while(++x<x1){
00557     err=err+ady;
00558     if(err>=adx){
00559       err-=adx;
00560       y+=sy;
00561     }else{
00562       y+=base;
00563     }
00564 
00565     val=vorbis_dBquant(mask+x);
00566     mse+=((y-val)*(y-val));
00567     n++;
00568     if(mdct[x]+info->twofitatten>=mask[x]){
00569       if(val){
00570         if(y+info->maxover<val)return(1);
00571         if(y-info->maxunder>val)return(1);
00572       }
00573     }
00574   }
00575 
00576   if(info->maxover*info->maxover/n>info->maxerr)return(0);
00577   if(info->maxunder*info->maxunder/n>info->maxerr)return(0);
00578   if(mse/n>info->maxerr)return(1);
00579   return(0);
00580 }

00582 static int post_Y(int *A,int *B,int pos){
00583   if(A[pos]<0)
00584     return B[pos];
00585   if(B[pos]<0)
00586     return A[pos];
00587 
00588   return (A[pos]+B[pos])>>1;
00589 }

00591 int *floor1_fit(vorbis_block *vb,vorbis_look_floor1 *look,
00592                           const float *logmdct,   /* in */
00593                           const float *logmask){
00594   long i,j;
00595   vorbis_info_floor1 *info=look->vi;
00596   long n=look->n;
00597   long posts=look->posts;
00598   long nonzero=0;
00599   lsfit_acc fits[VIF_POSIT+1];
00600   int fit_valueA[VIF_POSIT+2]; /* index by range list position */
00601   int fit_valueB[VIF_POSIT+2]; /* index by range list position */
00602 
00603   int loneighbor[VIF_POSIT+2]; /* sorted index of range list position (+2) */
00604   int hineighbor[VIF_POSIT+2];
00605   int *output=NULL;
00606   int memo[VIF_POSIT+2];
00607 
00608   for(i=0;i<posts;i++)fit_valueA[i]=-200; /* mark all unused */
00609   for(i=0;i<posts;i++)fit_valueB[i]=-200; /* mark all unused */
00610   for(i=0;i<posts;i++)loneighbor[i]=0; /* 0 for the implicit 0 post */
00611   for(i=0;i<posts;i++)hineighbor[i]=1; /* 1 for the implicit post at n */
00612   for(i=0;i<posts;i++)memo[i]=-1;      /* no neighbor yet */
00613 
00614   /* quantize the relevant floor points and collect them into line fit
00615      structures (one per minimal division) at the same time */
00616   if(posts==0){
00617     nonzero+=accumulate_fit(logmask,logmdct,0,n,fits,n,info);
00618   }else{
00619     for(i=0;i<posts-1;i++)
00620       nonzero+=accumulate_fit(logmask,logmdct,look->sorted_index[i],
00621                               look->sorted_index[i+1],fits+i,
00622                               n,info);
00623   }
00624 
00625   if(nonzero){
00626     /* start by fitting the implicit base case.... */
00627     int y0=-200;
00628     int y1=-200;
00629     fit_line(fits,posts-1,&y0,&y1,info);
00630 
00631     fit_valueA[0]=y0;
00632     fit_valueB[0]=y0;
00633     fit_valueB[1]=y1;
00634     fit_valueA[1]=y1;
00635 
00636     /* Non degenerate case */
00637     /* start progressive splitting.  This is a greedy, non-optimal
00638        algorithm, but simple and close enough to the best
00639        answer. */
00640     for(i=2;i<posts;i++){
00641       int sortpos=look->reverse_index[i];
00642       int ln=loneighbor[sortpos];
00643       int hn=hineighbor[sortpos];
00644 
00645       /* eliminate repeat searches of a particular range with a memo */
00646       if(memo[ln]!=hn){
00647         /* haven't performed this error search yet */
00648         int lsortpos=look->reverse_index[ln];
00649         int hsortpos=look->reverse_index[hn];
00650         memo[ln]=hn;
00651 
00652         {
00653           /* A note: we want to bound/minimize *local*, not global, error */
00654           int lx=info->postlist[ln];
00655           int hx=info->postlist[hn];
00656           int ly=post_Y(fit_valueA,fit_valueB,ln);
00657           int hy=post_Y(fit_valueA,fit_valueB,hn);
00658 
00659           if(ly==-1 || hy==-1){
00660             exit(1);
00661           }
00662 
00663           if(inspect_error(lx,hx,ly,hy,logmask,logmdct,info)){
00664             /* outside error bounds/begin search area.  Split it. */
00665             int ly0=-200;
00666             int ly1=-200;
00667             int hy0=-200;
00668             int hy1=-200;
00669             int ret0=fit_line(fits+lsortpos,sortpos-lsortpos,&ly0,&ly1,info);
00670             int ret1=fit_line(fits+sortpos,hsortpos-sortpos,&hy0,&hy1,info);
00671 
00672             if(ret0){
00673               ly0=ly;
00674               ly1=hy0;
00675             }
00676             if(ret1){
00677               hy0=ly1;
00678               hy1=hy;
00679             }
00680 
00681             if(ret0 && ret1){
00682               fit_valueA[i]=-200;
00683               fit_valueB[i]=-200;
00684             }else{
00685               /* store new edge values */
00686               fit_valueB[ln]=ly0;
00687               if(ln==0)fit_valueA[ln]=ly0;
00688               fit_valueA[i]=ly1;
00689               fit_valueB[i]=hy0;
00690               fit_valueA[hn]=hy1;
00691               if(hn==1)fit_valueB[hn]=hy1;
00692 
00693               if(ly1>=0 || hy0>=0){
00694                 /* store new neighbor values */
00695                 for(j=sortpos-1;j>=0;j--)
00696                   if(hineighbor[j]==hn)
00697                     hineighbor[j]=i;
00698                   else
00699                     break;
00700                 for(j=sortpos+1;j<posts;j++)
00701                   if(loneighbor[j]==ln)
00702                     loneighbor[j]=i;
00703                   else
00704                     break;
00705               }
00706             }
00707           }else{
00708             fit_valueA[i]=-200;
00709             fit_valueB[i]=-200;
00710           }
00711         }
00712       }
00713     }
00714 
00715     output=_vorbis_block_alloc(vb,sizeof(*output)*posts);
00716 
00717     output[0]=post_Y(fit_valueA,fit_valueB,0);
00718     output[1]=post_Y(fit_valueA,fit_valueB,1);
00719 
00720     /* fill in posts marked as not using a fit; we will zero
00721        back out to 'unused' when encoding them so long as curve
00722        interpolation doesn't force them into use */
00723     for(i=2;i<posts;i++){
00724       int ln=look->loneighbor[i-2];
00725       int hn=look->hineighbor[i-2];
00726       int x0=info->postlist[ln];
00727       int x1=info->postlist[hn];
00728       int y0=output[ln];
00729       int y1=output[hn];
00730 
00731       int predicted=render_point(x0,x1,y0,y1,info->postlist[i]);
00732       int vx=post_Y(fit_valueA,fit_valueB,i);
00733 
00734       if(vx>=0 && predicted!=vx){
00735         output[i]=vx;
00736       }else{
00737         output[i]= predicted|0x8000;
00738       }
00739     }
00740   }
00741 
00742   return(output);
00743 
00744 }

00749

00751   long posts=look->posts;

00753

00756

00768 int floor1_encode(oggpack_buffer *opb,vorbis_block *vb,
00769                   vorbis_look_floor1 *look,
00770                   int *post,int *ilogmask){
00771 
00772   long i,j;
00773   vorbis_info_floor1 *info=look->vi;
00774   long posts=look->posts;
00775   codec_setup_info *ci=vb->vd->vi->codec_setup;
00776   int out[VIF_POSIT+2];
00777   static_codebook **sbooks=ci->book_param;
00778   codebook *books=ci->fullbooks;
00779 
00780   /* quantize values to multiplier spec */
00781   if(post){
00782     for(i=0;i<posts;i++){
00783       int val=post[i]&0x7fff;
00784       switch(info->mult){
00785       case 1: /* 1024 -> 256 */
00786         val>>=2;
00787         break;
00788       case 2: /* 1024 -> 128 */
00789         val>>=3;
00790         break;
00791       case 3: /* 1024 -> 86 */
00792         val/=12;
00793         break;
00794       case 4: /* 1024 -> 64 */
00795         val>>=4;
00796         break;
00797       }
00798       post[i]=val | (post[i]&0x8000);
00799     }
00800 
00801     out[0]=post[0];
00802     out[1]=post[1];
00803 
00804     /* find prediction values for each post and subtract them */
00805     for(i=2;i<posts;i++){
00806       int ln=look->loneighbor[i-2];
00807       int hn=look->hineighbor[i-2];
00808       int x0=info->postlist[ln];
00809       int x1=info->postlist[hn];
00810       int y0=post[ln];
00811       int y1=post[hn];
00812 
00813       int predicted=render_point(x0,x1,y0,y1,info->postlist[i]);
00814 
00815       if((post[i]&0x8000) || (predicted==post[i])){
00816         post[i]=predicted|0x8000; /* in case there was roundoff jitter
00817                                      in interpolation */
00818         out[i]=0;
00819       }else{
00820         int headroom=(look->quant_q-predicted<predicted?
00821                       look->quant_q-predicted:predicted);
00822 
00823         int val=post[i]-predicted;
00824 
00825         /* at this point the 'deviation' value is in the range +/- max
00826            range, but the real, unique range can always be mapped to
00827            only [0-maxrange).  So we want to wrap the deviation into
00828            this limited range, but do it in the way that least screws
00829            an essentially gaussian probability distribution. */
00830 
00831         if(val<0)
00832           if(val<-headroom)
00833             val=headroom-val-1;
00834           else
00835             val=-1-(val<<1);
00836         else
00837           if(val>=headroom)
00838             val= val+headroom;
00839           else
00840             val<<=1;
00841 
00842         out[i]=val;
00843         post[ln]&=0x7fff;
00844         post[hn]&=0x7fff;
00845       }
00846     }
00847 
00848     /* we have everything we need. pack it out */
00849     /* mark nontrivial floor */
00850     oggpack_write(opb,1,1);
00851 
00852     /* beginning/end post */
00853     look->frames++;
00854     look->postbits+=ilog(look->quant_q-1)*2;
00855     oggpack_write(opb,out[0],ilog(look->quant_q-1));
00856     oggpack_write(opb,out[1],ilog(look->quant_q-1));
00857 
00858 
00859     /* partition by partition */
00860     for(i=0,j=2;i<info->partitions;i++){
00861       int class=info->partitionclass[i];
00862       int cdim=info->class_dim[class];
00863       int csubbits=info->class_subs[class];
00864       int csub=1<<csubbits;
00865       int bookas[8]={0,0,0,0,0,0,0,0};
00866       int cval=0;
00867       int cshift=0;
00868       int k,l;
00869 
00870       /* generate the partition's first stage cascade value */
00871       if(csubbits){
00872         int maxval[8];
00873         for(k=0;k<csub;k++){
00874           int booknum=info->class_subbook[class][k];
00875           if(booknum<0){
00876             maxval[k]=1;
00877           }else{
00878             maxval[k]=sbooks[info->class_subbook[class][k]]->entries;
00879           }
00880         }
00881         for(k=0;k<cdim;k++){
00882           for(l=0;l<csub;l++){
00883             int val=out[j+k];
00884             if(val<maxval[l]){
00885               bookas[k]=l;
00886               break;
00887             }
00888           }
00889           cval|= bookas[k]<<cshift;
00890           cshift+=csubbits;
00891         }
00892         /* write it */
00893         look->phrasebits+=
00894           vorbis_book_encode(books+info->class_book[class],cval,opb);
00895 
00896 #ifdef TRAIN_FLOOR1
00897         {
00898           FILE *of;
00899           char buffer[80];
00900           sprintf(buffer,"line_%dx%ld_class%d.vqd",
00901                   vb->pcmend/2,posts-2,class);
00902           of=fopen(buffer,"a");
00903           fprintf(of,"%d\n",cval);
00904           fclose(of);
00905         }
00906 #endif
00907       }
00908 
00909       /* write post values */
00910       for(k=0;k<cdim;k++){
00911         int book=info->class_subbook[class][bookas[k]];
00912         if(book>=0){
00913           /* hack to allow training with 'bad' books */
00914           if(out[j+k]<(books+book)->entries)
00915             look->postbits+=vorbis_book_encode(books+book,
00916                                                out[j+k],opb);
00917           /*else
00918             fprintf(stderr,"+!");*/
00919 
00920 #ifdef TRAIN_FLOOR1
00921           {
00922             FILE *of;
00923             char buffer[80];
00924             sprintf(buffer,"line_%dx%ld_%dsub%d.vqd",
00925                     vb->pcmend/2,posts-2,class,bookas[k]);
00926             of=fopen(buffer,"a");
00927             fprintf(of,"%d\n",out[j+k]);
00928             fclose(of);
00929           }
00930 #endif
00931         }
00932       }
00933       j+=cdim;
00934     }
00935 
00936     {
00937       /* generate quantized floor equivalent to what we'd unpack in decode */
00938       /* render the lines */
00939       int hx=0;
00940       int lx=0;
00941       int ly=post[0]*info->mult;
00942       int n=ci->blocksizes[vb->W]/2;
00943 
00944       for(j=1;j<look->posts;j++){
00945         int current=look->forward_index[j];
00946         int hy=post[current]&0x7fff;
00947         if(hy==post[current]){
00948 
00949           hy*=info->mult;
00950           hx=info->postlist[current];
00951 
00952           render_line0(n,lx,hx,ly,hy,ilogmask);
00953 
00954           lx=hx;
00955           ly=hy;
00956         }
00957       }
00958       for(j=hx;j<vb->pcmend/2;j++)ilogmask[j]=ly; /* be certain */
00959       return(1);
00960     }
00961   }else{
00962     oggpack_write(opb,0,1);
00963     memset(ilogmask,0,vb->pcmend/2*sizeof(*ilogmask));
00964     return(0);
00965   }
00966 }

00971   codec_setup_info   *ci=vb->vd->vi->codec_setup;
00972 
00973   int i,j,k;
00974   codebook *books=ci->fullbooks;
00975 
00976   /* unpack wrapped/predicted values from stream */
00977   if(oggpack_read(&vb->opb,1)==1){
00978     int *fit_value=_vorbis_block_alloc(vb,(look->posts)*sizeof(*fit_value));
00979 
00980     fit_value[0]=oggpack_read(&vb->opb,ilog(look->quant_q-1));
00981     fit_value[1]=oggpack_read(&vb->opb,ilog(look->quant_q-1));
00982 
00983     /* partition by partition */
00984     for(i=0,j=2;i<info->partitions;i++){
00985       int class=info->partitionclass[i];
00986       int cdim=info->class_dim[class];
00987       int csubbits=info->class_subs[class];
00988       int csub=1<<csubbits;

00994 
00995         if(cval==-1)goto eop;

00998       for(k=0;k<cdim;k++){
00999         int book=info->class_subbook[class][cval&(csub-1)];
01000         cval>>=csubbits;

01003             goto eop;

01005           fit_value[j+k]=0;
01006         }
01007       }
01008       j+=cdim;
01009     }
01010 
01011     /* unwrap positive values and reconsitute via linear interpolation */
01012     for(i=2;i<look->posts;i++){
01013       int predicted=render_point(info->postlist[look->loneighbor[i-2]],
01014                                  info->postlist[look->hineighbor[i-2]],
01015                                  fit_value[look->loneighbor[i-2]],
01016                                  fit_value[look->hineighbor[i-2]],
01017                                  info->postlist[i]);
01018       int hiroom=look->quant_q-predicted;
01019       int loroom=predicted;
01020       int room=(hiroom<loroom?hiroom:loroom)<<1;
01021       int val=fit_value[i];
01022 
01023       if(val){
01024         if(val>=room){
01025           if(hiroom>loroom){
01026             val = val-loroom;
01027           }else{
01028             val = -1-(val-hiroom);
01029           }
01030         }else{
01031           if(val&1){
01032             val= -((val+1)>>1);
01033           }else{
01034             val>>=1;
01035           }
01036         }
01037 
01038         //fit_value[i]=val+predicted&0x7fff;
01039         fit_value[i]=val+(predicted&0x7fff);
01040         fit_value[look->loneighbor[i-2]]&=0x7fff;
01041         fit_value[look->hineighbor[i-2]]&=0x7fff;
01042 
01043       }else{
01044         fit_value[i]=predicted|0x8000;
01045       }
01046 
01047     }
01048 
01049     return(fit_value);
01050   }
01051  eop:
01052   return(NULL);
01053 }

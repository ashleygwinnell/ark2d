00558          channels/samples. */
00559       int j,k,ch=0;
00560       for(i=0;i<ci->maps&&ch==0;i++){
00561         vorbis_info_mapping0 *mi=(vorbis_info_mapping0 *)ci->map_param[i];
00562         for(j=0;j<mi->submaps && ch==0;j++)
00563           if(mi->residuesubmap[j]==number) /* we found a submap referencing theis residue backend */
00564             for(k=0;k<vi->channels;k++)
00565               if(mi->chmuxlist[k]==j) /* this channel belongs to the submap */
00566                 ch++;
00567       }
00568 
00569       r->end=(int)((freq/nyq*blocksize*ch)/r->grouping+.9)* /* round up only if we're well past */
00570         r->grouping;
00571       /* the blocksize and grouping may disagree at the end */
00572       if(r->end>blocksize*ch)r->end=blocksize*ch/r->grouping*r->grouping;
00573 
00574     }else{
00575 
00576       r->end=(int)((freq/nyq*blocksize)/r->grouping+.9)* /* round up only if we're well past */
00577         r->grouping;
00578       /* the blocksize and grouping may disagree at the end */
00579       if(r->end>blocksize)r->end=blocksize/r->grouping*r->grouping;
00580 
00581     }
00582 
00583     if(r->end==0)r->end=r->grouping; /* LFE channel */
00584 
00585   }
00586 }

00633 static const void *get_setup_template(long ch,long srate,
00634                                       double req,int q_or_bitrate,
00635                                       double *base_setting){
00636   int i=0,j;
00637   if(q_or_bitrate)req/=ch;
00638 
00639   while(setup_list[i]){
00640     if(setup_list[i]->coupling_restriction==-1 ||
00641        setup_list[i]->coupling_restriction==ch){
00642       if(srate>=setup_list[i]->samplerate_min_restriction &&
00643          srate<=setup_list[i]->samplerate_max_restriction){
00644         int mappings=setup_list[i]->mappings;
00645         const double *map=(q_or_bitrate?
00646                      setup_list[i]->rate_mapping:
00647                      setup_list[i]->quality_mapping);
00648 
00649         /* the template matches.  Does the requested quality mode
00650            fall within this template's modes? */
00651         if(req<map[0]){++i;continue;}
00652         if(req>map[setup_list[i]->mappings]){++i;continue;}
00653         for(j=0;j<mappings;j++)
00654           if(req>=map[j] && req<map[j+1])break;
00655         /* an all-points match */
00656         if(j==mappings)
00657           *base_setting=j-.001;
00658         else{
00659           float low=map[j];
00660           float high=map[j+1];
00661           float del=(req-low)/(high-low);
00662           *base_setting=j+del;
00663         }
00664 
00665         return(setup_list[i]);
00666       }
00667     }
00668     i++;
00669   }
00670 
00671   return NULL;
00672 }

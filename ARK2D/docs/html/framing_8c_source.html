00649 /* sync the stream.  This is meant to be useful for finding page
00650    boundaries.
00651 
00652    return values for this:
00653   -n) skipped n bytes
00654    0) page not ready; more data (no bytes skipped)
00655    n) page synced at current location; page length n bytes
00656 
00657 */
00658 
00659 long ogg_sync_pageseek(ogg_sync_state *oy,ogg_page *og){
00660   unsigned char *page=oy->data+oy->returned;
00661   unsigned char *next;
00662   long bytes=oy->fill-oy->returned;
00663 
00664   if(ogg_sync_check(oy))return 0;
00665 
00666   if(oy->headerbytes==0){
00667     int headerbytes,i;
00668     if(bytes<27)return(0); /* not enough for a header */
00669 
00670     /* verify capture pattern */
00671     if(memcmp(page,"OggS",4))goto sync_fail;
00672 
00673     headerbytes=page[26]+27;
00674     if(bytes<headerbytes)return(0); /* not enough for header + seg table */
00675 
00676     /* count up body length in the segment table */
00677 
00678     for(i=0;i<page[26];i++)
00679       oy->bodybytes+=page[27+i];
00680     oy->headerbytes=headerbytes;
00681   }
00682 
00683   if(oy->bodybytes+oy->headerbytes>bytes)return(0);
00684 
00685   /* The whole test page is buffered.  Verify the checksum */
00686   {
00687     /* Grab the checksum bytes, set the header field to zero */
00688     char chksum[4];
00689     ogg_page log;
00690 
00691     memcpy(chksum,page+22,4);
00692     memset(page+22,0,4);
00693 
00694     /* set up a temp page struct and recompute the checksum */
00695     log.header=page;
00696     log.header_len=oy->headerbytes;
00697     log.body=page+oy->headerbytes;
00698     log.body_len=oy->bodybytes;
00699     ogg_page_checksum_set(&log);
00700 
00701     /* Compare */
00702     if(memcmp(chksum,page+22,4)){
00703       /* D'oh.  Mismatch! Corrupt page (or miscapture and not a page
00704          at all) */
00705       /* replace the computed checksum with the one actually read in */
00706       memcpy(page+22,chksum,4);
00707 
00708       /* Bad checksum. Lose sync */
00709       goto sync_fail;
00710     }
00711   }
00712 
00713   /* yes, have a whole page all ready to go */
00714   {
00715     unsigned char *page=oy->data+oy->returned;
00716     long bytes;
00717 
00718     if(og){
00719       og->header=page;
00720       og->header_len=oy->headerbytes;
00721       og->body=page+oy->headerbytes;
00722       og->body_len=oy->bodybytes;
00723     }
00724 
00725     oy->unsynced=0;
00726     oy->returned+=(bytes=oy->headerbytes+oy->bodybytes);
00727     oy->headerbytes=0;
00728     oy->bodybytes=0;
00729     return(bytes);
00730   }
00731 
00732  sync_fail:
00733 
00734   oy->headerbytes=0;
00735   oy->bodybytes=0;
00736 
00737   /* search for possible capture */
00738   next=memchr(page+1,'O',bytes-1);
00739   if(!next)
00740     next=oy->data+oy->fill;
00741 
00742   oy->returned=(int)(next-oy->data);
00743   return((long)-(next-page));
00744 }

00746 /* sync the stream and get a page.  Keep trying until we find a page.
00747    Suppress 'sync errors' after reporting the first.
00748 
00749    return values:
00750    -1) recapture (hole in data)
00751     0) need more data
00752     1) page returned
00753 
00754    Returns pointers into buffered data; invalidated by next call to
00755    _stream, _clear, _init, or _buffer */
00756 
00757 int ogg_sync_pageout(ogg_sync_state *oy, ogg_page *og){
00758 
00759   if(ogg_sync_check(oy))return 0;
00760 
00761   /* all we need to do is verify a page at the head of the stream
00762      buffer.  If it doesn't verify, we look for the next potential
00763      frame */
00764 
00765   for(;;){

00770     }
00771     if(ret==0){

00773       return(0);
00774     }
00775 
00776     /* head did not start a synced page... skipped some bytes */
00777     if(!oy->unsynced){
00778       oy->unsynced=1;
00779       return(-1);
00780     }
00781 
00782     /* loop. keep looking */
00783 
00784   }
00785 }

00792   unsigned char *body=og->body;
00793   long           bodysize=og->body_len;
00794   int            segptr=0;
00795 
00796   int version=ogg_page_version(og);
00797   int continued=ogg_page_continued(og);
00798   int bos=ogg_page_bos(og);
00799   int eos=ogg_page_eos(og);
00800   ogg_int64_t granulepos=ogg_page_granulepos(og);
00801   int serialno=ogg_page_serialno(og);
00802   long pageno=ogg_page_pageno(og);
00803   int segments=header[26];
00804 
00805   if(ogg_stream_check(os)) return -1;
00806 
00807   /* clean up 'returned data' */
00808   {
00809     long lr=os->lacing_returned;
00810     long br=os->body_returned;
00811 
00812     /* body data */
00813     if(br){
00814       os->body_fill-=br;
00815       if(os->body_fill)
00816         memmove(os->body_data,os->body_data+br,os->body_fill);
00817       os->body_returned=0;
00818     }
00819 
00820     if(lr){
00821       /* segment table */
00822       if(os->lacing_fill-lr){
00823         memmove(os->lacing_vals,os->lacing_vals+lr,
00824                 (os->lacing_fill-lr)*sizeof(*os->lacing_vals));
00825         memmove(os->granule_vals,os->granule_vals+lr,
00826                 (os->lacing_fill-lr)*sizeof(*os->granule_vals));
00827       }
00828       os->lacing_fill-=lr;
00829       os->lacing_packet-=lr;
00830       os->lacing_returned=0;
00831     }
00832   }
00833 
00834   /* check the serial number */
00835   if(serialno!=os->serialno)return(-1);
00836   if(version>0)return(-1);
00837 
00838   if(_os_lacing_expand(os,segments+1)) return -1;
00839 
00840   /* are we in sequence? */
00841   if(pageno!=os->pageno){
00842     int i;
00843 
00844     /* unroll previous partial packet (if any) */
00845     for(i=os->lacing_packet;i<os->lacing_fill;i++)
00846       os->body_fill-=os->lacing_vals[i]&0xff;
00847     os->lacing_fill=os->lacing_packet;
00848 
00849     /* make a note of dropped data in segment table */
00850     if(os->pageno!=-1){
00851       os->lacing_vals[os->lacing_fill++]=0x400;
00852       os->lacing_packet++;
00853     }
00854   }
00855 
00856   /* are we a 'continued packet' page?  If so, we may need to skip
00857      some segments */
00858   if(continued){
00859     if(os->lacing_fill<1 ||
00860        os->lacing_vals[os->lacing_fill-1]==0x400){
00861       bos=0;
00862       for(;segptr<segments;segptr++){
00863         int val=header[27+segptr];
00864         body+=val;
00865         bodysize-=val;
00866         if(val<255){
00867           segptr++;
00868           break;
00869         }
00870       }
00871     }
00872   }
00873 
00874   if(bodysize){
00875     if(_os_body_expand(os,bodysize)) return -1;
00876     memcpy(os->body_data+os->body_fill,body,bodysize);
00877     os->body_fill+=bodysize;
00878   }
00879 
00880   {
00881     int saved=-1;
00882     while(segptr<segments){
00883       int val=header[27+segptr];
00884       os->lacing_vals[os->lacing_fill]=val;
00885       os->granule_vals[os->lacing_fill]=-1;
00886 
00887       if(bos){
00888         os->lacing_vals[os->lacing_fill]|=0x100;
00889         bos=0;
00890       }
00891 
00892       if(val<255)saved=os->lacing_fill;
00893 
00894       os->lacing_fill++;
00895       segptr++;
00896 
00897       if(val<255)os->lacing_packet=os->lacing_fill;
00898     }
00899 
00900     /* set the granulepos on the last granuleval of the last full packet */
00901     if(saved!=-1){
00902       os->granule_vals[saved]=granulepos;
00903     }
00904 
00905   }
00906 
00907   if(eos){
00908     os->e_o_s=1;
00909     if(os->lacing_fill>0)
00910       os->lacing_vals[os->lacing_fill-1]|=0x200;
00911   }
00912 
00913   os->pageno=pageno+1;
00914 
00915   return(0);
00916 }

00958 static int _packetout(ogg_stream_state *os,ogg_packet *op,int adv){
00959 
00960   /* The last part of decode. We have the stream broken into packet
00961      segments.  Now we need to group them into packets (or return the
00962      out of sync markers) */
00963 
00964   int ptr=os->lacing_returned;
00965 
00966   if(os->lacing_packet<=ptr)return(0);
00967 
00968   if(os->lacing_vals[ptr]&0x400){
00969     /* we need to tell the codec there's a gap; it might need to
00970        handle previous packet dependencies. */
00971     os->lacing_returned++;
00972     os->packetno++;
00973     return(-1);
00974   }
00975 
00976   if(!op && !adv)return(1); /* just using peek as an inexpensive way
00977                                to ask if there's a whole packet
00978                                waiting */
00979 
00980   /* Gather the whole packet. We'll have no holes or a partial packet */
00981   {
00982     int size=os->lacing_vals[ptr]&0xff;
00983     long bytes=size;
00984     int eos=os->lacing_vals[ptr]&0x200; /* last packet of the stream? */
00985     int bos=os->lacing_vals[ptr]&0x100; /* first packet of the stream? */
00986 
00987     while(size==255){
00988       int val=os->lacing_vals[++ptr];
00989       size=val&0xff;
00990       if(val&0x200)eos=0x200;
00991       bytes+=size;
00992     }
00993 
00994     if(op){
00995       op->e_o_s=eos;
00996       op->b_o_s=bos;
00997       op->packet=os->body_data+os->body_returned;
00998       op->packetno=os->packetno;
00999       op->granulepos=os->granule_vals[ptr];
01000       op->bytes=bytes;
01001     }
01002 
01003     if(adv){
01004       os->body_returned+=bytes;
01005       os->lacing_returned=ptr+1;
01006       os->packetno++;
01007     }
01008   }
01009   return(1);
01010 }

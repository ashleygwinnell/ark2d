00665 /* fetch and process a packet.  Handles the case where we're at a
00666    bitstream boundary and dumps the decoding machine.  If the decoding
00667    machine is unloaded, it loads it.  It also keeps pcm_offset up to
00668    date (seek and read both use this.  seek uses a special hack with
00669    readp).
00670 
00671    return: <0) error, OV_HOLE (lost packet) or OV_EOF
00672             0) need more data (only if readp==0)
00673             1) got a packet
00674 */
00675 
00676 static int _fetch_and_process_packet(OggVorbis_File *vf,
00677                                      ogg_packet *op_in,
00678                                      int readp,
00679                                      int spanp){
00680   ogg_page og;
00681 
00682   /* handle one packet.  Try to fetch it from current stream state */
00683   /* extract packets from page */
00684   while(1){
00685 
00686     if(vf->ready_state==STREAMSET){
00687       int ret=_make_decode_ready(vf);
00688       if(ret<0)return ret;
00689     }
00690 
00691     /* process a packet if we can. */
00692 
00693     if(vf->ready_state==INITSET){
00694       int hs=vorbis_synthesis_halfrate_p(vf->vi);
00695 
00696       while(1) {
00697               ogg_packet op;
00698               ogg_packet *op_ptr=(op_in?op_in:&op);
00699         int result=ogg_stream_packetout(&vf->os,op_ptr);
00700         ogg_int64_t granulepos;
00701 
00702         op_in=NULL;
00703         if(result==-1)return(OV_HOLE); /* hole in the data. */
00704         if(result>0){
00705           /* got a packet.  process it */
00706           granulepos=op_ptr->granulepos;
00707           if(!vorbis_synthesis(&vf->vb,op_ptr)){ /* lazy check for lazy
00708                                                     header handling.  The
00709                                                     header packets aren't
00710                                                     audio, so if/when we
00711                                                     submit them,
00712                                                     vorbis_synthesis will
00713                                                     reject them */
00714 
00715             /* suck in the synthesis data and track bitrate */
00716             {
00717               int oldsamples=vorbis_synthesis_pcmout(&vf->vd,NULL);
00718               /* for proper use of libvorbis within libvorbisfile,
00719                  oldsamples will always be zero. */
00720               if(oldsamples)return(OV_EFAULT);
00721 
00722               vorbis_synthesis_blockin(&vf->vd,&vf->vb);
00723               vf->samptrack+=(vorbis_synthesis_pcmout(&vf->vd,NULL)<<hs);
00724               vf->bittrack+=op_ptr->bytes*8;
00725             }
00726 
00727             /* update the pcm offset. */
00728             if(granulepos!=-1 && !op_ptr->e_o_s){
00729               int link=(vf->seekable?vf->current_link:0);
00730               int i,samples;
00731 
00732               /* this packet has a pcm_offset on it (the last packet
00733                  completed on a page carries the offset) After processing
00734                  (above), we know the pcm position of the *last* sample
00735                  ready to be returned. Find the offset of the *first*
00736 
00737                  As an aside, this trick is inaccurate if we begin
00738                  reading anew right at the last page; the end-of-stream
00739                  granulepos declares the last frame in the stream, and the
00740                  last packet of the last page may be a partial frame.
00741                  So, we need a previous granulepos from an in-sequence page
00742                  to have a reference point.  Thus the !op_ptr->e_o_s clause
00743                  above */
00744 
00745               if(vf->seekable && link>0)
00746                 granulepos-=vf->pcmlengths[link*2];
00747               if(granulepos<0)granulepos=0; /* actually, this
00748                                                shouldn't be possible
00749                                                here unless the stream
00750                                                is very broken */
00751 
00752               samples=(vorbis_synthesis_pcmout(&vf->vd,NULL)<<hs);
00753 
00754               granulepos-=samples;
00755               for(i=0;i<link;i++)
00756                 granulepos+=vf->pcmlengths[i*2+1];
00757               vf->pcm_offset=granulepos;
00758             }
00759             return(1);
00760           }
00761         }
00762         else
00763           break;
00764       }
00765     }
00766 
00767     if(vf->ready_state>=OPENED){
00768       ogg_int64_t ret;
00769 
00770       while(1){
00771         /* the loop is not strictly necessary, but there's no sense in
00772            doing the extra checks of the larger loop for the common
00773            case in a multiplexed bistream where the page is simply
00774            part of a different logical bitstream; keep reading until
00775            we get one with the correct serialno */
00776 
00777         if(!readp)return(0);
00778         if((ret=_get_next_page(vf,&og,-1))<0){
00779           return(OV_EOF); /* eof. leave unitialized */
00780         }
00781 
00782         /* bitrate tracking; add the header's bytes here, the body bytes
00783            are done by packet above */
00784         vf->bittrack+=og.header_len*8;
00785 
00786         if(vf->ready_state==INITSET){
00787           if(vf->current_serialno!=ogg_page_serialno(&og)){
00788 
00789             /* two possibilities:
00790                1) our decoding just traversed a bitstream boundary
00791                2) another stream is multiplexed into this logical section */
00792 
00793             if(ogg_page_bos(&og)){
00794               /* boundary case */
00795               if(!spanp)
00796                 return(OV_EOF);
00797 
00798               _decode_clear(vf);
00799 
00800               if(!vf->seekable){
00801                 vorbis_info_clear(vf->vi);
00802                 vorbis_comment_clear(vf->vc);
00803               }
00804               break;
00805 
00806             }else
00807               continue; /* possibility #2 */
00808           }
00809         }
00810 
00811         break;
00812       }
00813     }
00814 
00815     /* Do we need to load a new machine before submitting the page? */
00816     /* This is different in the seekable and non-seekable cases.
00817 
00818        In the seekable case, we already have all the header
00819        information loaded and cached; we just initialize the machine
00820        with it and continue on our merry way.
00821 
00822        In the non-seekable (streaming) case, we'll only be at a
00823        boundary if we just left the previous logical bitstream and
00824        we're now nominally at the header of the next bitstream
00825     */
00826 
00827     if(vf->ready_state!=INITSET){
00828       int link;
00829 
00830       if(vf->ready_state<STREAMSET){
00831         if(vf->seekable){
00832           long serialno = ogg_page_serialno(&og);
00833 
00834           /* match the serialno to bitstream section.  We use this rather than
00835              offset positions to avoid problems near logical bitstream
00836              boundaries */
00837 
00838           for(link=0;link<vf->links;link++)
00839             if(vf->serialnos[link]==serialno)break;
00840 
00841           if(link==vf->links) continue; /* not the desired Vorbis
00842                                            bitstream section; keep
00843                                            trying */
00844 
00845           vf->current_serialno=serialno;
00846           vf->current_link=link;
00847 
00848           ogg_stream_reset_serialno(&vf->os,vf->current_serialno);
00849           vf->ready_state=STREAMSET;
00850 
00851         }else{
00852           /* we're streaming */
00853           /* fetch the three header packets, build the info struct */
00854 
00855           int ret=_fetch_headers(vf,vf->vi,vf->vc,NULL,NULL,&og);
00856           if(ret)return(ret);
00857           vf->current_serialno=vf->os.serialno;
00858           vf->current_link++;
00859           link=0;
00860         }
00861       }
00862     }
00863 
00864     /* the buffered page is the data we want, and we're ready for it;
00865        add it to the stream state */
00866     ogg_stream_pagein(&vf->os,&og);
00867 
00868   }
00869 }

01128           }else{
01129             return vf->vi[i].bitrate_upper;
01130           }
01131         }

01133       }
01134     }

01209   if(!vf->seekable || i>=vf->links)return(OV_EINVAL);

01221 /* seek to an offset relative to the *compressed* data. This also
01222    scans packets to update the PCM cursor. It will cross a logical
01223    bitstream boundary, but only if it can't get any packets out of the
01224    tail of the bitstream we seek to (so no surprises).
01225 
01226    returns zero on success, nonzero on failure */
01227 
01228 int ov_raw_seek(OggVorbis_File *vf,ogg_int64_t pos){
01229   ogg_stream_state work_os;
01230   int ret;
01231 
01232   if(vf->ready_state<OPENED)return(OV_EINVAL);
01233   if(!vf->seekable)
01234     return(OV_ENOSEEK); /* don't dump machine if we can't seek */
01235 
01236   if(pos<0 || pos>vf->end)return(OV_EINVAL);
01237 
01238   /* is the seek position outside our current link [if any]? */
01239   if(vf->ready_state>=STREAMSET){
01240     if(pos<vf->offsets[vf->current_link] || pos>=vf->offsets[vf->current_link+1])
01241       _decode_clear(vf); /* clear out stream state */
01242   }
01243 
01244   /* don't yet clear out decoding machine (if it's initialized), in
01245      the case we're in the same link.  Restart the decode lapping, and
01246      let _fetch_and_process_packet deal with a potential bitstream
01247      boundary */
01248   vf->pcm_offset=-1;
01249   ogg_stream_reset_serialno(&vf->os,
01250                             vf->current_serialno); /* must set serialno */
01251   vorbis_synthesis_restart(&vf->vd);
01252 
01253   ret=_seek_helper(vf,pos);
01254   if(ret)goto seek_error;
01255 
01256   /* we need to make sure the pcm_offset is set, but we don't want to
01257      advance the raw cursor past good packets just to get to the first
01258      with a granulepos.  That's not equivalent behavior to beginning
01259      decoding as immediately after the seek position as possible.
01260 
01261      So, a hack.  We use two stream states; a local scratch state and
01262      the shared vf->os stream state.  We use the local state to
01263      scan, and the shared state as a buffer for later decode.
01264 
01265      Unfortuantely, on the last page we still advance to last packet
01266      because the granulepos on the last page is not necessarily on a
01267      packet boundary, and we need to make sure the granpos is
01268      correct.
01269   */
01270 
01271   {
01272     ogg_page og;
01273     ogg_packet op;
01274     int lastblock=0;
01275     int accblock=0;
01276     int thisblock=0;
01277     int lastflag=0;
01278     int firstflag=0;
01279     ogg_int64_t pagepos=-1;
01280 
01281     ogg_stream_init(&work_os,vf->current_serialno); /* get the memory ready */
01282     ogg_stream_reset(&work_os); /* eliminate the spurious OV_HOLE
01283                                    return from not necessarily
01284                                    starting from the beginning */
01285 
01286     while(1){
01287       if(vf->ready_state>=STREAMSET){
01288         /* snarf/scan a packet if we can */
01289         int result=ogg_stream_packetout(&work_os,&op);
01290 
01291         if(result>0){
01292 
01293           if(vf->vi[vf->current_link].codec_setup){
01294             thisblock=vorbis_packet_blocksize(vf->vi+vf->current_link,&op);
01295             if(thisblock<0){
01296               ogg_stream_packetout(&vf->os,NULL);
01297               thisblock=0;
01298             }else{
01299 
01300               /* We can't get a guaranteed correct pcm position out of the
01301                  last page in a stream because it might have a 'short'
01302                  granpos, which can only be detected in the presence of a
01303                  preceding page.  However, if the last page is also the first
01304                  page, the granpos rules of a first page take precedence.  Not
01305                  only that, but for first==last, the EOS page must be treated
01306                  as if its a normal first page for the stream to open/play. */
01307               if(lastflag && !firstflag)
01308                 ogg_stream_packetout(&vf->os,NULL);
01309               else
01310                 if(lastblock)accblock+=(lastblock+thisblock)>>2;
01311             }
01312 
01313             if(op.granulepos!=-1){
01314               int i,link=vf->current_link;
01315               ogg_int64_t granulepos=op.granulepos-vf->pcmlengths[link*2];
01316               if(granulepos<0)granulepos=0;
01317 
01318               for(i=0;i<link;i++)
01319                 granulepos+=vf->pcmlengths[i*2+1];
01320               vf->pcm_offset=granulepos-accblock;
01321               if(vf->pcm_offset<0)vf->pcm_offset=0;
01322               break;
01323             }
01324             lastblock=thisblock;
01325             continue;
01326           }else
01327             ogg_stream_packetout(&vf->os,NULL);
01328         }
01329       }
01330 
01331       if(!lastblock){
01332         pagepos=_get_next_page(vf,&og,-1);
01333         if(pagepos<0){
01334           vf->pcm_offset=ov_pcm_total(vf,-1);
01335           break;
01336         }
01337       }else{
01338         /* huh?  Bogus stream with packets but no granulepos */
01339         vf->pcm_offset=-1;
01340         break;
01341       }
01342 
01343       /* has our decoding just traversed a bitstream boundary? */
01344       if(vf->ready_state>=STREAMSET){
01345         if(vf->current_serialno!=ogg_page_serialno(&og)){
01346 
01347           /* two possibilities:
01348              1) our decoding just traversed a bitstream boundary
01349              2) another stream is multiplexed into this logical section? */
01350 
01351           if(ogg_page_bos(&og)){
01352             /* we traversed */
01353             _decode_clear(vf); /* clear out stream state */
01354             ogg_stream_clear(&work_os);
01355           } /* else, do nothing; next loop will scoop another page */
01356         }
01357       }
01358 
01359       if(vf->ready_state<STREAMSET){
01360         int link;
01361         long serialno = ogg_page_serialno(&og);
01362 
01363         for(link=0;link<vf->links;link++)
01364           if(vf->serialnos[link]==serialno)break;
01365 
01366         if(link==vf->links) continue; /* not the desired Vorbis
01367                                          bitstream section; keep
01368                                          trying */
01369         vf->current_link=link;
01370         vf->current_serialno=serialno;
01371         ogg_stream_reset_serialno(&vf->os,serialno);
01372         ogg_stream_reset_serialno(&work_os,serialno);
01373         vf->ready_state=STREAMSET;
01374         firstflag=(pagepos<=vf->dataoffsets[link]);
01375       }
01376 
01377       ogg_stream_pagein(&vf->os,&og);
01378       ogg_stream_pagein(&work_os,&og);
01379       lastflag=ogg_page_eos(&og);
01380 
01381     }
01382   }
01383 
01384   ogg_stream_clear(&work_os);
01385   vf->bittrack=0.f;
01386   vf->samptrack=0.f;
01387   return(0);
01388 
01389  seek_error:
01390   /* dump the machine so we're in a known state */
01391   vf->pcm_offset=-1;
01392   ogg_stream_clear(&work_os);
01393   _decode_clear(vf);
01394   return OV_EBADLINK;
01395 }

01397 /* Page granularity seek (faster than sample granularity because we
01398    don't do the last bit of decode to find a specific sample).
01399 
01400    Seek to the last [granule marked] page preceding the specified pos
01401    location, such that decoding past the returned point will quickly
01402    arrive at the requested position. */
01403 int ov_pcm_seek_page(OggVorbis_File *vf,ogg_int64_t pos){
01404   int link=-1;
01405   ogg_int64_t result=0;
01406   ogg_int64_t total=ov_pcm_total(vf,-1);
01407 
01408   if(vf->ready_state<OPENED)return(OV_EINVAL);
01409   if(!vf->seekable)return(OV_ENOSEEK);
01410 
01411   if(pos<0 || pos>total)return(OV_EINVAL);
01412 
01413   /* which bitstream section does this pcm offset occur in? */
01414   for(link=vf->links-1;link>=0;link--){
01415     total-=vf->pcmlengths[link*2+1];
01416     if(pos>=total)break;
01417   }
01418 
01419   /* search within the logical bitstream for the page with the highest
01420      pcm_pos preceding (or equal to) pos.  There is a danger here;
01421      missing pages or incorrect frame number information in the
01422      bitstream could make our task impossible.  Account for that (it
01423      would be an error condition) */
01424 
01425   /* new search algorithm by HB (Nicholas Vinen) */
01426   {
01427     ogg_int64_t end=vf->offsets[link+1];
01428     ogg_int64_t begin=vf->offsets[link];
01429     ogg_int64_t begintime = vf->pcmlengths[link*2];
01430     ogg_int64_t endtime = vf->pcmlengths[link*2+1]+begintime;
01431     ogg_int64_t target=pos-total+begintime;
01432     ogg_int64_t best=begin;
01433 
01434     ogg_page og;
01435     while(begin<end){
01436       ogg_int64_t bisect;
01437 
01438       if(end-begin<CHUNKSIZE){
01439         bisect=begin;
01440       }else{
01441         /* take a (pretty decent) guess. */
01442         bisect=begin +
01443           (ogg_int64_t)((double)(target-begintime)*(end-begin)/(endtime-begintime))
01444           - CHUNKSIZE;
01445         if(bisect<begin+CHUNKSIZE)
01446           bisect=begin;
01447       }
01448 
01449       if(bisect!=vf->offset){
01450         result=_seek_helper(vf,bisect);
01451         if(result) goto seek_error;
01452       }
01453 
01454       while(begin<end){
01455         result=_get_next_page(vf,&og,end-vf->offset);
01456         if(result==OV_EREAD) goto seek_error;
01457         if(result<0){
01458           if(bisect<=begin+1)
01459             end=begin; /* found it */
01460           else{
01461             if(bisect==0) goto seek_error;
01462             bisect-=CHUNKSIZE;
01463             if(bisect<=begin)bisect=begin+1;
01464             result=_seek_helper(vf,bisect);
01465             if(result) goto seek_error;
01466           }
01467         }else{
01468           ogg_int64_t granulepos;
01469 
01470           if(ogg_page_serialno(&og)!=vf->serialnos[link])
01471             continue;
01472 
01473           granulepos=ogg_page_granulepos(&og);
01474           if(granulepos==-1)continue;
01475 
01476           if(granulepos<target){
01477             best=result;  /* raw offset of packet with granulepos */
01478             begin=vf->offset; /* raw offset of next page */
01479             begintime=granulepos;
01480 
01481             if(target-begintime>44100)break;
01482             bisect=begin; /* *not* begin + 1 */
01483           }else{
01484             if(bisect<=begin+1)
01485               end=begin;  /* found it */
01486             else{
01487               if(end==vf->offset){ /* we're pretty close - we'd be stuck in */
01488                 end=result;
01489                 bisect-=CHUNKSIZE; /* an endless loop otherwise. */
01490                 if(bisect<=begin)bisect=begin+1;
01491                 result=_seek_helper(vf,bisect);
01492                 if(result) goto seek_error;
01493               }else{
01494                 end=bisect;
01495                 endtime=granulepos;
01496                 break;
01497               }
01498             }
01499           }
01500         }
01501       }
01502     }
01503 
01504     /* found our page. seek to it, update pcm offset. Easier case than
01505        raw_seek, don't keep packets preceding granulepos. */
01506     {
01507       ogg_page og;
01508       ogg_packet op;
01509 
01510       /* seek */
01511       result=_seek_helper(vf,best);
01512       vf->pcm_offset=-1;
01513       if(result) goto seek_error;
01514       result=_get_next_page(vf,&og,-1);
01515       if(result<0) goto seek_error;
01516 
01517       if(link!=vf->current_link){
01518         /* Different link; dump entire decode machine */
01519         _decode_clear(vf);
01520 
01521         vf->current_link=link;
01522         vf->current_serialno=vf->serialnos[link];
01523         vf->ready_state=STREAMSET;
01524 
01525       }else{
01526         vorbis_synthesis_restart(&vf->vd);
01527       }
01528 
01529       ogg_stream_reset_serialno(&vf->os,vf->current_serialno);
01530       ogg_stream_pagein(&vf->os,&og);
01531 
01532       /* pull out all but last packet; the one with granulepos */
01533       while(1){
01534         result=ogg_stream_packetpeek(&vf->os,&op);
01535         if(result==0){
01536           /* !!! the packet finishing this page originated on a
01537              preceding page. Keep fetching previous pages until we
01538              get one with a granulepos or without the 'continued' flag
01539              set.  Then just use raw_seek for simplicity. */
01540 
01541           result=_seek_helper(vf,best);
01542           if(result<0) goto seek_error;
01543 
01544           while(1){
01545             result=_get_prev_page(vf,&og);
01546             if(result<0) goto seek_error;
01547             if(ogg_page_serialno(&og)==vf->current_serialno &&
01548                (ogg_page_granulepos(&og)>-1 ||
01549                 !ogg_page_continued(&og))){
01550               return ov_raw_seek(vf,result);
01551             }
01552             vf->offset=result;
01553           }
01554         }
01555         if(result<0){
01556           result = OV_EBADPACKET;
01557           goto seek_error;
01558         }
01559         if(op.granulepos!=-1){
01560           vf->pcm_offset=op.granulepos-vf->pcmlengths[vf->current_link*2];
01561           if(vf->pcm_offset<0)vf->pcm_offset=0;
01562           vf->pcm_offset+=total;
01563           break;
01564         }else
01565           result=ogg_stream_packetout(&vf->os,NULL);
01566       }
01567     }
01568   }
01569 
01570   /* verify result */
01571   if(vf->pcm_offset>pos || pos>ov_pcm_total(vf,-1)){
01572     result=OV_EFAULT;
01573     goto seek_error;
01574   }
01575   vf->bittrack=0.f;
01576   vf->samptrack=0.f;
01577   return(0);
01578 
01579  seek_error:
01580   /* dump machine so we're in a known state */
01581   vf->pcm_offset=-1;
01582   _decode_clear(vf);
01583   return (int)result;
01584 }

01586 /* seek to a sample offset relative to the decompressed pcm stream
01587    returns zero on success, nonzero on failure */
01588 
01589 int ov_pcm_seek(OggVorbis_File *vf,ogg_int64_t pos){
01590   int thisblock,lastblock=0;
01591   int ret=ov_pcm_seek_page(vf,pos);
01592   if(ret<0)return(ret);
01593   if((ret=_make_decode_ready(vf)))return ret;
01594 
01595   /* discard leading packets we don't need for the lapping of the
01596      position we want; don't decode them */
01597 
01598   while(1){
01599     ogg_packet op;
01600     ogg_page og;
01601 
01602     int ret=ogg_stream_packetpeek(&vf->os,&op);
01603     if(ret>0){
01604       thisblock=vorbis_packet_blocksize(vf->vi+vf->current_link,&op);
01605       if(thisblock<0){
01606         ogg_stream_packetout(&vf->os,NULL);
01607         continue; /* non audio packet */
01608       }
01609       if(lastblock)vf->pcm_offset+=(lastblock+thisblock)>>2;
01610 
01611       if(vf->pcm_offset+((thisblock+
01612                           vorbis_info_blocksize(vf->vi,1))>>2)>=pos)break;
01613 
01614       /* remove the packet from packet queue and track its granulepos */
01615       ogg_stream_packetout(&vf->os,NULL);
01616       vorbis_synthesis_trackonly(&vf->vb,&op);  /* set up a vb with
01617                                                    only tracking, no
01618                                                    pcm_decode */
01619       vorbis_synthesis_blockin(&vf->vd,&vf->vb);
01620 
01621       /* end of logical stream case is hard, especially with exact
01622          length positioning. */
01623 
01624       if(op.granulepos>-1){
01625         int i;
01626         /* always believe the stream markers */
01627         vf->pcm_offset=op.granulepos-vf->pcmlengths[vf->current_link*2];
01628         if(vf->pcm_offset<0)vf->pcm_offset=0;
01629         for(i=0;i<vf->current_link;i++)
01630           vf->pcm_offset+=vf->pcmlengths[i*2+1];
01631       }
01632 
01633       lastblock=thisblock;
01634 
01635     }else{
01636       if(ret<0 && ret!=OV_HOLE)break;
01637 
01638       /* suck in a new page */
01639       if(_get_next_page(vf,&og,-1)<0)break;
01640       if(ogg_page_bos(&og))_decode_clear(vf);
01641 
01642       if(vf->ready_state<STREAMSET){
01643         long serialno=ogg_page_serialno(&og);
01644         int link;
01645 
01646         for(link=0;link<vf->links;link++)
01647           if(vf->serialnos[link]==serialno)break;
01648         if(link==vf->links) continue;
01649         vf->current_link=link;
01650 
01651         vf->ready_state=STREAMSET;
01652         vf->current_serialno=ogg_page_serialno(&og);
01653         ogg_stream_reset_serialno(&vf->os,serialno);
01654         ret=_make_decode_ready(vf);
01655         if(ret)return ret;
01656         lastblock=0;
01657       }
01658 
01659       ogg_stream_pagein(&vf->os,&og);
01660     }
01661   }
01662 
01663   vf->bittrack=0.f;
01664   vf->samptrack=0.f;
01665   /* discard samples until we reach the desired position. Crossing a
01666      logical bitstream boundary with abandon is OK. */
01667   {
01668     /* note that halfrate could be set differently in each link, but
01669        vorbisfile encoforces all links are set or unset */
01670     int hs=vorbis_synthesis_halfrate_p(vf->vi);
01671     while(vf->pcm_offset<((pos>>hs)<<hs)){
01672       ogg_int64_t target=(pos-vf->pcm_offset)>>hs;
01673       long samples=vorbis_synthesis_pcmout(&vf->vd,NULL);
01674 
01675       if(samples>target)samples=target;
01676       vorbis_synthesis_read(&vf->vd,samples);
01677       vf->pcm_offset+=samples<<hs;
01678 
01679       if(samples<target)
01680         if(_fetch_and_process_packet(vf,NULL,1,1)<=0)
01681           vf->pcm_offset=ov_pcm_total(vf,-1); /* eof */
01682     }
01683   }
01684   return 0;
01685 }

00056    N. Levinson in 1947, modified by J. Durbin in 1959. */

00059    Output: m lpc coefficients, excitation energy */

00061 float vorbis_lpc_from_data(float *data,float *lpci,int n,int m){
00062   double *aut=alloca(sizeof(*aut)*(m+1));
00063   double *lpc=alloca(sizeof(*lpc)*(m));
00064   double error;
00065   double epsilon;
00066   int i,j;
00067 
00068   /* autocorrelation, p+1 lag coefficients */
00069   j=m+1;
00070   while(j--){
00071     double d=0; /* double needed for accumulator depth */
00072     for(i=j;i<n;i++)d+=(double)data[i]*data[i-j];
00073     aut[j]=d;
00074   }
00075 
00076   /* Generate lpc coefficients from autocorr values */
00077 
00078   /* set our noise floor to about -100dB */
00079   error=aut[0] * (1. + 1e-10);
00080   epsilon=1e-9*aut[0]+1e-10;
00081 
00082   for(i=0;i<m;i++){
00083     double r= -aut[i+1];
00084 
00085     if(error<epsilon){
00086       memset(lpc+i,0,(m-i)*sizeof(*lpc));
00087       goto done;
00088     }
00089 
00090     /* Sum up this iteration's reflection coefficient; note that in
00091        Vorbis we don't save it.  If anyone wants to recycle this code
00092        and needs reflection coefficients, save the results of 'r' from
00093        each iteration. */
00094 
00095     for(j=0;j<i;j++)r-=lpc[j]*aut[i-j];
00096     r/=error;
00097 
00098     /* Update LPC coefficients and total error */
00099 
00100     lpc[i]=r;
00101     for(j=0;j<i/2;j++){
00102       double tmp=lpc[j];
00103 
00104       lpc[j]+=r*lpc[i-1-j];
00105       lpc[i-1-j]+=r*tmp;
00106     }
00107     if(i&1)lpc[j]+=lpc[j]*r;
00108 
00109     error*=1.-r*r;
00110 
00111   }
00112 
00113  done:
00114 
00115   /* slightly damp the filter */
00116   {
00117     double g = .99;
00118     double damp = g;
00119     for(j=0;j<m;j++){
00120       lpc[j]*=damp;
00121       damp*=g;
00122     }
00123   }
00124 
00125   for(j=0;j<m;j++)lpci[j]=(float)lpc[j];
00126 
00127   /* we need the error value to know how big an impulse to hit the
00128      filter with later */
00129 
00130   return error;
00131 }

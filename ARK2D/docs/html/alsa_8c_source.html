00936 static ALCuint alsa_available_samples(ALCdevice *Device)
00937 {
00938     alsa_data *data = (alsa_data*)Device->ExtraData;
00939     snd_pcm_sframes_t avail;
00940 
00941     avail = (Device->Connected ? psnd_pcm_avail_update(data->pcmHandle) : 0);
00942     if(avail < 0)
00943     {
00944         AL_PRINT("avail update failed: %s\n", psnd_strerror(avail));
00945 
00946         if((avail=psnd_pcm_recover(data->pcmHandle, avail, 1)) >= 0 &&
00947            (avail=psnd_pcm_prepare(data->pcmHandle)) >= 0)
00948         {
00949             if(data->doCapture)
00950                 avail = psnd_pcm_start(data->pcmHandle);
00951             if(avail >= 0)
00952                 avail = psnd_pcm_avail_update(data->pcmHandle);
00953         }
00954         if(avail < 0)
00955         {
00956             AL_PRINT("restore error: %s\n", psnd_strerror(avail));
00957             aluHandleDisconnect(Device);
00958         }
00959     }
00960     while(avail > 0)
00961     {
00962         snd_pcm_sframes_t amt;
00963 
00964         amt = psnd_pcm_bytes_to_frames(data->pcmHandle, data->size);
00965         if(avail < amt) amt = avail;
00966 
00967         amt = psnd_pcm_readi(data->pcmHandle, data->buffer, amt);
00968         if(amt < 0)
00969         {
00970             AL_PRINT("read error: %s\n", psnd_strerror(amt));
00971 
00972             if(amt == -EAGAIN)
00973                 continue;
00974             if((amt=psnd_pcm_recover(data->pcmHandle, amt, 1)) >= 0 &&
00975                (amt=psnd_pcm_prepare(data->pcmHandle)) >= 0)
00976             {
00977                 if(data->doCapture)
00978                     amt = psnd_pcm_start(data->pcmHandle);
00979                 if(amt >= 0)
00980                     amt = psnd_pcm_avail_update(data->pcmHandle);
00981             }
00982             if(amt < 0)
00983             {
00984                 AL_PRINT("restore error: %s\n", psnd_strerror(amt));
00985                 aluHandleDisconnect(Device);
00986                 break;
00987             }
00988             avail = amt;
00989             continue;
00990         }
00991 
00992         WriteRingBuffer(data->ring, data->buffer, amt);
00993         avail -= amt;
00994     }
00995 
00996     return RingBufferSize(data->ring);
00997 }

00072   /* A simple technical note on how the raster works                       */
00073   /* -----------------------------------------------                       */
00074   /*                                                                       */
00075   /*   Converting an outline into a bitmap is achieved in several steps:   */
00076   /*                                                                       */
00077   /*   1 - Decomposing the outline into successive `profiles'.  Each       */
00078   /*       profile is simply an array of scanline intersections on a given */
00079   /*       dimension.  A profile's main attributes are                     */
00080   /*                                                                       */
00081   /*       o its scanline position boundaries, i.e. `Ymin' and `Ymax'      */
00082   /*                                                                       */
00083   /*       o an array of intersection coordinates for each scanline        */
00084   /*         between `Ymin' and `Ymax'                                     */
00085   /*                                                                       */
00086   /*       o a direction, indicating whether it was built going `up' or    */
00087   /*         `down', as this is very important for filling rules           */
00088   /*                                                                       */
00089   /*       o its drop-out mode                                             */
00090   /*                                                                       */
00091   /*   2 - Sweeping the target map's scanlines in order to compute segment */
00092   /*       `spans' which are then filled.  Additionally, this pass         */
00093   /*       performs drop-out control.                                      */
00094   /*                                                                       */
00095   /*   The outline data is parsed during step 1 only.  The profiles are    */
00096   /*   built from the bottom of the render pool, used as a stack.  The     */
00097   /*   following graphics shows the profile list under construction:       */
00098   /*                                                                       */
00099   /*     __________________________________________________________ _ _    */
00100   /*    |         |                 |         |                 |          */
00101   /*    | profile | coordinates for | profile | coordinates for |-->       */
00102   /*    |    1    |  profile 1      |    2    |  profile 2      |-->       */
00103   /*    |_________|_________________|_________|_________________|__ _ _    */
00104   /*                                                                       */
00105   /*    ^                                                       ^          */
00106   /*    |                                                       |          */
00107   /* start of render pool                                      top         */
00108   /*                                                                       */
00109   /*   The top of the profile stack is kept in the `top' variable.         */
00110   /*                                                                       */
00111   /*   As you can see, a profile record is pushed on top of the render     */
00112   /*   pool, which is then followed by its coordinates/intersections.  If  */
00113   /*   a change of direction is detected in the outline, a new profile is  */
00114   /*   generated until the end of the outline.                             */
00115   /*                                                                       */
00116   /*   Note that when all profiles have been generated, the function       */
00117   /*   Finalize_Profile_Table() is used to record, for each profile, its   */
00118   /*   bottom-most scanline as well as the scanline above its upmost       */
00119   /*   boundary.  These positions are called `y-turns' because they (sort  */
00120   /*   of) correspond to local extrema.  They are stored in a sorted list  */
00121   /*   built from the top of the render pool as a downwards stack:         */
00122   /*                                                                       */
00123   /*      _ _ _______________________________________                      */
00124   /*                            |                    |                     */
00125   /*                         <--| sorted list of     |                     */
00126   /*                         <--|  extrema scanlines |                     */
00127   /*      _ _ __________________|____________________|                     */
00128   /*                                                                       */
00129   /*                            ^                    ^                     */
00130   /*                            |                    |                     */
00131   /*                         maxBuff           sizeBuff = end of pool      */
00132   /*                                                                       */
00133   /*   This list is later used during the sweep phase in order to          */
00134   /*   optimize performance (see technical note on the sweep below).       */
00135   /*                                                                       */
00136   /*   Of course, the raster detects whether the two stacks collide and    */
00137   /*   handles the situation properly.                                     */
00138   /*                                                                       */

01754   /*    Beziers by calling Line_To() and Bezier_To().  It handles all      */

01760   /*                                                                       */

01762   /*                                                                       */

01764   /*                                                                       */
01765   /* <Return>                                                              */
01766   /*    SUCCESS on success, FAILURE on error.                              */
01767   /*                                                                       */
01768   static Bool
01769   Decompose_Curve( RAS_ARGS UShort  first,
01770                             UShort  last,

01774     FT_Vector   v_control;

01776 
01777     FT_Vector*  points;
01778     FT_Vector*  point;
01779     FT_Vector*  limit;
01780     char*       tags;
01781 
01782     unsigned    tag;       /* current point's state           */
01783 
01784 
01785     points = ras.outline.points;
01786     limit  = points + last;
01787

01790     v_last.x  = SCALED( points[last].x );
01791     v_last.y  = SCALED( points[last].y );

01794     {
01795       SWAP_( v_start.x, v_start.y );
01796       SWAP_( v_last.x, v_last.y );

01800

01802     tags  = ras.outline.tags + first;

01806       ras.dropOutControl = (Byte)tags[0] >> 5;

01809

01811     if ( tag == FT_CURVE_TAG_CUBIC )

01815     if ( tag == FT_CURVE_TAG_CONIC )

01821         v_start = v_last;
01822         limit--;

01827         /* start at their middle and record its position    */
01828         /* for closure                                      */

01830         v_start.y = ( v_start.y + v_last.y ) / 2;

01875           FT_Vector  v_middle;
01876           Long       x, y;

01880           tags++;
01881           tag = FT_CURVE_TAG( tags[0] );

01884           y = SCALED( point[0].y );

01900           v_middle.y = ( v_control.y + y ) / 2;

01903                                   v_middle.x,  v_middle.y ) )
01904             goto Fail;
01905 
01906           v_control.x = x;
01907           v_control.y = y;
01908 
01909           goto Do_Conic;
01910         }
01911 
01912         if ( Conic_To( RAS_VARS v_control.x, v_control.y,
01913                                 v_start.x,   v_start.y ) )
01914           goto Fail;
01915

01917 
01918       default:  /* FT_CURVE_TAG_CUBIC */
01919         {
01920           Long  x1, y1, x2, y2, x3, y3;
01921 
01922

01924                FT_CURVE_TAG( tags[1] ) != FT_CURVE_TAG_CUBIC )
01925             goto Invalid_Outline;
01926 
01927           point += 2;
01928           tags  += 2;
01929

01931           y1 = SCALED( point[-2].y );
01932           x2 = SCALED( point[-1].x );
01933           y2 = SCALED( point[-1].y );
01934

01936           {
01937             SWAP_( x1, y1 );
01938             SWAP_( x2, y2 );
01939           }
01940 
01941           if ( point <= limit )
01942           {
01943             x3 = SCALED( point[0].x );
01944             y3 = SCALED( point[0].y );
01945 
01946             if ( flipped )
01947               SWAP_( x3, y3 );
01948 
01949             if ( Cubic_To( RAS_VARS x1, y1, x2, y2, x3, y3 ) )
01950               goto Fail;
01951             continue;
01952           }
01953 
01954           if ( Cubic_To( RAS_VARS x1, y1, x2, y2, v_start.x, v_start.y ) )

02303                                 FT_F26Dot6  x1,
02304                                 FT_F26Dot6  x2,
02305                                 PProfile    left,
02306                                 PProfile    right )
02307   {
02308     Long   e1, e2, pxl;
02309     Short  c1, f1;
02310 
02311 
02312     /* Drop-out control */
02313 
02314     /*   e2            x2                    x1           e1   */
02315     /*                                                         */
02316     /*                 ^                     |                 */
02317     /*                 |                     |                 */
02318     /*   +-------------+---------------------+------------+    */
02319     /*                 |                     |                 */
02320     /*                 |                     v                 */
02321     /*                                                         */
02322     /* pixel         contour              contour       pixel  */
02323     /* center                                           center */
02324 
02325     /* drop-out mode    scan conversion rules (as defined in OpenType) */
02326     /* --------------------------------------------------------------- */
02327     /*  0                1, 2, 3                                       */
02328     /*  1                1, 2, 4                                       */
02329     /*  2                1, 2                                          */
02330     /*  3                same as mode 2                                */
02331     /*  4                1, 2, 5                                       */
02332     /*  5                1, 2, 6                                       */
02333     /*  6, 7             same as mode 2                                */
02334 
02335     e1  = CEILING( x1 );
02336     e2  = FLOOR  ( x2 );
02337     pxl = e1;
02338 
02339     if ( e1 > e2 )
02340     {
02341       Int  dropOutControl = left->flags & 7;
02342 
02343 
02344       if ( e1 == e2 + ras.precision )
02345       {
02346         switch ( dropOutControl )
02347         {
02348         case 0: /* simple drop-outs including stubs */
02349           pxl = e2;
02350           break;
02351 
02352         case 4: /* smart drop-outs including stubs */
02353           pxl = FLOOR( ( x1 + x2 - 1 ) / 2 + ras.precision_half );
02354           break;
02355 
02356         case 1: /* simple drop-outs excluding stubs */
02357         case 5: /* smart drop-outs excluding stubs  */
02358 
02359           /* Drop-out Control Rules #4 and #6 */
02360 
02361           /* The specification neither provides an exact definition */
02362           /* of a `stub' nor gives exact rules to exclude them.     */
02363           /*                                                        */
02364           /* Here the constraints we use to recognize a stub.       */
02365           /*                                                        */
02366           /*  upper stub:                                           */
02367           /*                                                        */
02368           /*   - P_Left and P_Right are in the same contour         */
02369           /*   - P_Right is the successor of P_Left in that contour */
02370           /*   - y is the top of P_Left and P_Right                 */
02371           /*                                                        */
02372           /*  lower stub:                                           */
02373           /*                                                        */
02374           /*   - P_Left and P_Right are in the same contour         */
02375           /*   - P_Left is the successor of P_Right in that contour */
02376           /*   - y is the bottom of P_Left                          */
02377           /*                                                        */
02378           /* We draw a stub if the following constraints are met.   */
02379           /*                                                        */
02380           /*   - for an upper or lower stub, there is top or bottom */
02381           /*     overshoot, respectively                            */
02382           /*   - the covered interval is greater or equal to a half */
02383           /*     pixel                                              */
02384 
02385           /* upper stub test */
02386           if ( left->next == right                &&
02387                left->height <= 0                  &&
02388                !( left->flags & Overshoot_Top   &&
02389                   x2 - x1 >= ras.precision_half ) )
02390             return;
02391 
02392           /* lower stub test */
02393           if ( right->next == left                 &&
02394                left->start == y                    &&
02395                !( left->flags & Overshoot_Bottom &&
02396                   x2 - x1 >= ras.precision_half  ) )
02397             return;
02398 
02399           if ( dropOutControl == 1 )
02400             pxl = e2;
02401           else
02402             pxl = FLOOR( ( x1 + x2 - 1 ) / 2 + ras.precision_half );
02403           break;
02404 
02405         default: /* modes 2, 3, 6, 7 */
02406           return;  /* no drop-out control */
02407         }
02408 
02409         /* check that the other pixel isn't set */
02410         e1 = pxl == e1 ? e2 : e1;
02411 
02412         e1 = TRUNC( e1 );
02413 
02414         c1 = (Short)( e1 >> 3 );
02415         f1 = (Short)( e1 &  7 );
02416 
02417         if ( e1 >= 0 && e1 < ras.bWidth                      &&
02418              ras.bTarget[ras.traceOfs + c1] & ( 0x80 >> f1 ) )
02419           return;
02420       }
02421       else
02422         return;
02423     }
02424 
02425     e1 = TRUNC( pxl );
02426 
02427     if ( e1 >= 0 && e1 < ras.bWidth )
02428     {
02429       c1 = (Short)( e1 >> 3 );
02430       f1 = (Short)( e1 & 7 );
02431 
02432       if ( ras.gray_min_x > c1 )
02433         ras.gray_min_x = c1;
02434       if ( ras.gray_max_x < c1 )
02435         ras.gray_max_x = c1;
02436 
02437       ras.bTarget[ras.traceOfs + c1] |= (char)( 0x80 >> f1 );
02438     }
02439   }

02700         if ( ras.gray_max_x >= last_cell && last_bit != 3 )
02701         {
02702           ras.gray_max_x = last_cell - 1;
02703           over = 1;
02704         }
02705 
02706         if ( ras.gray_min_x < 0 )
02707           ras.gray_min_x = 0;
02708 
02709         bit  = ras.bTarget + ras.gray_min_x;
02710         bit2 = bit + ras.gray_width;
02711 
02712         c1 = ras.gray_max_x - ras.gray_min_x;
02713 
02714         while ( c1 >= 0 )
02715         {
02716           c2 = count[*bit] + count[*bit2];
02717 
02718           if ( c2 )
02719           {
02720             pix[0] = grays[(c2 >> 12) & 0x000F];
02721             pix[1] = grays[(c2 >> 8 ) & 0x000F];
02722             pix[2] = grays[(c2 >> 4 ) & 0x000F];
02723             pix[3] = grays[ c2        & 0x000F];
02724 
02725             *bit  = 0;
02726             *bit2 = 0;
02727           }
02728 
02729           bit++;
02730           bit2++;
02731           pix += 4;
02732           c1--;
02733         }
02734 
02735         if ( over )
02736         {
02737           c2 = count[*bit] + count[*bit2];
02738           if ( c2 )
02739           {
02740             switch ( last_bit )
02741             {
02742             case 2:
02743               pix[2] = grays[(c2 >> 4 ) & 0x000F];
02744             case 1:
02745               pix[1] = grays[(c2 >> 8 ) & 0x000F];
02746             default:
02747               pix[0] = grays[(c2 >> 12) & 0x000F];
02748             }
02749 
02750             *bit  = 0;
02751             *bit2 = 0;
02752           }
02753         }
02754       }
02755 
02756       ras.traceOfs = 0;
02757       ras.traceG  += ras.traceIncr;
02758 
02759       ras.gray_min_x =  32000;
02760       ras.gray_max_x = -32000;
02761     }

02782   static void
02783   Horizontal_Gray_Sweep_Drop( RAS_ARGS Short       y,
02784                                        FT_F26Dot6  x1,
02785                                        FT_F26Dot6  x2,
02786                                        PProfile    left,
02787                                        PProfile    right )
02788   {
02789     Long   e1, e2;
02790     PByte  pixel;
02791     Byte   color;
02792 
02793 
02794     /* During the horizontal sweep, we only take care of drop-outs */
02795 
02796     e1 = CEILING( x1 );
02797     e2 = FLOOR  ( x2 );
02798 
02799     if ( e1 > e2 )
02800     {
02801       Int  dropOutControl = left->flags & 7;
02802 
02803 
02804       if ( e1 == e2 + ras.precision )
02805       {
02806         switch ( dropOutControl )
02807         {
02808         case 0: /* simple drop-outs including stubs */
02809           e1 = e2;
02810           break;
02811 
02812         case 4: /* smart drop-outs including stubs */
02813           e1 = FLOOR( ( x1 + x2 - 1 ) / 2 + ras.precision_half );
02814           break;
02815 
02816         case 1: /* simple drop-outs excluding stubs */
02817         case 5: /* smart drop-outs excluding stubs  */
02818           /* see Vertical_Sweep_Drop for details */
02819 
02820           /* rightmost stub test */
02821           if ( left->next == right && left->height <= 0 )
02822             return;
02823 
02824           /* leftmost stub test */
02825           if ( right->next == left && left->start == y )
02826             return;
02827 
02828           if ( dropOutControl == 1 )
02829             e1 = e2;
02830           else
02831             e1 = FLOOR( ( x1 + x2 - 1 ) / 2 + ras.precision_half );
02832 
02833           break;
02834 
02835         default: /* modes 2, 3, 6, 7 */
02836           return;  /* no drop-out control */
02837         }
02838       }
02839       else
02840         return;
02841     }
02842 
02843     if ( e1 >= 0 )
02844     {
02845       if ( x2 - x1 >= ras.precision_half )
02846         color = ras.grays[2];
02847       else
02848         color = ras.grays[1];
02849 
02850       e1 = TRUNC( e1 ) / 2;
02851       if ( e1 < ras.target.rows )
02852       {
02853         pixel = ras.gTarget - e1 * ras.target.pitch + y / 2;
02854         if ( ras.target.pitch > 0 )
02855           pixel += ( ras.target.rows - 1 ) * ras.target.pitch;
02856 
02857         if ( pixel[0] == ras.grays[0] )
02858           pixel[0] = color;
02859       }
02860     }

02868   /*                                                                       */
02869   /*  Generic Sweep Drawing routine                                        */
02870   /*                                                                       */
02871   /*************************************************************************/
02872 
02873   static Bool
02874   Draw_Sweep( RAS_ARG )
02875   {
02876     Short         y, y_change, y_height;
02877 
02878     PProfile      P, Q, P_Left, P_Right;
02879 
02880     Short         min_Y, max_Y, top, bottom, dropouts;
02881 
02882     Long          x1, x2, xs, e1, e2;
02883 
02884     TProfileList  waiting;
02885     TProfileList  draw_left, draw_right;
02886 
02887 
02888     /* initialize empty linked lists */
02889 
02890     Init_Linked( &waiting );
02891 
02892     Init_Linked( &draw_left  );
02893     Init_Linked( &draw_right );
02894 
02895     /* first, compute min and max Y */
02896 
02897     P     = ras.fProfile;
02898     max_Y = (Short)TRUNC( ras.minY );
02899     min_Y = (Short)TRUNC( ras.maxY );
02900 
02901     while ( P )
02902     {
02903       Q = P->link;
02904 
02905       bottom = (Short)P->start;
02906       top    = (Short)( P->start + P->height - 1 );
02907 
02908       if ( min_Y > bottom )
02909         min_Y = bottom;
02910       if ( max_Y < top )
02911         max_Y = top;
02912 
02913       P->X = 0;
02914       InsNew( &waiting, P );
02915 
02916       P = Q;
02917     }
02918 
02919     /* check the Y-turns */
02920     if ( ras.numTurns == 0 )
02921     {
02922       ras.error = Raster_Err_Invalid;
02923       return FAILURE;
02924     }
02925 
02926     /* now initialize the sweep */
02927 
02928     ras.Proc_Sweep_Init( RAS_VARS &min_Y, &max_Y );
02929 
02930     /* then compute the distance of each profile from min_Y */
02931 
02932     P = waiting;
02933 
02934     while ( P )
02935     {
02936       P->countL = (UShort)( P->start - min_Y );
02937       P = P->link;
02938     }
02939 
02940     /* let's go */
02941 
02942     y        = min_Y;
02943     y_height = 0;
02944 
02945     if ( ras.numTurns > 0                     &&
02946          ras.sizeBuff[-ras.numTurns] == min_Y )
02947       ras.numTurns--;
02948 
02949     while ( ras.numTurns > 0 )
02950     {
02951       /* check waiting list for new activations */
02952 
02953       P = waiting;
02954 
02955       while ( P )
02956       {
02957         Q = P->link;
02958         P->countL -= y_height;
02959         if ( P->countL == 0 )
02960         {
02961           DelOld( &waiting, P );
02962 
02963           if ( P->flags & Flow_Up )
02964             InsNew( &draw_left,  P );
02965           else
02966             InsNew( &draw_right, P );
02967         }
02968 
02969         P = Q;
02970       }
02971 
02972       /* sort the drawing lists */
02973 
02974       Sort( &draw_left );
02975       Sort( &draw_right );
02976 
02977       y_change = (Short)ras.sizeBuff[-ras.numTurns--];
02978       y_height = (Short)( y_change - y );
02979 
02980       while ( y < y_change )
02981       {
02982         /* let's trace */
02983 
02984         dropouts = 0;
02985 
02986         P_Left  = draw_left;
02987         P_Right = draw_right;
02988 
02989         while ( P_Left )
02990         {
02991           x1 = P_Left ->X;
02992           x2 = P_Right->X;
02993 
02994           if ( x1 > x2 )
02995           {
02996             xs = x1;
02997             x1 = x2;
02998             x2 = xs;
02999           }
03000 
03001           e1 = FLOOR( x1 );
03002           e2 = CEILING( x2 );
03003 
03004           if ( x2 - x1 <= ras.precision &&
03005                e1 != x1 && e2 != x2     )
03006           {
03007             if ( e1 > e2 || e2 == e1 + ras.precision )
03008             {
03009               Int  dropOutControl = P_Left->flags & 7;
03010 
03011 
03012               if ( dropOutControl != 2 )
03013               {
03014                 /* a drop-out was detected */
03015 
03016                 P_Left ->X = x1;
03017                 P_Right->X = x2;
03018 
03019                 /* mark profile for drop-out processing */
03020                 P_Left->countL = 1;
03021                 dropouts++;
03022               }
03023 
03024               goto Skip_To_Next;
03025             }
03026           }
03027 
03028           ras.Proc_Sweep_Span( RAS_VARS y, x1, x2, P_Left, P_Right );
03029 
03030         Skip_To_Next:
03031 
03032           P_Left  = P_Left->link;
03033           P_Right = P_Right->link;
03034         }
03035 
03036         /* handle drop-outs _after_ the span drawing --       */
03037         /* drop-out processing has been moved out of the loop */
03038         /* for performance tuning                             */
03039         if ( dropouts > 0 )
03040           goto Scan_DropOuts;
03041 
03042       Next_Line:
03043 
03044         ras.Proc_Sweep_Step( RAS_VAR );
03045 
03046         y++;
03047 
03048         if ( y < y_change )
03049         {
03050           Sort( &draw_left  );
03051           Sort( &draw_right );
03052         }
03053       }
03054 
03055       /* now finalize the profiles that need it */
03056 
03057       P = draw_left;
03058       while ( P )
03059       {
03060         Q = P->link;
03061         if ( P->height == 0 )
03062           DelOld( &draw_left, P );
03063         P = Q;
03064       }
03065 
03066       P = draw_right;
03067       while ( P )
03068       {
03069         Q = P->link;
03070         if ( P->height == 0 )
03071           DelOld( &draw_right, P );
03072         P = Q;
03073       }
03074     }
03075 
03076     /* for gray-scaling, flush the bitmap scanline cache */
03077     while ( y <= max_Y )
03078     {
03079       ras.Proc_Sweep_Step( RAS_VAR );
03080       y++;
03081     }
03082 
03083     return SUCCESS;
03084 
03085   Scan_DropOuts:
03086 
03087     P_Left  = draw_left;
03088     P_Right = draw_right;
03089 
03090     while ( P_Left )
03091     {
03092       if ( P_Left->countL )
03093       {
03094         P_Left->countL = 0;
03095 #if 0
03096         dropouts--;  /* -- this is useful when debugging only */
03097 #endif
03098         ras.Proc_Sweep_Drop( RAS_VARS y,
03099                                       P_Left->X,
03100                                       P_Right->X,
03101                                       P_Left,
03102                                       P_Right );
03103       }
03104 
03105       P_Left  = P_Left->link;
03106       P_Right = P_Right->link;
03107     }
03108 
03109     goto Next_Line;
03110   }

03113   /*************************************************************************/
03114   /*                                                                       */
03115   /* <Function>                                                            */
03116   /*    Render_Single_Pass                                                 */
03117   /*                                                                       */
03118   /* <Description>                                                         */
03119   /*    Perform one sweep with sub-banding.                                */
03120   /*                                                                       */
03121   /* <Input>                                                               */
03122   /*    flipped :: If set, flip the direction of the outline.              */
03123   /*                                                                       */
03124   /* <Return>                                                              */
03125   /*    Renderer error code.                                               */
03126   /*                                                                       */
03127   static int
03128   Render_Single_Pass( RAS_ARGS Bool  flipped )
03129   {
03130     Short  i, j, k;
03131 
03132 
03133     while ( ras.band_top >= 0 )
03134     {
03135       ras.maxY = (Long)ras.band_stack[ras.band_top].y_max * ras.precision;
03136       ras.minY = (Long)ras.band_stack[ras.band_top].y_min * ras.precision;
03137 
03138       ras.top = ras.buff;
03139 
03140       ras.error = Raster_Err_None;
03141 
03142       if ( Convert_Glyph( RAS_VARS flipped ) )
03143       {
03144         if ( ras.error != Raster_Err_Overflow )
03145           return FAILURE;
03146 
03147         ras.error = Raster_Err_None;
03148 
03149         /* sub-banding */
03150 
03151 #ifdef DEBUG_RASTER
03152         ClearBand( RAS_VARS TRUNC( ras.minY ), TRUNC( ras.maxY ) );
03153 #endif
03154 
03155         i = ras.band_stack[ras.band_top].y_min;
03156         j = ras.band_stack[ras.band_top].y_max;
03157 
03158         k = (Short)( ( i + j ) / 2 );
03159 
03160         if ( ras.band_top >= 7 || k < i )
03161         {
03162           ras.band_top = 0;
03163           ras.error    = Raster_Err_Invalid;
03164 
03165           return ras.error;
03166         }
03167 
03168         ras.band_stack[ras.band_top + 1].y_min = k;
03169         ras.band_stack[ras.band_top + 1].y_max = j;
03170 
03171         ras.band_stack[ras.band_top].y_max = (Short)( k - 1 );
03172 
03173         ras.band_top++;
03174       }
03175       else
03176       {
03177         if ( ras.fProfile )
03178           if ( Draw_Sweep( RAS_VAR ) )
03179              return ras.error;
03180         ras.band_top--;
03181       }
03182     }
03183 
03184     return SUCCESS;
03185   }

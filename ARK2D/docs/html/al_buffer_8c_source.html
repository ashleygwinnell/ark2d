01053 /*
01054  * LoadData
01055  *
01056  * Loads the specified data into the buffer, using the specified formats.
01057  * Currently, the new format must be 32-bit float, and must have the same
01058  * channel configuration as the original format. This does NOT handle
01059  * compressed formats (eg. IMA4).
01060  */
01061 static ALenum LoadData(ALbuffer *ALBuf, const ALvoid *data, ALsizei size, ALuint freq, ALenum OrigFormat, ALenum NewFormat)
01062 {
01063     ALuint NewBytes = aluBytesFromFormat(NewFormat);
01064     ALuint NewChannels = aluChannelsFromFormat(NewFormat);
01065     ALuint OrigBytes = aluBytesFromFormat(OrigFormat);
01066     ALuint OrigChannels = aluChannelsFromFormat(OrigFormat);
01067     ALuint64 newsize, allocsize;
01068     ALvoid *temp;
01069 
01070     assert(NewBytes == 4);
01071     assert(NewChannels == OrigChannels);
01072 
01073     if ((size%(OrigBytes*OrigChannels)) != 0)
01074         return AL_INVALID_VALUE;
01075 
01076     // Allocate extra padding samples
01077     newsize = size / OrigBytes;
01078     allocsize = (BUFFER_PADDING*NewChannels + newsize)*NewBytes;
01079     if(allocsize > INT_MAX)
01080         return AL_OUT_OF_MEMORY;
01081 
01082     temp = realloc(ALBuf->data, allocsize);
01083     if(!temp) return AL_OUT_OF_MEMORY;
01084     ALBuf->data = temp;
01085 
01086     // Samples are converted here
01087     ConvertData(ALBuf->data, data, OrigBytes, newsize);
01088 
01089     ALBuf->format = NewFormat;
01090     ALBuf->eOriginalFormat = OrigFormat;
01091     ALBuf->size = newsize*NewBytes;
01092     ALBuf->frequency = freq;
01093 
01094     ALBuf->LoopStart = 0;
01095     ALBuf->LoopEnd = newsize / NewChannels;
01096 
01097     ALBuf->OriginalSize = size;
01098     ALBuf->OriginalAlign = OrigBytes * OrigChannels;
01099 
01100     return AL_NO_ERROR;
01101 }

01105     ALsizei i;

01109     switch(origBytes)

01189 static void ConvertDataIMA4(ALfloat *dst, const ALvoid *src, ALint chans, ALsizei len)
01190 {
01191     const ALubyte *IMAData;
01192     ALint Sample[2],Index[2];
01193     ALuint IMACode[2];
01194     ALsizei i,j,k,c;
01195 
01196     if(src == NULL)
01197         return;
01198 
01199     IMAData = src;
01200     for(i = 0;i < len;i++)
01201     {
01202         for(c = 0;c < chans;c++)
01203         {
01204             Sample[c]  = *(IMAData++);
01205             Sample[c] |= *(IMAData++) << 8;
01206             Sample[c]  = (Sample[c]^0x8000) - 32768;
01207             Index[c]  = *(IMAData++);
01208             Index[c] |= *(IMAData++) << 8;
01209             Index[c]  = (Index[c]^0x8000) - 32768;
01210 
01211             Index[c] = ((Index[c]<0) ? 0 : Index[c]);
01212             Index[c] = ((Index[c]>88) ? 88 : Index[c]);
01213 
01214             dst[i*65*chans + c] = ((Sample[c] < 0) ? (Sample[c]/32768.0f) : (Sample[c]/32767.0f));
01215         }
01216 
01217         for(j = 1;j < 65;j += 8)
01218         {
01219             for(c = 0;c < chans;c++)
01220             {
01221                 IMACode[c]  = *(IMAData++);
01222                 IMACode[c] |= *(IMAData++) << 8;
01223                 IMACode[c] |= *(IMAData++) << 16;
01224                 IMACode[c] |= *(IMAData++) << 24;
01225             }
01226 
01227             for(k = 0;k < 8;k++)
01228             {
01229                 for(c = 0;c < chans;c++)
01230                 {
01231                     Sample[c] += ((g_IMAStep_size[Index[c]]*g_IMACodeword_4[IMACode[c]&15])/8);
01232                     Index[c] += g_IMAIndex_adjust_4[IMACode[c]&15];
01233 
01234                     if(Sample[c] < -32768) Sample[c] = -32768;
01235                     else if(Sample[c] > 32767) Sample[c] = 32767;
01236 
01237                     if(Index[c]<0) Index[c] = 0;
01238                     else if(Index[c]>88) Index[c] = 88;
01239 
01240                     dst[(i*65+j+k)*chans + c] = ((Sample[c] < 0) ? (Sample[c]/32768.0f) : (Sample[c]/32767.0f));
01241                     IMACode[c] >>= 4;
01242                 }
01243             }
01244         }
01245     }
01246 }

00038 /* 32 bit float (not IEEE; nonnormalized mantissa +
00039    biased exponent) : neeeeeee eeemmmmm mmmmmmmm mmmmmmmm
00040    Why not IEEE?  It's just not that important here. */
00041 
00042 #define VQ_FEXP 10
00043 #define VQ_FMAN 21
00044 #define VQ_FEXP_BIAS 768 /* bias toward values smaller than 1. */

00046 /* doesn't currently guard under/overflow */
00047 long _float32_pack(float val){
00048   int sign=0;
00049   long exp;
00050   long mant;
00051   if(val<0){
00052     sign=0x80000000;
00053     val= -val;
00054   }
00055   exp= floor(log(val)/log(2.f)+.001); //+epsilon
00056   mant=rint(ldexp(val,(VQ_FMAN-1)-exp));
00057   exp=(exp+VQ_FEXP_BIAS)<<VQ_FMAN;
00058 
00059   return(sign|exp|mant);
00060 }

00070 /* given a list of word lengths, generate a list of codewords.  Works
00071    for length ordered or unordered, always assigns the lowest valued
00072    codewords first.  Extended to handle unused entries (length 0) */
00073 ogg_uint32_t *_make_words(long *l,long n,long sparsecount){
00074   long i,j,count=0;
00075   ogg_uint32_t marker[33];
00076   ogg_uint32_t *r=_ogg_malloc((sparsecount?sparsecount:n)*sizeof(*r));
00077   memset(marker,0,sizeof(marker));
00078 
00079   for(i=0;i<n;i++){
00080     long length=l[i];
00081     if(length>0){
00082       ogg_uint32_t entry=marker[length];
00083 
00084       /* when we claim a node for an entry, we also claim the nodes
00085          below it (pruning off the imagined tree that may have dangled
00086          from it) as well as blocking the use of any nodes directly
00087          above for leaves */
00088 
00089       /* update ourself */
00090       if(length<32 && (entry>>length)){
00091         /* error condition; the lengths must specify an overpopulated tree */
00092         _ogg_free(r);
00093         return(NULL);
00094       }
00095       r[count++]=entry;
00096 
00097       /* Look to see if the next shorter marker points to the node
00098          above. if so, update it and repeat.  */
00099       {
00100         for(j=length;j>0;j--){
00101 
00102           if(marker[j]&1){
00103             /* have to jump branches */
00104             if(j==1)
00105               marker[1]++;
00106             else
00107               marker[j]=marker[j-1]<<1;
00108             break; /* invariant says next upper marker would already
00109                       have been moved if it was on the same path */
00110           }
00111           marker[j]++;
00112         }
00113       }
00114 
00115       /* prune the tree; the implicit invariant says all the longer
00116          markers were dangling from our just-taken node.  Dangle them
00117          from our *new* node. */
00118       for(j=length+1;j<33;j++)
00119         if((marker[j]>>1) == entry){
00120           entry=marker[j];
00121           marker[j]=marker[j-1]<<1;
00122         }else
00123           break;
00124     }else
00125       if(sparsecount==0)count++;
00126   }
00127 
00128   /* sanity check the huffman tree; an underpopulated tree must be
00129      rejected. The only exception is the one-node pseudo-nil tree,
00130      which appears to be underpopulated because the tree doesn't
00131      really exist; there's only one possible 'codeword' or zero bits,
00132      but the above tree-gen code doesn't mark that. */
00133   if(sparsecount != 1){
00134     for(i=1;i<33;i++)
00135       if(marker[i] & (0xffffffffUL>>(32-i))){
00136         _ogg_free(r);
00137         return(NULL);
00138       }
00139   }
00140 
00141   /* bitreverse the words because our bitwise packer/unpacker is LSb
00142      endian */
00143   for(i=0,count=0;i<n;i++){
00144     ogg_uint32_t temp=0;
00145     for(j=0;j<l[i];j++){
00146       temp<<=1;
00147       temp|=(r[count]>>j)&1;
00148     }
00149 
00150     if(sparsecount){
00151       if(l[i])
00152         r[count++]=temp;
00153     }else
00154       r[count++]=temp;
00155   }
00156 
00157   return(r);
00158 }

00163 long _book_maptype1_quantvals(const static_codebook *b){

00165 
00166   /* the above *should* be reliable, but we'll not assume that FP is
00167      ever reliable when bitstream sync is at stake; verify via integer
00168      means that vals really is the greatest value of dim for which
00169      vals^b->bim <= b->entries */
00170   /* treat the above as an initial guess */
00171   while(1){
00172     long acc=1;
00173     long acc1=1;
00174     int i;
00175     for(i=0;i<b->dim;i++){
00176       acc*=vals;
00177       acc1*=vals+1;
00178     }
00179     if(acc<=b->entries && acc1>b->entries){
00180       return(vals);
00181     }else{
00182       if(acc>b->entries){
00183         vals--;
00184       }else{
00185         vals++;
00186       }
00187     }

00192 /* we need to deal with two map types: in map type 1, the values are
00193    generated algorithmically (each column of the vector counts through
00194    the values in the quant vector). in map type 2, all the values came
00195    in in an explicit list.  Both value lists must be unpacked */
00196 float *_book_unquantize(const static_codebook *b,int n,int *sparsemap){

00203

00205        different sizes */

00207     case 1:
00208       /* most of the time, entries%dimensions == 0, but we need to be
00209          well defined.  We define that the possible vales at each
00210          scalar is values == entries/dim.  If entries%dim != 0, we'll
00211          have 'too few' values (values*dim<entries), which means that
00212          we'll have 'left over' entries; left over entries use zeroed
00213          values (and are wasted).  So don't generate codebooks like
00214          that */
00215       quantvals=_book_maptype1_quantvals(b);
00216       for(j=0;j<b->entries;j++){
00217         if((sparsemap && b->lengthlist[j]) || !sparsemap){

00219           int indexdiv=1;
00220           for(k=0;k<b->dim;k++){
00221             int index= (j/indexdiv)%quantvals;
00222             float val=b->quantlist[index];
00223             val=fabs(val)*delta+mindel+last;

00227             else
00228               r[count*b->dim+k]=val;

00230           }
00231           count++;
00232         }
00233 
00234       }
00235       break;
00236     case 2:

00238         if((sparsemap && b->lengthlist[j]) || !sparsemap){
00239           float last=0.f;
00240

107  if (m_frames.size() == 0) { return; }
108  m_timer += delta;
109  while (m_timer > m_frameTime) {
110  m_timer -= m_frameTime;
111 
112  // next frame (+1 or -1)
113  m_currentFrameIndex += m_nextFrameModifier;
114 
115  if (m_pingPong) {
116  if (m_currentFrameIndex >= (signed int) m_frames.size()) {
117  m_currentFrameIndex = m_frames.size() - 1;
118  m_nextFrameModifier *= -1;
119  } else if (m_currentFrameIndex < 0) {
120  m_currentFrameIndex = 1;
121  m_nextFrameModifier *= -1;
122  }
123

129  }

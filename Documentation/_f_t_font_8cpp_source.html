22  void FTFontData::make_dlist(FT_Face face, unsigned char ch, GLuint list_base, GLuint* tex_base) {
23 
24  // get freetype to render out character in to a bitmap.
25  // this requires a couple of freetype commands.
26 
27  // load the glyph for our character.
28  if (FT_Load_Glyph(face, FT_Get_Char_Index(face, ch), FT_LOAD_DEFAULT)) {
29  ErrorDialog::createAndShow("FT_Load_Glyph failed");
30  exit(0);
31  }
32 
33  // move the face's glyph into a glyph object.
34  FT_Glyph glyph;
35  if (FT_Get_Glyph(face->glyph, &glyph)) {
36  ErrorDialog::createAndShow("FT_Get_Glyph failed");
37  exit(0);
38  }
39 
40  // convert the glyph in to a bitmap.
41  FT_Glyph_To_Bitmap(&glyph, ft_render_mode_normal, 0, 1);
42  FT_BitmapGlyph bitmap_glyph = (FT_BitmapGlyph) glyph;
43 
44  // this reference makes referencing the bitmap easier.
45  FT_Bitmap& bitmap = bitmap_glyph->bitmap;
46 
47  // get the width of the bitmap data and pad it the texture.
48  int width = next_p2(bitmap.width);
49  int height = next_p2(bitmap.rows);
50 
51  // allocate memory for texture data.
52  GLubyte* expanded_data = new GLubyte[2 * width * height];
53 
54  // fill the data for the expanded bitmap.
55  for(int j = 0; j < height; j++) {
56  for(int i = 0; i < width; i++) {
57  expanded_data[2 * (i+j*width)] = 255;
58  expanded_data[2 * (i+j*width)+1] =
59  (i >= bitmap.width || j >= bitmap.rows)
60  ?0
61  :bitmap.buffer[i + bitmap.width*j];
62  }
63  }
64 
65  // do some texture shizzle.
66  //glBindTexture(GL_TEXTURE_2D, tex_base[ch]);
67  RendererState::start(RendererState::TEXTURE, tex_base[ch]);
68  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
69  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
70 
71  // create the texture with the data.
72  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, width, height, 0, GL_RGBA, GL_UNSIGNED_BYTE, expanded_data);
73 
74  // we don't need this any more.
75  delete[] expanded_data;
76 
77 
78  // Now We Need To Account For The Fact That Many Of
79  // Our Textures Are Filled With Empty Padding Space.
80  // We Figure What Portion Of The Texture Is Used By
81  // The Actual Character And Store That Information In
82  // The x And y Variables, Then When We Draw The
83  // Quad, We Will Only Reference The Parts Of The Texture
84  // That Contains The Character Itself.
85  float x = (float) bitmap.width / (float) width;
86  float y = (float) bitmap.rows / (float) height;
87 
88 
89  #if (defined(ARK2D_ANDROID) || defined(ARK2D_IPHONE) || defined(ARK2D_MACINTOSH))
90 
91  #else
92  // create the display list...
93  glNewList(list_base+ch, GL_COMPILE);
94  //glBindTexture(GL_TEXTURE_2D, tex_base[ch]);
95  RendererState::start(RendererState::TEXTURE, tex_base[ch]);
96 
97  glPushMatrix();
98 
99  // First We Need To Move Over A Little So That
100  // The Character Has The Right Amount Of Space
101  // Between It And The One Before It.
102  glTranslatef(bitmap_glyph->left, 0, 0);
103 
104  // Now We Move Down A Little In The Case That The
105  // Bitmap Extends Past The Bottom Of The Line
106  // This Is Only True For Characters Like 'g' Or 'y'.
107  glTranslatef(0, bitmap_glyph->top - bitmap.rows, 0);
108  //glTranslatef(0, bitmap_glyph->top, 0);
109 
110 
111  // Here We Draw The Texturemapped Quads.
112  // The Bitmap That We Got From FreeType Was Not
113  // Oriented Quite Like We Would Like It To Be,
114  // But We Link The Texture To The Quad
115  // In Such A Way That The Result Will Be Properly Aligned.
116  glBegin(GL_QUADS);
117  glTexCoord2d(0,0); glVertex2f(0,0);
118  glTexCoord2d(0,y); glVertex2f(0,bitmap.rows);
119  glTexCoord2d(x,y); glVertex2f(bitmap.width,bitmap.rows);
120  glTexCoord2d(x,0); glVertex2f(bitmap.width,0);
121  glEnd();
122 
123  glPopMatrix();
124  glTranslatef(face->glyph->advance.x >> 6 ,0,0);
125 
126  // Finish The Display List
127  glEndList();
128 
129  #endif
130 
131  // scorecharacter data.
132  FTFontCharData* charData = &characterDatas[ch];// + (sizeof(FTFontCharData) * ch_ui);
133  charData->offsetx = bitmap_glyph->left;
134  charData->offsety = bitmap_glyph->top * -1;
135  charData->width = bitmap.width;
136  charData->height = bitmap.rows;
137  charData->advancex = face->glyph->advance.x >> 6; // * 64
138  charData->advancey = face->glyph->advance.y >> 6; // * 64
139  charData->x = x;
140  charData->y = y;
141  charData->tw = x;
142  charData->th = y;
143  charData->tid = tex_base[ch];
144  // Increment The Raster Position As If We Were A Bitmap Font.
145  // (Only Needed If You Want To Calculate Text Length)
146  //glBitmap(0,0,0,0,face->glyph->advance.x >> 6,0,NULL);
147 
148 
149 
150  FT_Done_Glyph(glyph);
151

155  void FTFontData::init(const char* fname, unsigned int h) {
156 
157  // allocate memory to store texture ids
158  textures = new GLuint[128];
159  characterDatas = new FTFontCharData[128];
160 
161  m_height = h;
162 
163  // create & initialise the freetype font library.
164  FT_Library library;
165  if (FT_Init_FreeType(&library)) {
166  ErrorDialog::createAndShow("FT_Init_FreeType failed");
167  exit(0);
168  }
169 
170  // The object in which freetype holds information on a given font is called a "Face".
171  FT_Face face = FT_Face();
172  m_lineHeight = face->height;
173 
174  // Load in the font information from the file.
175  // Code is most likely to die here as FT_New_Face
176  // will fail if the font does not exist or is (somehow) broken.
177  if (FT_New_Face(library, fname, 0, &face)) {
178  ErrorDialog::createAndShow("FT_New_Face failed (there is probably a problem with your font file)");
179  exit(0);
180  }
181 
182  // FreeType measures font size in terms of 1/64ths of pixels.
183  // so to make a font H pixels high, we need to make it h*64 or h << 6.
184  FT_Set_Char_Size(face, h << 6, h << 6, 96, 96);
185 
186  #if (defined(ARK2D_ANDROID) || defined(ARK2D_IPHONE) || defined(ARK2D_MACINTOSH))
187 
188  #else
189 
190  // here we ask opengl to allocate resources for all the
191  // textures and display lists which we are about to create.
192  list_base = glGenLists(128);
193  glGenTextures(128, textures);
194 
195  // this is where we actually create each of the fonts display lists.
196  for(unsigned char i = 0; i < 128; i++) {
197  make_dlist(face, i, list_base, textures);
198  }
199 
200  #endif
201 
202  // we fon#t need the face information now the the display lists have been created.
203  // free the resources.
204  FT_Done_Face(face);
205  FT_Done_FreeType(library);
206  }

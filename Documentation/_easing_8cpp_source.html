126  double Easing::ease(unsigned int easing, double t, double start, double change, double duration)
127  {
128  if (t >= duration) {
129  return start + change;
130  }
131  switch(easing) {
132  case Easing::LINEAR:
133  return change * t / duration + start;
134 
135  case Easing::QUADRATIC_IN:
136  return change * (t /= duration) * t + start;
137 
138  case Easing::QUADRATIC_OUT:
139  return -change * (t /= duration) * (t - 2) + start;
140 
141  case Easing::QUADRATIC_IN_OUT:
142  if ((t /= duration / 2) < 1) {
143  return change / 2 * t * t + start;
144  }
145  return -change / 2 * ((--t) * (t - 2) - 1) + start;
146 
147  case Easing::CUBIC_IN:
148  return change * (t /= duration) * t * t + start;
149 
150  case Easing::CUBIC_OUT:
151  return change * ((t = t / duration - 1) * t * t + 1) + start;
152 
153  case Easing::CUBIC_IN_OUT:
154  if ((t /= duration / 2) < 1) {
155  return change / 2 * t * t * t + start;
156  }
157  return change / 2 * ((t -= 2) * t * t + 2) + start;
158 
159  case Easing::QUARTIC_IN:
160  return change * (t /= duration) * t * t * t + start;
161 
162  case Easing::QUARTIC_OUT:
163  return -change * ((t = t / duration - 1) * t * t * t - 1) + start;
164 
165  case Easing::QUARTIC_IN_OUT:
166  if ((t /= duration / 2) < 1) {
167  return change / 2 * t * t * t * t + start;
168  }
169  return -change / 2 * ((t -= 2) * t * t * t - 2) + start;
170 
171  case Easing::SINE_IN:
172 
173  return -change * (float) cos(t / duration * (MY_PI / 2)) + change + start;
174 
175  case Easing::SINE_OUT:
176  return change * (float) sin(t / duration * (MY_PI / 2)) + start;
177 
178  case Easing::SINE_IN_OUT:
179  return -change / 2 * ((float) cos(MY_PI * t / duration) - 1) + start;
180 
181  case Easing::EXPONENTIAL_IN:
182  return (t==0) ? start : change * (float)pow(2, 10 * (t/duration - 1)) + start;
183 
184  case Easing::EXPONENTIAL_OUT:
185  return (t==duration) ? start+change : change * (-(float)pow(2, -10 * t/duration) + 1) + start;
186 
187  case Easing::EXPONENTIAL_IN_OUT:
188  if (t==0) return start;
189  if (t==duration) return start+change;
190  if ((t/=duration/2) < 1) return change/2 * (float)pow(2, 10 * (t - 1)) + start;
191  return change/2 * (-(float)pow(2, -10 * --t) + 2) + start;
192 
193  case Easing::CIRCLE_IN:
194  return -change * ((float)sqrt(1 - (t/=duration) * t) - 1) + start;
195 
196  case Easing::CIRCLE_OUT:
197  return change * (float)sqrt(1 - (t=t/duration-1) * t) + start;
198 
199  case Easing::CIRCLE_IN_OUT:
200  if ((t/=duration/2) < 1) return -change/2 * ((float)sqrt(1 - t*t) - 1) + start;
201  return change/2 * ((float)sqrt(1 - (t-=2)*t) + 1) + start;
202 
203  case Easing::ELASTIC_IN: {
204  if (t==0) return start;
205  if ((t/=duration)==1) return start+change;
206 
207  double amplitude = 1.0;
208  double period = duration * 0.3; //if (!p) p=duration*.3;
209  double s = 0;
210  if (amplitude < abs(change)) {
211  amplitude = change;
212  s = period / 4;
213  } else {
214  s = period / (2 * MathUtil::PIE) * asin(change/amplitude);
215  }
216  return -(amplitude * pow(2, 10 * (t-=1)) * sin( (t*duration-s)*(2* MathUtil::PIE)/period )) + start;
217  }
218  case Easing::ELASTIC_OUT: {
219  if (t==0) return start;
220  if ((t/=duration)==1) return start+change;
221 
222  double amplitude = 1.0;
223  double period = duration * 0.3; //if (!p) p=duration*.3;
224  double s = 0;
225  if (amplitude < abs(change)) {
226  amplitude = change;
227  s = period/4;
228  } else {
229  s = period/(2 * MathUtil::PIE) * asin (change/amplitude);
230  }
231  return amplitude * pow(2,-10*t) * sin( (t*duration-s)*(2 * MathUtil::PIE)/period ) + change + start;
232  }
233  case Easing::ELASTIC_IN_OUT: {
234  if (t==0) return start;
235  if ((t/=duration/2)==2) return start+change;
236  //if (!p) p=d*(.3*1.5);
237  double amplitude = 1.0;
238  double period = duration*(0.3*1.5);
239  double s = 0;
240 
241  if (amplitude < abs(change)) {
242  amplitude=change;
243  s = period/4;
244  } else {
245  s = period/(2 * MathUtil::PIE) * asin (change/amplitude);
246  }
247 
248  if (t < 1) {
249  return -0.5 * (amplitude * pow(2,10*(t-=1)) * sin( (t*duration-s)*(2*MathUtil::PIE)/period )) + start;
250  }
251  return amplitude * pow(2,-10*(t-=1)) * sin( (t*duration-s)*(2*MathUtil::PIE)/period )*.5 + change + start;
252  }
253  case Easing::BOUNCE_IN:
254  return change - Easing::ease(Easing::BOUNCE_OUT, duration - t, 0, change, duration) + start;
255  case Easing::BOUNCE_OUT:
256  if ((t/=duration) < (1/2.75)) {
257  return change*(7.5625*t*t) + start;
258  } else if (t < (2/2.75)) {
259  return change*(7.5625*(t-=(1.5/2.75))*t + .75) + start;
260  } else if (t < (2.5/2.75)) {
261  return change*(7.5625*(t-=(2.25/2.75))*t + .9375) + start;
262  } else {
263  return change*(7.5625*(t-=(2.625/2.75))*t + .984375) + start;
264  }
265  return start;
266 
267  case Easing::BOUNCE_IN_OUT:
268  if (t < duration/2) { return Easing::ease(Easing::BOUNCE_IN, t*2, 0, change, duration) * 0.5 + start; }
269  return Easing::ease(Easing::BOUNCE_OUT, t*2-duration, 0, change, duration) * .5 + change * 0.5 + start;
270 
271  case Easing::NONE:
272  return start;
273  default:
274  return start;
275  }

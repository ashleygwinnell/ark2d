84  Vector2<float>* p0 = m_curve.at(0);
85  for (unsigned int i = 1; i < m_curve.size(); i++) {
86  Vector2<float>* p1 = m_curve.at(i);
87  if (time >= p0->getX() && time <= p1->getX()) {
88  // found the segment?
89  float st = (time - p0->getX()) / (p1->getX() - p0->getX());
90  float r = p0->getY() + st * (p1->getY() - p0->getY());
91  return r;
92  }

94  }

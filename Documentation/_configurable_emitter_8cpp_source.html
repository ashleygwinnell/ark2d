213  void ConfigurableEmitter::updateParticle(Particle* particle, float delta) {
214  m_particleCount++;
215 
216  //std::cout << "a" << std::endl;
217 
218  particle->m_x += m_adjustx;
219  particle->m_y += m_adjusty;
220 
221  particle->adjustVelocity(
222  m_windFactor.getValue(0) * 0.00005f * (delta*1000),
223  m_gravityFactor.getValue(0) * 0.00005f * (delta*1000));
224 
225  float offset = particle->getLife() / particle->getOriginalLife();
226  float inv = 1 - offset;
227  float colOffset = 0;
228  float colInv = 1;
229 
230  Color* startColor = NULL;
231  Color* endColor = NULL;
232 
233 
234 
235  for (unsigned int i = 0; i < m_colors.size() - 1; i++) {
236  ColorRecord* rec1 = m_colors.at(i);
237  ColorRecord* rec2 = m_colors.at(i + 1);
238 
239  if ((inv >= rec1->pos) && (inv <= rec2->pos)) {
240  startColor = rec1->col;
241  endColor = rec2->col;
242 
243  float step = rec2->pos - rec1->pos;
244  colOffset = inv - rec1->pos;
245  colOffset /= step;
246  colOffset = 1 - colOffset;
247  colInv = 1 - colOffset;
248  }
249  }
250 
251  //std::cout << "b" << std::endl;
252 
253  if (startColor != NULL) {
254  float r = ((startColor->getRed()/255.0f) * colOffset) + ((endColor->getRed()/255.0f) * colInv);
255  float g = ((startColor->getGreen()/255.0f) * colOffset) + ((endColor->getGreen()/255.0f) * colInv);
256  float b = ((startColor->getBlue()/255.0f) * colOffset) + ((endColor->getBlue()/255.0f) * colInv);
257 
258  float a;
259  if (m_alpha->isActive()) {
260  a = m_alpha->getValue(inv) / 255.0f;
261  } else {
262  a = ((m_startAlpha.getValue(0) / 255.0f) * offset)
263  + ((m_endAlpha.getValue(0) / 255.0f) * inv);
264  }
265  particle->setColor(r, g, b, a);
266  }
267 
268  //std::cout << "c" << std::endl;
269 
270  if (m_size->isActive()) {
271  float s = m_size->getValue(inv);
272  particle->setSize(s);
273  } else {
274  particle->adjustSize((delta*1000) * m_growthFactor.getValue(0) * 0.001f);
275  }
276

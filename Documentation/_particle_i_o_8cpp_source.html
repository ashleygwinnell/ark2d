183  void ParticleIO::parseValueElement(TiXmlElement* element, Value* value) {
184  const char* type = element->Attribute("type");
185  const char* v = element->Attribute("value");
186 
187  if (type == NULL || strlen(type) == 0) {
188  // support for old style which did not write the type
189  SimpleValue* simpleValue = NULL;
190  RandomValue* randomValue = NULL;
191  if ((simpleValue = dynamic_cast<SimpleValue*>(value)) != NULL) {
192  simpleValue->setValue(Cast::fromString<float>(v));
193  } else if ((randomValue = dynamic_cast<RandomValue*>(value)) != NULL) {
194  randomValue->setValue(Cast::fromString<float>(v));
195  } else {
196  std::cout << "problems reading element, skipping..." << std::endl;
197  }
198  } else {
199  if (strcmp(type, "simple") == 0) {
200  (dynamic_cast<SimpleValue*>(value))->setValue(Cast::fromString<float>(v));
201  } else if (strcmp(type, "random") == 0) {
202  (dynamic_cast<RandomValue*>(value))->setValue(Cast::fromString<float>(v));
203  } else if (strcmp(type, "linear") == 0) {
204  const char* min = element->Attribute("min");
205  const char* max = element->Attribute("max");
206  const char* active = element->Attribute("active");
207 
208  TiXmlElement* point = 0;
209  vector<Vector2<float>* > curve;
210  for (point = element->FirstChildElement("point");
211  point;
212  point = point->NextSiblingElement("point")) {
213  float x = Cast::fromString<float>(point->Attribute("x"));
214  float y = Cast::fromString<float>(point->Attribute("y"));
215  curve.push_back(new Vector2<float>(x, y));
216  }
217 
218  LinearInterpolator* li = static_cast<LinearInterpolator*>(value);
219  li->setCurve(curve);
220  li->setMin(Cast::fromString<int>(min));
221  li->setMax(Cast::fromString<int>(max));
222  li->setActive(strcmp(active, "true") == 0);
223 
224  } else {
225  std::cout << "unknown type detected: " << type << std::endl;
226  }
227  }

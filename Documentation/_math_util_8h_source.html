152  static void rotateVectorAroundAxis(float* vecx, float* vecy, float* vecz, float axisx, float axisy, float axisz, float degreeAngle) {
153 
154  // Main algorithm
155  double radianAngle = degreeAngle * MY_PI/180.0;
156  double cosAngle = cos(radianAngle);
157  double sinAngle = sin(radianAngle);
158 
159  float crossX = axisy * *vecz - axisz * *vecy;
160  float crossY = axisz * *vecx - axisx * *vecz;
161  float crossZ = axisx * *vecy - axisy * *vecx;
162 
163  *vecx = *vecx * cosAngle;
164  *vecy = *vecy * cosAngle;
165  *vecz = *vecz * cosAngle;
166 
167  *vecx = *vecx + (sinAngle * crossX);
168  *vecy = *vecy + (sinAngle * crossY);
169  *vecz = *vecz + (sinAngle * crossZ);
170 
171  float dotp = ((axisx * *vecx) + (axisy * *vecy) + (axisz * *vecz));
172  float factor = dotp * (1 - cosAngle);
173  *vecx = *vecx + (factor * axisx);
174  *vecy = *vecy + (factor * axisy);
175  *vecz = *vecz + (factor * axisz);
176  }
